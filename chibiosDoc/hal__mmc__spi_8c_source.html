98 static bool mmc_read(void *instance, uint32_t startblk,
99  uint8_t *buffer, uint32_t n) {
100 
101  if (mmcStartSequentialRead((MMCDriver *)instance, startblk)) {
102  return HAL_FAILED;
103  }
104 
105  while (n > 0U) {
106  if (mmcSequentialRead((MMCDriver *)instance, buffer)) {
107  return HAL_FAILED;
108  }
109  buffer += MMCSD_BLOCK_SIZE;
110  n--;
111  }
112 
113  if (mmcStopSequentialRead((MMCDriver *)instance)) {
114  return HAL_FAILED;
115  }
116  return HAL_SUCCESS;
117 }

140 /**
141  * @brief Calculate the MMC standard CRC-7 based on a lookup table.
142  *
143  * @param[in] crc start value for CRC
144  * @param[in] buffer pointer to data buffer
145  * @param[in] len length of data
146  * @return Calculated CRC
147  */
148 static uint8_t crc7(uint8_t crc, const uint8_t *buffer, size_t len) {
149 
150  while (len > 0U) {
151  crc = crc7_lookup_table[(crc << 1) ^ (*buffer++)];
152  len--;
153  }
154  return crc;
155 }

157 /**
158  * @brief Waits an idle condition.
159  *
160  * @param[in] mmcp pointer to the @p MMCDriver object
161  *
162  * @notapi
163  */
164 static void wait(MMCDriver *mmcp) {
165  int i;
166  uint8_t buf[4];
167 
168  for (i = 0; i < 16; i++) {
169  spiReceive(mmcp->config->spip, 1, buf);
170  if (buf[0] == 0xFFU) {
171  return;
172  }
173  }
174  /* Looks like it is a long wait.*/
175  while (true) {
176  spiReceive(mmcp->config->spip, 1, buf);
177  if (buf[0] == 0xFFU) {
178  break;
179  }
180 #if MMC_NICE_WAITING == TRUE
181  /* Trying to be nice with the other threads.*/
182  osalThreadSleepMilliseconds(1);
183 #endif
184  }
185 }

187 /**
188  * @brief Sends a command header.
189  *
190  * @param[in] mmcp pointer to the @p MMCDriver object
191  * @param[in] cmd the command id
192  * @param[in] arg the command argument
193  *
194  * @notapi
195  */
196 static void send_hdr(MMCDriver *mmcp, uint8_t cmd, uint32_t arg) {
197  uint8_t buf[6];
198 
199  /* Wait for the bus to become idle if a write operation was in progress.*/
200  wait(mmcp);
201 
202  buf[0] = (uint8_t)0x40U | cmd;
203  buf[1] = (uint8_t)(arg >> 24U);
204  buf[2] = (uint8_t)(arg >> 16U);
205  buf[3] = (uint8_t)(arg >> 8U);
206  buf[4] = (uint8_t)arg;
207  /* Calculate CRC for command header, shift to right position, add stop bit.*/
208  buf[5] = ((crc7(0, buf, 5U) & 0x7FU) << 1U) | 0x01U;
209 
210  spiSend(mmcp->config->spip, 6, buf);
211 }

213 /**
214  * @brief Receives a single byte response.
215  *
216  * @param[in] mmcp pointer to the @p MMCDriver object
217  * @return The response as an @p uint8_t value.
218  * @retval 0xFF timed out.
219  *
220  * @notapi
221  */
222 static uint8_t recvr1(MMCDriver *mmcp) {
223  int i;
224  uint8_t r1[1];
225 
226  for (i = 0; i < 9; i++) {
227  spiReceive(mmcp->config->spip, 1, r1);
228  if (r1[0] != 0xFFU) {
229  return r1[0];
230  }
231  }
232  return 0xFFU;
233 }

299 /**
300  * @brief Reads the CSD.
301  *
302  * @param[in] mmcp pointer to the @p MMCDriver object
303  * @param[out] cmd command
304  * @param[out] cxd pointer to the CSD/CID buffer
305  *
306  * @return The operation status.
307  * @retval HAL_SUCCESS the operation succeeded.
308  * @retval HAL_FAILED the operation failed.
309  *
310  * @notapi
311  */
312 static bool read_CxD(MMCDriver *mmcp, uint8_t cmd, uint32_t cxd[4]) {
313  unsigned i;
314  uint8_t *bp, buf[16];
315 
316  spiSelect(mmcp->config->spip);
317  send_hdr(mmcp, cmd, 0);
318  if (recvr1(mmcp) != 0x00U) {
319  spiUnselect(mmcp->config->spip);
320  return HAL_FAILED;
321  }
322 
323  /* Wait for data availability.*/
324  for (i = 0U; i < MMC_WAIT_DATA; i++) {
325  spiReceive(mmcp->config->spip, 1, buf);
326  if (buf[0] == 0xFEU) {
327  uint32_t *wp;
328 
329  spiReceive(mmcp->config->spip, 16, buf);
330  bp = buf;
331  for (wp = &cxd[3]; wp >= cxd; wp--) {
332  *wp = ((uint32_t)bp[0] << 24U) | ((uint32_t)bp[1] << 16U) |
333  ((uint32_t)bp[2] << 8U) | (uint32_t)bp[3];
334  bp += 4;
335  }
336 
337  /* CRC ignored then end of transaction. */
338  spiIgnore(mmcp->config->spip, 2);
339  spiUnselect(mmcp->config->spip);
340 
341  return HAL_SUCCESS;
342  }
343  }
344  return HAL_FAILED;
345 }

437 /**
438  * @brief Performs the initialization procedure on the inserted card.
439  * @details This function should be invoked when a card is inserted and
440  * brings the driver in the @p MMC_READY state where it is possible
441  * to perform read and write operations.
442  * @note It is possible to invoke this function from the insertion event
443  * handler.
444  *
445  * @param[in] mmcp pointer to the @p MMCDriver object
446  *
447  * @return The operation status.
448  * @retval HAL_SUCCESS the operation succeeded and the driver is now
449  * in the @p MMC_READY state.
450  * @retval HAL_FAILED the operation failed.
451  *
452  * @api
453  */
454 bool mmcConnect(MMCDriver *mmcp) {
455  unsigned i;
456  uint8_t r3[4];
457 
458  osalDbgCheck(mmcp != NULL);
459 
460  osalDbgAssert((mmcp->state == BLK_ACTIVE) || (mmcp->state == BLK_READY),
461  "invalid state");
462 
463  /* Connection procedure in progress.*/
464  mmcp->state = BLK_CONNECTING;
465  mmcp->block_addresses = false;
466 
467  /* Slow clock mode and 128 clock pulses.*/
468  spiStart(mmcp->config->spip, mmcp->config->lscfg);
469  spiIgnore(mmcp->config->spip, 16);
470 
471  /* SPI mode selection.*/
472  i = 0;
473  while (true) {
474  if (send_command_R1(mmcp, MMCSD_CMD_GO_IDLE_STATE, 0) == 0x01U) {
475  break;
476  }
477  if (++i >= MMC_CMD0_RETRY) {
478  goto failed;
479  }
480  osalThreadSleepMilliseconds(10);
481  }
482 
483  /* Try to detect if this is a high capacity card and switch to block
484  addresses if possible.
485  This method is based on "How to support SDC Ver2 and high capacity cards"
486  by ElmChan.*/
487  if (send_command_R3(mmcp, MMCSD_CMD_SEND_IF_COND,
488  MMCSD_CMD8_PATTERN, r3) != 0x05U) {
489 
490  /* Switch to SDHC mode.*/
491  i = 0;
492  while (true) {
493  /*lint -save -e9007 [13.5] Side effect unimportant.*/
494  if ((send_command_R1(mmcp, MMCSD_CMD_APP_CMD, 0) == 0x01U) &&
495  (send_command_R3(mmcp, MMCSD_CMD_APP_OP_COND, 0x400001AAU, r3) == 0x00U)) {
496  /*lint -restore*/
497  break;
498  }
499 
500  if (++i >= MMC_ACMD41_RETRY) {
501  goto failed;
502  }
503  osalThreadSleepMilliseconds(10);
504  }
505 
506  /* Execute dedicated read on OCR register */
507  (void) send_command_R3(mmcp, MMCSD_CMD_READ_OCR, 0, r3);
508 
509  /* Check if CCS is set in response. Card operates in block mode if set.*/
510  if ((r3[0] & 0x40U) != 0U) {
511  mmcp->block_addresses = true;
512  }
513  }
514 
515  /* Initialization.*/
516  i = 0;
517  while (true) {
518  uint8_t b = send_command_R1(mmcp, MMCSD_CMD_INIT, 0);
519  if (b == 0x00U) {
520  break;
521  }
522  if (b != 0x01U) {
523  goto failed;
524  }
525  if (++i >= MMC_CMD1_RETRY) {
526  goto failed;
527  }
528  osalThreadSleepMilliseconds(10);
529  }
530 
531  /* Initialization complete, full speed.*/
532  spiStart(mmcp->config->spip, mmcp->config->hscfg);
533 
534  /* Setting block size.*/
535  if (send_command_R1(mmcp, MMCSD_CMD_SET_BLOCKLEN,
536  MMCSD_BLOCK_SIZE) != 0x00U) {
537  goto failed;
538  }
539 
540  /* Determine capacity.*/
541  if (read_CxD(mmcp, MMCSD_CMD_SEND_CSD, mmcp->csd)) {
542  goto failed;
543  }
544 
545  mmcp->capacity = _mmcsd_get_capacity(mmcp->csd);
546  if (mmcp->capacity == 0U) {
547  goto failed;
548  }
549 
550  if (read_CxD(mmcp, MMCSD_CMD_SEND_CID, mmcp->cid)) {
551  goto failed;
552  }
553 
554  mmcp->state = BLK_READY;
555  return HAL_SUCCESS;
556 
557  /* Connection failed, state reset to BLK_ACTIVE.*/
558 failed:
559  spiStop(mmcp->config->spip);
560  mmcp->state = BLK_ACTIVE;
561  return HAL_FAILED;
562 }

564 /**
565  * @brief Brings the driver in a state safe for card removal.
566  *
567  * @param[in] mmcp pointer to the @p MMCDriver object
568  * @return The operation status.
569  *
570  * @retval HAL_SUCCESS the operation succeeded and the driver is now

572  * @retval HAL_FAILED the operation failed.
573  *
574  * @api
575  */
576 bool mmcDisconnect(MMCDriver *mmcp) {
577 
578  osalDbgCheck(mmcp != NULL);
579 
580  osalSysLock();
581  osalDbgAssert((mmcp->state == BLK_ACTIVE) || (mmcp->state == BLK_READY),
582  "invalid state");
583  if (mmcp->state == BLK_ACTIVE) {
584  osalSysUnlock();
585  return HAL_SUCCESS;
586  }
587  mmcp->state = BLK_DISCONNECTING;
588  osalSysUnlock();
589 
590  /* Wait for the pending write operations to complete.*/
591  spiStart(mmcp->config->spip, mmcp->config->hscfg);
592  sync(mmcp);
593 
594  spiStop(mmcp->config->spip);
595  mmcp->state = BLK_ACTIVE;
596  return HAL_SUCCESS;
597 }

599 /**
600  * @brief Starts a sequential read.
601  *
602  * @param[in] mmcp pointer to the @p MMCDriver object
603  * @param[in] startblk first block to read
604  *
605  * @return The operation status.
606  * @retval HAL_SUCCESS the operation succeeded.
607  * @retval HAL_FAILED the operation failed.
608  *
609  * @api
610  */
611 bool mmcStartSequentialRead(MMCDriver *mmcp, uint32_t startblk) {
612 
613  osalDbgCheck(mmcp != NULL);
614  osalDbgAssert(mmcp->state == BLK_READY, "invalid state");
615 
616  /* Read operation in progress.*/
617  mmcp->state = BLK_READING;
618 
619  /* (Re)starting the SPI in case it has been reprogrammed externally, it can
620  happen if the SPI bus is shared among multiple peripherals.*/
621  spiStart(mmcp->config->spip, mmcp->config->hscfg);
622  spiSelect(mmcp->config->spip);
623 
624  if (mmcp->block_addresses) {
625  send_hdr(mmcp, MMCSD_CMD_READ_MULTIPLE_BLOCK, startblk);
626  }
627  else {
628  send_hdr(mmcp, MMCSD_CMD_READ_MULTIPLE_BLOCK, startblk * MMCSD_BLOCK_SIZE);
