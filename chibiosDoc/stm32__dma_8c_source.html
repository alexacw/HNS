24  * streams at runtime in order to allow all the other device
25  * drivers to coordinate the access to the resource.

437  *
438  * @init
439  */
440 void dmaInit(void) {
441  int i;

443  dma.streams_mask = 0U;

446  _stm32_dma_streams[i].channel->CCR = 0U;
447  _stm32_dma_isr_redir[i].dma_func = NULL;

451  DMA2->IFCR = 0xFFFFFFFFU;

455 /**
456  * @brief Allocates a DMA stream.
457  * @details The stream is allocated and, if required, the DMA clock enabled.
458  * The function also enables the IRQ vector associated to the stream
459  * and initializes its priority.
460  * @pre The stream must not be already in use or an error is returned.
461  * @post The stream is allocated and the default ISR handler redirected
462  * to the specified function.
463  * @post The stream ISR vector is enabled and its priority configured.
464  * @post The stream must be freed using @p dmaStreamRelease() before it can
465  * be reused with another peripheral.
466  * @post The stream is in its post-reset state.
467  * @note This function can be invoked in both ISR or thread context.
468  *
469  * @param[in] dmastp pointer to a stm32_dma_stream_t structure
470  * @param[in] priority IRQ priority for the DMA stream
471  * @param[in] func handling function pointer, can be @p NULL
472  * @param[in] param a parameter to be passed to the handling function
473  * @return The operation status.
474  * @retval false no error, stream taken.
475  * @retval true error, stream already taken.
476  *
477  * @special
478  */
479 bool dmaStreamAllocate(const stm32_dma_stream_t *dmastp,
480  uint32_t priority,
481  stm32_dmaisr_t func,
482  void *param) {
483 
484  osalDbgCheck(dmastp != NULL);
485 
486  /* Checks if the stream is already taken.*/
487  if ((dma.streams_mask & (1U << dmastp->selfindex)) != 0U)
488  return true;
489 
490  /* Installs the DMA handler.*/
491  _stm32_dma_isr_redir[dmastp->selfindex].dma_func = func;
492  _stm32_dma_isr_redir[dmastp->selfindex].dma_param = param;
493 
494  /* Enabling DMA clocks required by the current streams set.*/
495  if ((dma.streams_mask & STM32_DMA1_STREAMS_MASK) == 0U) {
496  rccEnableDMA1(true);
497  }
498 #if STM32_DMA2_NUM_CHANNELS > 0
499  if ((dma.streams_mask & STM32_DMA2_STREAMS_MASK) == 0U) {
500  rccEnableDMA2(true);
501  }
502 #endif
503 
504  /* Putting the stream in a safe state.*/
505  dmaStreamDisable(dmastp);
506  dmastp->channel->CCR = STM32_DMA_CCR_RESET_VALUE;
507 
508  /* Enables the associated IRQ vector if not already enabled and if a
509  callback is defined.*/
510  if (func != NULL) {
511  if ((dma.isr_mask & dmastp->cmask) == 0U) {
512  nvicEnableVector(dmastp->vector, priority);
513  }
514  dma.isr_mask |= (1U << dmastp->selfindex);
515  }
516 
517  /* Marks the stream as allocated.*/
518  dma.streams_mask |= (1U << dmastp->selfindex);
519 
520  return false;
521 }

523 /**
524  * @brief Releases a DMA stream.
525  * @details The stream is freed and, if required, the DMA clock disabled.
526  * Trying to release a unallocated stream is an illegal operation
527  * and is trapped if assertions are enabled.
528  * @pre The stream must have been allocated using @p dmaStreamAllocate().
529  * @post The stream is again available.
530  * @note This function can be invoked in both ISR or thread context.
531  *
532  * @param[in] dmastp pointer to a stm32_dma_stream_t structure
533  *
534  * @special
535  */
536 void dmaStreamRelease(const stm32_dma_stream_t *dmastp) {
537 
538  osalDbgCheck(dmastp != NULL);
539 
540  /* Check if the streams is not taken.*/
541  osalDbgAssert((dma.streams_mask & (1 << dmastp->selfindex)) != 0U,
542  "not allocated");
543 
544  /* Marks the stream as not allocated.*/
545  dma.streams_mask &= ~(1U << dmastp->selfindex);
546  dma.isr_mask &= ~(1U << dmastp->selfindex);
547 
548  /* Disables the associated IRQ vector if it is no more in use.*/
549  if ((dma.streams_mask & dmastp->cmask) == 0U) {
550  nvicDisableVector(dmastp->vector);
551  }
552 
553  /* Removes the DMA handler.*/
554  _stm32_dma_isr_redir[dmastp->selfindex].dma_func = NULL;
555  _stm32_dma_isr_redir[dmastp->selfindex].dma_param = NULL;
556 
557  /* Shutting down clocks that are no more required, if any.*/
558  if ((dma.streams_mask & STM32_DMA1_STREAMS_MASK) == 0U) {
559  rccDisableDMA1();
560  }
561 #if STM32_DMA2_NUM_CHANNELS > 0
562  if ((dma.streams_mask & STM32_DMA2_STREAMS_MASK) == 0U) {
563  rccDisableDMA2();
564  }
565 #endif
566 }
