248  * @brief TIM2 interrupt handler.

251  * @isr

253 OSAL_IRQ_HANDLER(ST_HANDLER) {
254 
255  OSAL_IRQ_PROLOGUE();
256 
257  /* Note, under rare circumstances an interrupt can remain latched even if
258  the timer SR register has been cleared, in those cases the interrupt
259  is simply ignored.*/
260  if ((STM32_ST_TIM->SR & TIM_SR_CC1IF) != 0U) {
261  STM32_ST_TIM->SR = 0U;
262 
263  osalSysLockFromISR();
264  osalOsTimerHandlerI();
265  osalSysUnlockFromISR();
266  }
267 
268  OSAL_IRQ_EPILOGUE();

276 /**
277  * @brief Low level ST driver initialization.
278  *
279  * @notapi
280  */
281 void st_lld_init(void) {
282 
283 #if OSAL_ST_MODE == OSAL_ST_MODE_FREERUNNING
284  /* Free running counter mode.*/
285 
286  /* Enabling timer clock.*/
287  ST_ENABLE_CLOCK();
288 
289  /* Enabling the stop mode during debug for this timer.*/
290  ST_ENABLE_STOP();
291 
292  /* Initializing the counter in free running mode.*/
293  STM32_ST_TIM->PSC = (ST_CLOCK_SRC / OSAL_ST_FREQUENCY) - 1;
294  STM32_ST_TIM->ARR = ST_ARR_INIT;
295  STM32_ST_TIM->CCMR1 = 0;
296  STM32_ST_TIM->CCR[0] = 0;
297  STM32_ST_TIM->DIER = 0;
298  STM32_ST_TIM->CR2 = 0;
299  STM32_ST_TIM->EGR = TIM_EGR_UG;
300  STM32_ST_TIM->CR1 = TIM_CR1_CEN;
301 
302  /* IRQ enabled.*/
303  nvicEnableVector(ST_NUMBER, STM32_ST_IRQ_PRIORITY);
304 #endif /* OSAL_ST_MODE == OSAL_ST_MODE_FREERUNNING */
305 
306 #if OSAL_ST_MODE == OSAL_ST_MODE_PERIODIC
307  /* Periodic systick mode, the Cortex-Mx internal systick timer is used
308  in this mode.*/
309  SysTick->LOAD = (SYSTICK_CK / OSAL_ST_FREQUENCY) - 1;
310  SysTick->VAL = 0;
311  SysTick->CTRL = SysTick_CTRL_CLKSOURCE_Msk |
312  SysTick_CTRL_ENABLE_Msk |
313  SysTick_CTRL_TICKINT_Msk;
314 
315  /* IRQ enabled.*/
316  nvicSetSystemHandlerPriority(HANDLER_SYSTICK, STM32_ST_IRQ_PRIORITY);
317 #endif /* OSAL_ST_MODE == OSAL_ST_MODE_PERIODIC */
318 }
319 
320 #endif /* OSAL_ST_MODE != OSAL_ST_MODE_NONE */
