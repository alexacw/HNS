68 /**
69  * @brief Programs the filters of CAN 1 and CAN 2.
70  *
71  * @param[in] canp pointer to the @p CANDriver object
72  * @param[in] can2sb number of the first filter assigned to CAN2
73  * @param[in] num number of entries in the filters array, if zero then
74  * a default filter is programmed
75  * @param[in] cfp pointer to the filters array, can be @p NULL if
76  * (num == 0)
77  *
78  * @notapi
79  */
80 static void can_lld_set_filters(CANDriver* canp,
81  uint32_t can2sb,
82  uint32_t num,
83  const CANFilter *cfp) {
84 
85 #if STM32_CAN_USE_CAN2
86  if(canp == &CAND2) {
87  /* Set handle to CAN1, because CAN1 manages the filters of CAN2.*/
88  canp = &CAND1;
89  }
90 #endif
91 
92  /* Temporarily enabling CAN clock.*/
93 #if STM32_CAN_USE_CAN1
94  if(canp == &CAND1) {
95  rccEnableCAN1(true);
96  /* Filters initialization.*/
97  canp->can->FMR = (canp->can->FMR & 0xFFFF0000) | CAN_FMR_FINIT;
98  canp->can->FMR = (canp->can->FMR & 0xFFFF0000) | (can2sb << 8) | CAN_FMR_FINIT;
99  }
100 #endif
101 
102 #if STM32_CAN_USE_CAN3
103  if(canp == &CAND3) {
104  rccEnableCAN3(true);
105  /* Filters initialization.*/
106  canp->can->FMR = (canp->can->FMR & 0xFFFF0000) | CAN_FMR_FINIT;
107  }
108 #endif
109 
110  if (num > 0) {
111  uint32_t i, fmask;
112 
113  /* All filters cleared.*/
114  canp->can->FA1R = 0;
115  canp->can->FM1R = 0;
116  canp->can->FS1R = 0;
117  canp->can->FFA1R = 0;
118 
119 #if STM32_CAN_USE_CAN1
120  if(canp == &CAND1) {
121  for (i = 0; i < STM32_CAN_MAX_FILTERS; i++) {
122  canp->can->sFilterRegister[i].FR1 = 0;
123  canp->can->sFilterRegister[i].FR2 = 0;
124  }
125  }
126 #endif
127 
128 #if STM32_CAN_USE_CAN3
129  if(canp == &CAND3) {
130  for (i = 0; i < STM32_CAN3_MAX_FILTERS; i++) {
131  canp->can->sFilterRegister[i].FR1 = 0;
132  canp->can->sFilterRegister[i].FR2 = 0;
133  }
134  }
135 #endif
136 
137  /* Scanning the filters array.*/
138  for (i = 0; i < num; i++) {
139  fmask = 1 << cfp->filter;
140  if (cfp->mode)
141  canp->can->FM1R |= fmask;
142  if (cfp->scale)
143  canp->can->FS1R |= fmask;
144  if (cfp->assignment)
145  canp->can->FFA1R |= fmask;
146  canp->can->sFilterRegister[cfp->filter].FR1 = cfp->register1;
147  canp->can->sFilterRegister[cfp->filter].FR2 = cfp->register2;
148  canp->can->FA1R |= fmask;
149  cfp++;
150  }
151  }
152  else {
153  /* Setting up a single default filter that enables everything for both
154  CANs.*/
155  canp->can->sFilterRegister[0].FR1 = 0;
156  canp->can->sFilterRegister[0].FR2 = 0;
157 #if STM32_CAN_USE_CAN2
158  if(canp == &CAND1) {
159  canp->can->sFilterRegister[can2sb].FR1 = 0;
160  canp->can->sFilterRegister[can2sb].FR2 = 0;
161  }
162 #endif
163  canp->can->FM1R = 0;
164  canp->can->FFA1R = 0;
165  canp->can->FS1R = 1;
166  canp->can->FA1R = 1;
167 #if STM32_CAN_USE_CAN2
168  if(canp == &CAND1) {
169  canp->can->FS1R |= 1 << can2sb;
170  canp->can->FA1R |= 1 << can2sb;
171  }
172 #endif
173  }
174  canp->can->FMR &= ~CAN_FMR_FINIT;
175 
176  /* Clock disabled, it will be enabled again in can_lld_start().*/
177  /* Temporarily enabling CAN clock.*/
178 #if STM32_CAN_USE_CAN1
179  if(canp == &CAND1) {
180  rccDisableCAN1();
181  }
182 #endif
183 #if STM32_CAN_USE_CAN3
184  if(canp == &CAND3) {
185  rccDisableCAN3();
186  }
187 #endif
188 }

190 /**
191  * @brief Common TX ISR handler.
192  *
193  * @param[in] canp pointer to the @p CANDriver object
194  *
195  * @notapi
196  */
197 static void can_lld_tx_handler(CANDriver *canp) {
198  uint32_t tsr;
199  eventflags_t flags;
200 
201  /* Clearing IRQ sources.*/
202  tsr = canp->can->TSR;
203  canp->can->TSR = tsr;
204 
205  /* Flags to be signaled through the TX event source.*/
206  flags = 0U;
207 
208  /* Checking mailbox 0.*/
209  if ((tsr & CAN_TSR_RQCP0) != 0U) {
210  if ((tsr & (CAN_TSR_ALST0 | CAN_TSR_TERR0)) != 0U) {
211  flags |= CAN_MAILBOX_TO_MASK(1U) << 16U;
212  }
213  else {
214  flags |= CAN_MAILBOX_TO_MASK(1U);
215  }
216  }
217 
218  /* Checking mailbox 1.*/
219  if ((tsr & CAN_TSR_RQCP1) != 0U) {
220  if ((tsr & (CAN_TSR_ALST1 | CAN_TSR_TERR1)) != 0U) {
221  flags |= CAN_MAILBOX_TO_MASK(2U) << 16U;
222  }
223  else {
224  flags |= CAN_MAILBOX_TO_MASK(2U);
225  }
226  }
227 
228  /* Checking mailbox 2.*/
229  if ((tsr & CAN_TSR_RQCP2) != 0U) {
230  if ((tsr & (CAN_TSR_ALST2 | CAN_TSR_TERR2)) != 0U) {
231  flags |= CAN_MAILBOX_TO_MASK(3U) << 16U;
232  }
233  else {
234  flags |= CAN_MAILBOX_TO_MASK(3U);
235  }
236  }
237 
238  /* Signaling flags and waking up threads waiting for a transmission slot.*/
239  _can_tx_empty_isr(canp, flags);
240 }

242 /**
243  * @brief Common RX0 ISR handler.
244  *
245  * @param[in] canp pointer to the @p CANDriver object
246  *
247  * @notapi
248  */
249 static void can_lld_rx0_handler(CANDriver *canp) {
250  uint32_t rf0r;
251 
252  rf0r = canp->can->RF0R;
253  if ((rf0r & CAN_RF0R_FMP0) > 0) {
254  /* No more receive events until the queue 0 has been emptied.*/
255  canp->can->IER &= ~CAN_IER_FMPIE0;
256  _can_rx_full_isr(canp, CAN_MAILBOX_TO_MASK(1U));
257  }
258  if ((rf0r & CAN_RF0R_FOVR0) > 0) {
259  /* Overflow events handling.*/
260  canp->can->RF0R = CAN_RF0R_FOVR0;
261  _can_error_isr(canp, CAN_OVERFLOW_ERROR);
262  }
263 }

288 /**
289  * @brief Common SCE ISR handler.
290  *
291  * @param[in] canp pointer to the @p CANDriver object
292  *
293  * @notapi
294  */
295 static void can_lld_sce_handler(CANDriver *canp) {
296  uint32_t msr;
297 
298  /* Clearing IRQ sources.*/
299  msr = canp->can->MSR;
300  canp->can->MSR = msr;
301 
302  /* Wakeup event.*/
303 #if CAN_USE_SLEEP_MODE
304  if (msr & CAN_MSR_WKUI) {
305  canp->state = CAN_READY;
306  canp->can->MCR &= ~CAN_MCR_SLEEP;
307  _can_wakeup_isr(canp);
308  }
309 #endif /* CAN_USE_SLEEP_MODE */
310  /* Error event.*/
311  if (msr & CAN_MSR_ERRI) {
312  eventflags_t flags;
313  uint32_t esr = canp->can->ESR;
314 
315 #if STM32_CAN_REPORT_ALL_ERRORS
316  flags = (eventflags_t)(esr & 7);
317  if ((esr & CAN_ESR_LEC) > 0)
318  flags |= CAN_FRAMING_ERROR;
319 #else
320  flags = 0;
321 #endif
322 
323  /* The content of the ESR register is copied unchanged in the upper
324  half word of the listener flags mask.*/
325  _can_error_isr(canp, flags | (eventflags_t)(esr << 16U));
326  }
327 }

795 /**
796  * @brief Inserts a frame into the transmit queue.
797  *
798  * @param[in] canp pointer to the @p CANDriver object
799  * @param[in] ctfp pointer to the CAN frame to be transmitted
800  * @param[in] mailbox mailbox number, @p CAN_ANY_MAILBOX for any mailbox
801  *
802  * @notapi
803  */
804 void can_lld_transmit(CANDriver *canp,
805  canmbx_t mailbox,
806  const CANTxFrame *ctfp) {
807  uint32_t tir;
808  CAN_TxMailBox_TypeDef *tmbp;
809 
810  /* Pointer to a free transmission mailbox.*/
811  switch (mailbox) {
812  case CAN_ANY_MAILBOX:
813  tmbp = &canp->can->sTxMailBox[(canp->can->TSR & CAN_TSR_CODE) >> 24];
814  break;
815  case 1:
816  tmbp = &canp->can->sTxMailBox[0];
817  break;
818  case 2:
819  tmbp = &canp->can->sTxMailBox[1];
820  break;
821  case 3:
822  tmbp = &canp->can->sTxMailBox[2];
823  break;
824  default:
825  return;
826  }
827 
828  /* Preparing the message.*/
829  if (ctfp->IDE)
830  tir = ((uint32_t)ctfp->EID << 3) | ((uint32_t)ctfp->RTR << 1) |
831  CAN_TI0R_IDE;
832  else
833  tir = ((uint32_t)ctfp->SID << 21) | ((uint32_t)ctfp->RTR << 1);
834  tmbp->TDTR = ctfp->DLC;
835  tmbp->TDLR = ctfp->data32[0];
836  tmbp->TDHR = ctfp->data32[1];
837  tmbp->TIR = tir | CAN_TI0R_TXRQ;
838 }

867 /**
868  * @brief Receives a frame from the input queue.
869  *
870  * @param[in] canp pointer to the @p CANDriver object
871  * @param[in] mailbox mailbox number, @p CAN_ANY_MAILBOX for any mailbox
872  * @param[out] crfp pointer to the buffer where the CAN frame is copied
873  *
874  * @notapi
875  */
876 void can_lld_receive(CANDriver *canp,
877  canmbx_t mailbox,
878  CANRxFrame *crfp) {
879  uint32_t rir, rdtr;
880 
881  if (mailbox == CAN_ANY_MAILBOX) {
882  if ((canp->can->RF0R & CAN_RF0R_FMP0) != 0)
883  mailbox = 1;
884  else if ((canp->can->RF1R & CAN_RF1R_FMP1) != 0)
885  mailbox = 2;
886  else {
887  /* Should not happen, do nothing.*/
888  return;
889  }
890  }
891  switch (mailbox) {
892  case 1:
893  /* Fetches the message.*/
894  rir = canp->can->sFIFOMailBox[0].RIR;
895  rdtr = canp->can->sFIFOMailBox[0].RDTR;
896  crfp->data32[0] = canp->can->sFIFOMailBox[0].RDLR;
897  crfp->data32[1] = canp->can->sFIFOMailBox[0].RDHR;
898 
899  /* Releases the mailbox.*/
900  canp->can->RF0R = CAN_RF0R_RFOM0;
901 
902  /* If the queue is empty re-enables the interrupt in order to generate
903  events again.*/
904  if ((canp->can->RF0R & CAN_RF0R_FMP0) == 0)
905  canp->can->IER |= CAN_IER_FMPIE0;
906  break;
907  case 2:
908  /* Fetches the message.*/
909  rir = canp->can->sFIFOMailBox[1].RIR;
910  rdtr = canp->can->sFIFOMailBox[1].RDTR;
911  crfp->data32[0] = canp->can->sFIFOMailBox[1].RDLR;
912  crfp->data32[1] = canp->can->sFIFOMailBox[1].RDHR;
913 
914  /* Releases the mailbox.*/
915  canp->can->RF1R = CAN_RF1R_RFOM1;
916 
917  /* If the queue is empty re-enables the interrupt in order to generate
918  events again.*/
919  if ((canp->can->RF1R & CAN_RF1R_FMP1) == 0)
920  canp->can->IER |= CAN_IER_FMPIE1;
921  break;
922  default:
923  /* Should not happen, do nothing.*/
924  return;
925  }
926 
927  /* Decodes the various fields in the RX frame.*/
928  crfp->RTR = (rir & CAN_RI0R_RTR) >> 1;
929  crfp->IDE = (rir & CAN_RI0R_IDE) >> 2;
930  if (crfp->IDE)
931  crfp->EID = rir >> 3;
932  else
933  crfp->SID = rir >> 21;
934  crfp->DLC = rdtr & CAN_RDT0R_DLC;
935  crfp->FMI = (uint8_t)(rdtr >> 8);
936  crfp->TIME = (uint16_t)(rdtr >> 16);
937 }
