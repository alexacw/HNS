131 /**
132  * @brief Status bits translation.
133  *
134  * @param[in] sr USART SR register value
135  *
136  * @return The error flags.
137  */
138 static uartflags_t translate_errors(uint16_t sr) {
139  uartflags_t sts = 0;
140 
141  if (sr & USART_SR_ORE)
142  sts |= UART_OVERRUN_ERROR;
143  if (sr & USART_SR_PE)
144  sts |= UART_PARITY_ERROR;
145  if (sr & USART_SR_FE)
146  sts |= UART_FRAMING_ERROR;
147  if (sr & USART_SR_NE)
148  sts |= UART_NOISE_ERROR;
149  if (sr & USART_SR_LBD)
150  sts |= UART_BREAK_DETECTED;
151  return sts;
152 }

154 /**
155  * @brief Puts the receiver in the UART_RX_IDLE state.
156  *
157  * @param[in] uartp pointer to the @p UARTDriver object
158  */
159 static void uart_enter_rx_idle_loop(UARTDriver *uartp) {
160  uint32_t mode;
161 
162  /* RX DMA channel preparation, if the char callback is defined then the
163  TCIE interrupt is enabled too.*/
164  if (uartp->config->rxchar_cb == NULL)
165  mode = STM32_DMA_CR_DIR_P2M | STM32_DMA_CR_CIRC;
166  else
167  mode = STM32_DMA_CR_DIR_P2M | STM32_DMA_CR_CIRC | STM32_DMA_CR_TCIE;
168  dmaStreamSetMemory0(uartp->dmarx, &uartp->rxbuf);
169  dmaStreamSetTransactionSize(uartp->dmarx, 1);
170  dmaStreamSetMode(uartp->dmarx, uartp->dmamode | mode);
171  dmaStreamEnable(uartp->dmarx);
172 }

174 /**
175  * @brief USART de-initialization.
176  * @details This function must be invoked with interrupts disabled.
177  *
178  * @param[in] uartp pointer to the @p UARTDriver object
179  */
180 static void usart_stop(UARTDriver *uartp) {
181 
182  /* Stops RX and TX DMA channels.*/
183  dmaStreamDisable(uartp->dmarx);
184  dmaStreamDisable(uartp->dmatx);
185 
186  /* Stops USART operations.*/
187  uartp->usart->CR1 = 0;
188  uartp->usart->CR2 = 0;
189  uartp->usart->CR3 = 0;
190 }

192 /**
193  * @brief USART initialization.
194  * @details This function must be invoked with interrupts disabled.
195  *
196  * @param[in] uartp pointer to the @p UARTDriver object
197  */
198 static void usart_start(UARTDriver *uartp) {
199  uint16_t cr1;
200  USART_TypeDef *u = uartp->usart;
201 
202  /* Defensive programming, starting from a clean state.*/
203  usart_stop(uartp);
204 
205  /* Baud rate setting.*/
206 #if STM32_HAS_USART6
207  if ((uartp->usart == USART1) || (uartp->usart == USART6))
208 #else
209  if (uartp->usart == USART1)
210 #endif
211  u->BRR = STM32_PCLK2 / uartp->config->speed;
212  else
213  u->BRR = STM32_PCLK1 / uartp->config->speed;
214 
215  /* Resetting eventual pending status flags.*/
216  (void)u->SR; /* SR reset step 1.*/
217  (void)u->DR; /* SR reset step 2.*/
218  u->SR = 0;
219 
220  /* Note that some bits are enforced because required for correct driver
221  operations.*/
222  u->CR2 = uartp->config->cr2 | USART_CR2_LBDIE;
223  u->CR3 = uartp->config->cr3 | USART_CR3_DMAT | USART_CR3_DMAR |
224  USART_CR3_EIE;
225 
226  /* Mustn't ever set TCIE here - if done, it causes an immediate
227  interrupt.*/
228  cr1 = USART_CR1_UE | USART_CR1_PEIE | USART_CR1_TE | USART_CR1_RE;
229  u->CR1 = uartp->config->cr1 | cr1;
230 
231  /* Starting the receiver idle loop.*/
232  uart_enter_rx_idle_loop(uartp);
233 }

235 /**
236  * @brief RX DMA common service routine.
237  *
238  * @param[in] uartp pointer to the @p UARTDriver object
239  * @param[in] flags pre-shifted content of the ISR register
240  */
241 static void uart_lld_serve_rx_end_irq(UARTDriver *uartp, uint32_t flags) {
242 
243  /* DMA errors handling.*/
244 #if defined(STM32_UART_DMA_ERROR_HOOK)
245  if ((flags & (STM32_DMA_ISR_TEIF | STM32_DMA_ISR_DMEIF)) != 0) {
246  STM32_UART_DMA_ERROR_HOOK(uartp);
247  }
248 #else
249  (void)flags;
250 #endif
251 
252  if (uartp->rxstate == UART_RX_IDLE) {
253  /* Receiver in idle state, a callback is generated, if enabled, for each
254  received character and then the driver stays in the same state.*/
255  _uart_rx_idle_code(uartp);
256  }
257  else {
258  /* Receiver in active state, a callback is generated, if enabled, after
259  a completed transfer.*/
260  dmaStreamDisable(uartp->dmarx);
261  _uart_rx_complete_isr_code(uartp);
262  }
263 }

265 /**
266  * @brief TX DMA common service routine.
267  *
268  * @param[in] uartp pointer to the @p UARTDriver object
269  * @param[in] flags pre-shifted content of the ISR register
270  */
271 static void uart_lld_serve_tx_end_irq(UARTDriver *uartp, uint32_t flags) {
272 
273  /* DMA errors handling.*/
274 #if defined(STM32_UART_DMA_ERROR_HOOK)
275  if ((flags & (STM32_DMA_ISR_TEIF | STM32_DMA_ISR_DMEIF)) != 0) {
276  STM32_UART_DMA_ERROR_HOOK(uartp);
277  }
278 #else
279  (void)flags;
280 #endif
281 
282  dmaStreamDisable(uartp->dmatx);
283 
284  /* A callback is generated, if enabled, after a completed transfer.*/
285  _uart_tx1_isr_code(uartp);
286 }

288 /**
289  * @brief USART common service routine.
290  *
291  * @param[in] uartp pointer to the @p UARTDriver object
292  */
293 static void serve_usart_irq(UARTDriver *uartp) {
294  uint16_t sr;
295  USART_TypeDef *u = uartp->usart;
296  uint32_t cr1 = u->CR1;
297 
298  sr = u->SR; /* SR reset step 1.*/
299  (void)u->DR; /* SR reset step 2.*/
300 
301  if (sr & (USART_SR_LBD | USART_SR_ORE | USART_SR_NE |
302  USART_SR_FE | USART_SR_PE)) {
303  u->SR = ~USART_SR_LBD;
304  _uart_rx_error_isr_code(uartp, translate_errors(sr));
305  }
306 
307  if ((sr & USART_SR_TC) && (cr1 & USART_CR1_TCIE)) {
308  /* TC interrupt cleared and disabled.*/
309  u->SR = ~USART_SR_TC;
310  u->CR1 = cr1 & ~USART_CR1_TCIE;
311 
312  /* End of transmission, a callback is generated.*/
313  _uart_tx2_isr_code(uartp);
314  }
315 }

605  STM32_UART_UART5_IRQ_PRIORITY,

607  (void *)uartp);
608  osalDbgAssert(!b, "stream already allocated");
609  rccEnableUART5(true);
610  nvicEnableVector(STM32_UART5_NUMBER, STM32_UART_UART5_IRQ_PRIORITY);
611  uartp->dmamode |= STM32_DMA_CR_CHSEL(UART5_RX_DMA_CHANNEL) |
612  STM32_DMA_CR_PL(STM32_UART_UART5_DMA_PRIORITY);
613  }
614 #endif
615 
616 #if STM32_UART_USE_USART6

619  b = dmaStreamAllocate(uartp->dmarx,
620  STM32_UART_USART6_IRQ_PRIORITY,

622  (void *)uartp);

624  b = dmaStreamAllocate(uartp->dmatx,
625  STM32_UART_USART6_IRQ_PRIORITY,
626  (stm32_dmaisr_t)uart_lld_serve_tx_end_irq,
627  (void *)uartp);
628  osalDbgAssert(!b, "stream already allocated");
629  rccEnableUSART6(true);
630  nvicEnableVector(STM32_USART6_NUMBER, STM32_UART_USART6_IRQ_PRIORITY);
631  uartp->dmamode |= STM32_DMA_CR_CHSEL(USART6_RX_DMA_CHANNEL) |
632  STM32_DMA_CR_PL(STM32_UART_USART6_DMA_PRIORITY);
633  }
634 #endif
635 
636  /* Static DMA setup, the transfer size depends on the USART settings,
637  it is 16 bits if M=1 and PCE=0 else it is 8 bits.*/
638  if ((uartp->config->cr1 & (USART_CR1_M | USART_CR1_PCE)) == USART_CR1_M)
639  uartp->dmamode |= STM32_DMA_CR_PSIZE_HWORD | STM32_DMA_CR_MSIZE_HWORD;
640  dmaStreamSetPeripheral(uartp->dmarx, &uartp->usart->DR);
641  dmaStreamSetPeripheral(uartp->dmatx, &uartp->usart->DR);
642  uartp->rxbuf = 0;
643  }
644

646  uartp->txstate = UART_TX_IDLE;
647  usart_start(uartp);
648 }

729  dmaStreamSetTransactionSize(uartp->dmatx, n);
730  dmaStreamSetMode(uartp->dmatx, uartp->dmamode | STM32_DMA_CR_DIR_M2P |

735  which could be set before.*/

738 #else
739  if (uartp->config->txend2_cb != NULL) {
740 #endif
741  uartp->usart->SR = ~USART_SR_TC;
742  uartp->usart->CR1 |= USART_CR1_TCIE;
743  }
744 
745  /* Starting transfer.*/
746  dmaStreamEnable(uartp->dmatx);
747 }
