218  * @brief Writes to a TX FIFO.
219  *
220  * @param[in] fifop pointer to the FIFO register

222  * @param[in] n maximum number of bytes to copy

226 static void otg_fifo_write_from_buffer(volatile uint32_t *fifop,
227  const uint8_t *buf,
228  size_t n) {

230  osalDbgAssert(n > 0, "is zero");
231 
232  while (true) {

235  break;
236  }
237  n -= 4;
238  buf += 4;

242 /**
243  * @brief Reads a packet from the RXFIFO.
244  *
245  * @param[in] fifop pointer to the FIFO register
246  * @param[out] buf buffer where to copy the endpoint data
247  * @param[in] n number of bytes to pull from the FIFO
248  * @param[in] max number of bytes to copy into the buffer
249  *
250  * @notapi
251  */
252 static void otg_fifo_read_to_buffer(volatile uint32_t *fifop,
253  uint8_t *buf,
254  size_t n,
255  size_t max) {
256  uint32_t w = 0;
257  size_t i = 0;
258 
259  while (i < n) {
260  if ((i & 3) == 0){
261  w = *fifop;
262  }
263  if (i < max) {
264  *buf++ = (uint8_t)w;
265  w >>= 8;
266  }
267  i++;
268  }
269 }

271 /**
272  * @brief Incoming packets handler.
273  *
274  * @param[in] usbp pointer to the @p USBDriver object
275  *
276  * @notapi
277  */
278 static void otg_rxfifo_handler(USBDriver *usbp) {
279  uint32_t sts, cnt, ep;
280 
281  /* Popping the event word out of the RX FIFO.*/
282  sts = usbp->otg->GRXSTSP;
283 
284  /* Event details.*/
285  cnt = (sts & GRXSTSP_BCNT_MASK) >> GRXSTSP_BCNT_OFF;
286  ep = (sts & GRXSTSP_EPNUM_MASK) >> GRXSTSP_EPNUM_OFF;
287 
288  switch (sts & GRXSTSP_PKTSTS_MASK) {
289  case GRXSTSP_SETUP_DATA:
290  otg_fifo_read_to_buffer(usbp->otg->FIFO[0], usbp->epc[ep]->setup_buf,
291  cnt, 8);
292  break;
293  case GRXSTSP_SETUP_COMP:
294  break;
295  case GRXSTSP_OUT_DATA:
296  otg_fifo_read_to_buffer(usbp->otg->FIFO[0],
297  usbp->epc[ep]->out_state->rxbuf,
298  cnt,
299  usbp->epc[ep]->out_state->rxsize -
300  usbp->epc[ep]->out_state->rxcnt);
301  usbp->epc[ep]->out_state->rxbuf += cnt;
302  usbp->epc[ep]->out_state->rxcnt += cnt;
303  break;
304  case GRXSTSP_OUT_COMP:
305  break;
306  case GRXSTSP_OUT_GLOBAL_NAK:
307  break;
308  default:
309  break;
310  }
311 }

313 /**
314  * @brief Outgoing packets handler.
315  *
316  * @param[in] usbp pointer to the @p USBDriver object
317  * @param[in] ep endpoint number
318  *
319  * @notapi
320  */
321 static bool otg_txfifo_handler(USBDriver *usbp, usbep_t ep) {
322 
323  /* The TXFIFO is filled until there is space and data to be transmitted.*/
324  while (true) {
325  uint32_t n;
326 
327  /* Transaction end condition.*/
328  if (usbp->epc[ep]->in_state->txcnt >= usbp->epc[ep]->in_state->txsize) {
329 #if 1
330  usbp->otg->DIEPEMPMSK &= ~DIEPEMPMSK_INEPTXFEM(ep);
331 #endif
332  return true;
333  }
334 
335  /* Number of bytes remaining in current transaction.*/
336  n = usbp->epc[ep]->in_state->txsize - usbp->epc[ep]->in_state->txcnt;
337  if (n > usbp->epc[ep]->in_maxsize)
338  n = usbp->epc[ep]->in_maxsize;
339 
340  /* Checks if in the TXFIFO there is enough space to accommodate the
341  next packet.*/
342  if (((usbp->otg->ie[ep].DTXFSTS & DTXFSTS_INEPTFSAV_MASK) * 4) < n)
343  return false;
344 
345 #if STM32_USB_OTGFIFO_FILL_BASEPRI
346  __set_BASEPRI(CORTEX_PRIO_MASK(STM32_USB_OTGFIFO_FILL_BASEPRI));
347 #endif
348  otg_fifo_write_from_buffer(usbp->otg->FIFO[ep],
349  usbp->epc[ep]->in_state->txbuf,
350  n);
351  usbp->epc[ep]->in_state->txbuf += n;
352  usbp->epc[ep]->in_state->txcnt += n;
353 #if STM32_USB_OTGFIFO_FILL_BASEPRI
354  __set_BASEPRI(0);
355 #endif
356  }
357 }

368  stm32_otg_t *otgp = usbp->otg;
369  uint32_t epint = otgp->ie[ep].DIEPINT;
370 
371  otgp->ie[ep].DIEPINT = epint;
372 
373  if (epint & DIEPINT_TOC) {
374  /* Timeouts not handled yet, not sure how to handle.*/
375  }
376  if ((epint & DIEPINT_XFRC) && (otgp->DIEPMSK & DIEPMSK_XFRCM)) {
377  /* Transmit transfer complete.*/
378  USBInEndpointState *isp = usbp->epc[ep]->in_state;
379 
380  if (isp->txsize < isp->totsize) {
381  /* In case the transaction covered only part of the total transfer
382  then another transaction is immediately started in order to
383  cover the remaining.*/
384  isp->txsize = isp->totsize - isp->txsize;
385  isp->txcnt = 0;
386  osalSysLockFromISR();
387  usb_lld_start_in(usbp, ep);
388  osalSysUnlockFromISR();
389  }
390  else {
391  /* End on IN transfer.*/
392  _usb_isr_invoke_in_cb(usbp, ep);
393  }
394  }
395  if ((epint & DIEPINT_TXFE) &&
396  (otgp->DIEPEMPMSK & DIEPEMPMSK_INEPTXFEM(ep))) {
397 #if 0
398  /* The thread is made ready, it will be scheduled on ISR exit.*/
399  osalSysLockFromISR();
400  usbp->txpending |= (1 << ep);
401  otgp->DIEPEMPMSK &= ~(1 << ep);

411 /**
412  * @brief Generic endpoint OUT handler.
413  *
414  * @param[in] usbp pointer to the @p USBDriver object
415  * @param[in] ep endpoint number
416  *
417  * @notapi
418  */
419 static void otg_epout_handler(USBDriver *usbp, usbep_t ep) {
420  stm32_otg_t *otgp = usbp->otg;
421  uint32_t epint = otgp->oe[ep].DOEPINT;
422 
423  /* Resets all EP IRQ sources.*/
424  otgp->oe[ep].DOEPINT = epint;
425 
426  if ((epint & DOEPINT_STUP) && (otgp->DOEPMSK & DOEPMSK_STUPM)) {
427  /* Setup packets handling, setup packets are handled using a
428  specific callback.*/
429  _usb_isr_invoke_setup_cb(usbp, ep);
430  }
431 
432  if ((epint & DOEPINT_XFRC) && (otgp->DOEPMSK & DOEPMSK_XFRCM)) {
433  USBOutEndpointState *osp;
434 
435  /* OUT state structure pointer for this endpoint.*/
436  osp = usbp->epc[ep]->out_state;
437 
438  /* EP0 requires special handling.*/
439  if (ep == 0) {
440 
441 #if defined(STM32_OTG_SEQUENCE_WORKAROUND)
442  /* If an OUT transaction end interrupt is processed while the state
443  machine is not in an OUT state then it is ignored, this is caused
444  on some devices (L4) apparently injecting spurious data complete
445  words in the RX FIFO.*/
446  if ((usbp->ep0state & USB_OUT_STATE) == 0)
447  return;
448 #endif
449 
450  /* In case the transaction covered only part of the total transfer
451  then another transaction is immediately started in order to
452  cover the remaining.*/
453  if (((osp->rxcnt % usbp->epc[ep]->out_maxsize) == 0) &&
454  (osp->rxsize < osp->totsize)) {
455  osp->rxsize = osp->totsize - osp->rxsize;
456  osp->rxcnt = 0;
457  osalSysLockFromISR();
458  usb_lld_start_out(usbp, ep);
459  osalSysUnlockFromISR();
460  return;
461  }
462  }
463 
464  /* End on OUT transfer.*/
465  _usb_isr_invoke_out_cb(usbp, ep);
466  }
467 }

537  * @brief OTG shared ISR.
538  *
539  * @param[in] usbp pointer to the @p USBDriver object
540  *
541  * @notapi
542  */
543 static void usb_lld_serve_interrupt(USBDriver *usbp) {
544  stm32_otg_t *otgp = usbp->otg;
545  uint32_t sts, src;
546 
547  sts = otgp->GINTSTS;
548  sts &= otgp->GINTMSK;
549  otgp->GINTSTS = sts;
550 
551  /* Reset interrupt handling.*/
552  if (sts & GINTSTS_USBRST) {
553 #if 0
554  /* Resetting pending operations.*/
555  usbp->txpending = 0;
556 #endif
557  /* Default reset action.*/
558  _usb_reset(usbp);
559 
560  /* Preventing execution of more handlers, the core has been reset.*/
561  return;
562  }
563 
564  /* Wake-up handling.*/
565  if (sts & GINTSTS_WKUPINT) {
566  /* If clocks are gated off, turn them back on (may be the case if
567  coming out of suspend mode).*/
568  if (otgp->PCGCCTL & (PCGCCTL_STPPCLK | PCGCCTL_GATEHCLK)) {
569  /* Set to zero to un-gate the USB core clocks.*/
570  otgp->PCGCCTL &= ~(PCGCCTL_STPPCLK | PCGCCTL_GATEHCLK);
571  }
572 
573  /* Clear the Remote Wake-up Signaling.*/
574  otgp->DCTL &= ~DCTL_RWUSIG;
575 
576  _usb_wakeup(usbp);
577  }
578 
579  /* Suspend handling.*/
580  if (sts & GINTSTS_USBSUSP) {
581 #if 0
582  /* Resetting pending operations.*/
583  usbp->txpending = 0;
584 #endif
585  /* Default suspend action.*/
586  _usb_suspend(usbp);
587  }
588 
589  /* Enumeration done.*/
590  if (sts & GINTSTS_ENUMDNE) {
591  /* Full or High speed timing selection.*/
592  if ((otgp->DSTS & DSTS_ENUMSPD_MASK) == DSTS_ENUMSPD_HS_480) {
593  otgp->GUSBCFG = (otgp->GUSBCFG & ~(GUSBCFG_TRDT_MASK)) |
594  GUSBCFG_TRDT(TRDT_VALUE_HS);
595  }
596  else {
597  otgp->GUSBCFG = (otgp->GUSBCFG & ~(GUSBCFG_TRDT_MASK)) |
598  GUSBCFG_TRDT(TRDT_VALUE_FS);
599  }
600  }
601 
602  /* SOF interrupt handling.*/
603  if (sts & GINTSTS_SOF) {
604  _usb_isr_invoke_sof_cb(usbp);
605  }
606 
607  /* Isochronous IN failed handling */
608  if (sts & GINTSTS_IISOIXFR) {
609  otg_isoc_in_failed_handler(usbp);
610  }
611 
612  /* Isochronous OUT failed handling */
613  if (sts & GINTSTS_IISOOXFR) {
614  otg_isoc_out_failed_handler(usbp);
615  }
616 
617  /* RX FIFO not empty handling.*/
618 #if 0
619  if (sts & GINTSTS_RXFLVL) {
620  /* The interrupt is masked while the thread has control or it would
621  be triggered again.*/
622  osalSysLockFromISR();
623  otgp->GINTMSK &= ~GINTMSK_RXFLVLM;
624  osalThreadResumeI(&usbp->wait, MSG_OK);
625  osalSysUnlockFromISR();
626  }
627 #else
628  /* Performing the whole FIFO emptying in the ISR, it is advised to keep
629  this IRQ at a very low priority level.*/
630  if ((sts & GINTSTS_RXFLVL) != 0U) {
631  otg_rxfifo_handler(usbp);
632  }
633 #endif
634 
635  /* IN/OUT endpoints event handling.*/
636  src = otgp->DAINT;
637  if (sts & GINTSTS_OEPINT) {
638  if (src & (1 << 16))
639  otg_epout_handler(usbp, 0);
640  if (src & (1 << 17))
641  otg_epout_handler(usbp, 1);
642  if (src & (1 << 18))
643  otg_epout_handler(usbp, 2);
644  if (src & (1 << 19))
645  otg_epout_handler(usbp, 3);
646 #if USB_MAX_ENDPOINTS >= 4
647  if (src & (1 << 20))
648  otg_epout_handler(usbp, 4);
649 #endif
650 #if USB_MAX_ENDPOINTS >= 5
651  if (src & (1 << 21))
652  otg_epout_handler(usbp, 5);
653 #endif
654 #if USB_MAX_ENDPOINTS >= 6
655  if (src & (1 << 22))
656  otg_epout_handler(usbp, 6);

1000  *
1001  * @param[in] usbp pointer to the @p USBDriver object
1002  *
1003  * @notapi
1004  */
1005 void usb_lld_reset(USBDriver *usbp) {

1007  stm32_otg_t *otgp = usbp->otg;
1008 
1009  /* Flush the Tx FIFO.*/
1010  otg_txfifo_flush(usbp, 0);
1011 
1012  /* Endpoint interrupts all disabled and cleared.*/
1013  otgp->DIEPEMPMSK = 0;
1014  otgp->DAINTMSK = DAINTMSK_OEPM(0) | DAINTMSK_IEPM(0);
1015 
1016  /* All endpoints in NAK mode, interrupts cleared.*/
1017  for (i = 0; i <= usbp->otgparams->num_endpoints; i++) {
1018  otgp->ie[i].DIEPCTL = DIEPCTL_SNAK;
1019  otgp->oe[i].DOEPCTL = DOEPCTL_SNAK;
1020  otgp->ie[i].DIEPINT = 0xFFFFFFFF;
1021  otgp->oe[i].DOEPINT = 0xFFFFFFFF;

1026

1033

1036  otgp->DIEPMSK = DIEPMSK_TOCM | DIEPMSK_XFRCM;
1037  otgp->DOEPMSK = DOEPMSK_STUPM | DOEPMSK_XFRCM;
1038

1041  otgp->oe[0].DOEPTSIZ = DOEPTSIZ_STUPCNT(3);
1042  otgp->oe[0].DOEPCTL = DOEPCTL_SD0PID | DOEPCTL_USBAEP | DOEPCTL_EPTYP_CTRL |

1067  *

1069  * @param[in] ep endpoint number
1070  *
1071  * @notapi
1072  */

1075  stm32_otg_t *otgp = usbp->otg;
1076 
1077  /* IN and OUT common parameters.*/
1078  switch (usbp->epc[ep]->ep_mode & USB_EP_MODE_TYPE) {
1079  case USB_EP_MODE_TYPE_CTRL:
1080  ctl = DIEPCTL_SD0PID | DIEPCTL_USBAEP | DIEPCTL_EPTYP_CTRL;

1085  case USB_EP_MODE_TYPE_BULK:
1086  ctl = DIEPCTL_SD0PID | DIEPCTL_USBAEP | DIEPCTL_EPTYP_BULK;
1087  break;
1088  case USB_EP_MODE_TYPE_INTR:
1089  ctl = DIEPCTL_SD0PID | DIEPCTL_USBAEP | DIEPCTL_EPTYP_INTR;
1090  break;
1091  default:
1092  return;
1093  }

1098  otgp->oe[ep].DOEPCTL = ctl | DOEPCTL_MPSIZ(usbp->epc[ep]->out_maxsize);
1099  otgp->DAINTMSK |= DAINTMSK_OEPM(ep);
1100  }
1101  else {
1102  otgp->oe[ep].DOEPCTL &= ~DOEPCTL_USBAEP;
1103  otgp->DAINTMSK &= ~DAINTMSK_OEPM(ep);
1104  }

1111  if (usbp->epc[ep]->in_multiplier > 1)
1112  fsize *= usbp->epc[ep]->in_multiplier;

1114  DIEPTXF_INEPTXSA(otg_ram_alloc(usbp, fsize));
1115  otg_txfifo_flush(usbp, ep);
1116 
1117  otgp->ie[ep].DIEPCTL = ctl |
1118  DIEPCTL_TXFNUM(ep) |
1119  DIEPCTL_MPSIZ(usbp->epc[ep]->in_maxsize);
1120  otgp->DAINTMSK |= DAINTMSK_IEPM(ep);
1121  }
1122  else {
1123  otgp->DIEPTXF[ep - 1] = 0x02000400; /* Reset value.*/
1124  otg_txfifo_flush(usbp, ep);
1125  otgp->ie[ep].DIEPCTL &= ~DIEPCTL_USBAEP;
1126  otgp->DAINTMSK &= ~DAINTMSK_IEPM(ep);
1127  }
1128 }

1215 /**
1216  * @brief Starts a receive operation on an OUT endpoint.
1217  *
1218  * @param[in] usbp pointer to the @p USBDriver object
1219  * @param[in] ep endpoint number
1220  *
1221  * @notapi
1222  */
1223 void usb_lld_start_out(USBDriver *usbp, usbep_t ep) {
1224  uint32_t pcnt, rxsize;
1225  USBOutEndpointState *osp = usbp->epc[ep]->out_state;
1226 
1227  /* Transfer initialization.*/
1228  osp->totsize = osp->rxsize;
1229  if ((ep == 0) && (osp->rxsize > EP0_MAX_OUTSIZE))
1230  osp->rxsize = EP0_MAX_OUTSIZE;
1231 
1232  /* Transaction size is rounded to a multiple of packet size because the
1233  following requirement in the RM:
1234  "For OUT transfers, the transfer size field in the endpoint's transfer
1235  size register must be a multiple of the maximum packet size of the
1236  endpoint, adjusted to the Word boundary".*/
1237  pcnt = (osp->rxsize + usbp->epc[ep]->out_maxsize - 1U) /
1238  usbp->epc[ep]->out_maxsize;
1239  rxsize = (pcnt * usbp->epc[ep]->out_maxsize + 3U) & 0xFFFFFFFCU;
1240 
1241  /*Setting up transaction parameters in DOEPTSIZ.*/
1242  usbp->otg->oe[ep].DOEPTSIZ = DOEPTSIZ_STUPCNT(3) | DOEPTSIZ_PKTCNT(pcnt) |
1243  DOEPTSIZ_XFRSIZ(rxsize);
1244 
1245  /* Special case of isochronous endpoint.*/
1246  if ((usbp->epc[ep]->ep_mode & USB_EP_MODE_TYPE) == USB_EP_MODE_TYPE_ISOC) {
1247  /* Odd/even bit toggling for isochronous endpoint.*/
1248  if (usbp->otg->DSTS & DSTS_FNSOF_ODD)
