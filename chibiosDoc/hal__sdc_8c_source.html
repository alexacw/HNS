88 /**
89  * @brief Detects card mode.
90  *
91  * @param[in] sdcp pointer to the @p SDCDriver object
92  *
93  * @return The operation status.
94  * @retval HAL_SUCCESS operation succeeded.
95  * @retval HAL_FAILED operation failed.
96  *
97  * @notapi
98  */
99 static bool mode_detect(SDCDriver *sdcp) {
100  uint32_t resp[1];
101 
102  /* V2.0 cards detection.*/
103  if (!sdc_lld_send_cmd_short_crc(sdcp, MMCSD_CMD_SEND_IF_COND,
104  MMCSD_CMD8_PATTERN, resp)) {
105  sdcp->cardmode = SDC_MODE_CARDTYPE_SDV20;
106  /* Voltage verification.*/
107  if (((resp[0] >> 8U) & 0xFU) != 1U) {
108  return HAL_FAILED;
109  }
110  if (sdc_lld_send_cmd_short_crc(sdcp, MMCSD_CMD_APP_CMD, 0, resp) ||
111  MMCSD_R1_ERROR(resp[0])) {
112  return HAL_FAILED;
113  }
114  }
115  else {
116  /* MMC or SD V1.1 detection.*/
117  if (sdc_lld_send_cmd_short_crc(sdcp, MMCSD_CMD_APP_CMD, 0, resp) ||
118  MMCSD_R1_ERROR(resp[0])) {
119  sdcp->cardmode = SDC_MODE_CARDTYPE_MMC;
120  }
121  else {
122  sdcp->cardmode = SDC_MODE_CARDTYPE_SDV11;
123 
124  /* Reset error flag illegal command.*/
125  sdc_lld_send_cmd_none(sdcp, MMCSD_CMD_GO_IDLE_STATE, 0);
126  }
127  }
128 
129  return HAL_SUCCESS;
130 }

132 /**
133  * @brief Init procedure for MMC.
134  *
135  * @param[in] sdcp pointer to the @p SDCDriver object
136  *
137  * @return The operation status.
138  * @retval HAL_SUCCESS operation succeeded.
139  * @retval HAL_FAILED operation failed.
140  *
141  * @notapi
142  */
143 static bool mmc_init(SDCDriver *sdcp) {
144  uint32_t ocr;
145  unsigned i;
146  uint32_t resp[1];
147 
148  ocr = 0xC0FF8000U;
149  i = 0;
150  while (true) {
151  if (sdc_lld_send_cmd_short(sdcp, MMCSD_CMD_INIT, ocr, resp)) {
152  return HAL_FAILED;
153  }
154  if ((resp[0] & 0x80000000U) != 0U) {
155  if ((resp[0] & 0x40000000U) != 0U) {
156  sdcp->cardmode |= SDC_MODE_HIGH_CAPACITY;
157  }
158  break;
159  }
160  if (++i >= (unsigned)SDC_INIT_RETRY) {
161  return HAL_FAILED;
162  }
163  osalThreadSleepMilliseconds(10);
164  }
165 
166  return HAL_SUCCESS;
167 }

169 /**
170  * @brief Init procedure for SDC.
171  *
172  * @param[in] sdcp pointer to the @p SDCDriver object
173  *
174  * @return The operation status.
175  * @retval HAL_SUCCESS operation succeeded.
176  * @retval HAL_FAILED operation failed.
177  *
178  * @notapi
179  */
180 static bool sdc_init(SDCDriver *sdcp) {
181  unsigned i;
182  uint32_t ocr;
183  uint32_t resp[1];
184 
185  if ((sdcp->cardmode & SDC_MODE_CARDTYPE_MASK) == SDC_MODE_CARDTYPE_SDV20) {
186  ocr = SDC_INIT_OCR_V20;
187  }
188  else {
189  ocr = SDC_INIT_OCR;
190  }
191 
192  i = 0;
193  while (true) {
194  if (sdc_lld_send_cmd_short_crc(sdcp, MMCSD_CMD_APP_CMD, 0, resp) ||
195  MMCSD_R1_ERROR(resp[0])) {
196  return HAL_FAILED;
197  }
198  if (sdc_lld_send_cmd_short(sdcp, MMCSD_CMD_APP_OP_COND, ocr, resp)) {
199  return HAL_FAILED;
200  }
201  if ((resp[0] & 0x80000000U) != 0U) {
202  if ((resp[0] & 0x40000000U) != 0U) {
203  sdcp->cardmode |= SDC_MODE_HIGH_CAPACITY;
204  }
205  break;
206  }
207  if (++i >= (unsigned)SDC_INIT_RETRY) {
208  return HAL_FAILED;
209  }
210  osalThreadSleepMilliseconds(10);
211  }
212 
213  return HAL_SUCCESS;
214 }

216 /**
217  * @brief Constructs CMD6 argument for MMC.
218  *
219  * @param[in] access EXT_CSD access mode
220  * @param[in] idx EXT_CSD byte number
221  * @param[in] value value to be written in target field
222  * @param[in] cmd_set switch current command set
223  *
224  * @return CMD6 argument.
225  *
226  * @notapi
227  */
228 static uint32_t mmc_cmd6_construct(mmc_switch_t access, uint32_t idx,
229  uint32_t value, uint32_t cmd_set) {
230 
231  osalDbgAssert(idx <= 191U, "This field is not writable");
232  osalDbgAssert(cmd_set < 8U, "This field has only 3 bits");
233 
234  return ((uint32_t)access << 24U) | (idx << 16U) | (value << 8U) | cmd_set;
235 }
236 
237 /**
238  * @brief Constructs CMD6 argument for SDC.
239  *
240  * @param[in] mode switch/test mode
241  * @param[in] function function number to be switched
242  * @param[in] value value to be written in target function
243  *
244  * @return CMD6 argument.
245  *
246  * @notapi
247  */
248 static uint32_t sdc_cmd6_construct(sd_switch_t mode,
249  sd_switch_function_t function,
250  uint32_t value) {
251  uint32_t ret = 0xFFFFFF;
252 
253  osalDbgAssert((value < 16U), "This field has only 4 bits");
254 
255  ret &= ~((uint32_t)0xFU << ((uint32_t)function * 4U));
256  ret |= value << ((uint32_t)function * 4U);
257  return ret | ((uint32_t)mode << 31U);
258 }

260 /**
261  * @brief Extracts information from CMD6 answer.
262  *
263  * @param[in] function function number to be switched
264  * @param[in] buf buffer with answer
265  *
266  * @return extracted answer.
267  *
268  * @notapi
269  */
270 static uint16_t sdc_cmd6_extract_info(sd_switch_function_t function,
271  const uint8_t *buf) {
272 
273  unsigned start = 12U - ((unsigned)function * 2U);
274 
275  return ((uint16_t)buf[start] << 8U) | (uint16_t)buf[start + 1U];
276 }

278 /**
279  * @brief Checks status after switching using CMD6.
280  *
281  * @param[in] function function number to be switched
282  * @param[in] buf buffer with answer
283  *
284  * @return The operation status.
285  * @retval HAL_SUCCESS operation succeeded.
286  * @retval HAL_FAILED operation failed.
287  *
288  * @notapi
289  */
290 static bool sdc_cmd6_check_status(sd_switch_function_t function,
291  const uint8_t *buf) {
292 
293  uint32_t tmp;
294  uint32_t status;
295 
296  tmp = ((uint32_t)buf[14] << 16U) |
297  ((uint32_t)buf[15] << 8U) |
298  (uint32_t)buf[16];
299  status = (tmp >> ((uint32_t)function * 4U)) & 0xFU;
300  if (0xFU != status) {
301  return HAL_SUCCESS;
302  }
303  return HAL_FAILED;
304 }

306 /**
307  * @brief Reads supported bus clock and switch SDC to appropriate mode.
308  *
309  * @param[in] sdcp pointer to the @p SDCDriver object
310  * @param[out] clk pointer to clock enum
311  *
312  * @return The operation status.
313  * @retval HAL_SUCCESS operation succeeded.
314  * @retval HAL_FAILED operation failed.
315  *
316  * @notapi
317  */
318 static bool sdc_detect_bus_clk(SDCDriver *sdcp, sdcbusclk_t *clk) {
319  uint32_t cmdarg;
320  const size_t N = 64;
321  uint8_t tmp[N];
322 
323  /* Safe default.*/
324  *clk = SDC_CLK_25MHz;
325 
326  /* Looks like only "high capacity" cards produce meaningful results during
327  this clock detection procedure.*/
328  if (0U == _mmcsd_get_slice(sdcp->csd, MMCSD_CSD_10_CSD_STRUCTURE_SLICE)) {
329  *clk = SDC_CLK_25MHz;
330  return HAL_SUCCESS;
331  }
332 
333  /* Read switch functions' register.*/
334  if (sdc_lld_read_special(sdcp, tmp, N, MMCSD_CMD_SWITCH, 0)) {
335  return HAL_FAILED;
336  }
337 
338  /* Check card capabilities parsing acquired data.*/
339  if ((sdc_cmd6_extract_info(SD_SWITCH_FUNCTION_SPEED, tmp) & 2U) == 2U) {
340  /* Construct command to set the bus speed.*/
341  cmdarg = sdc_cmd6_construct(SD_SWITCH_SET, SD_SWITCH_FUNCTION_SPEED, 1);
342 
343  /* Write constructed command and read operation status in single call.*/
344  if (sdc_lld_read_special(sdcp, tmp, N, MMCSD_CMD_SWITCH, cmdarg)) {
345  return HAL_FAILED;
346  }
347 
348  /* Check card answer for success status bits.*/
349  if (HAL_SUCCESS == sdc_cmd6_check_status(SD_SWITCH_FUNCTION_SPEED, tmp)) {
350  *clk = SDC_CLK_50MHz;
351  }
352  else {
353  *clk = SDC_CLK_25MHz;
354  }
355  }
356 
357  return HAL_SUCCESS;
358 }

360 /**
361  * @brief Reads supported bus clock and switch MMC to appropriate mode.
362  *
363  * @param[in] sdcp pointer to the @p SDCDriver object
364  * @param[out] clk pointer to clock enum
365  *
366  * @return The operation status.
367  * @retval HAL_SUCCESS operation succeeded.
368  * @retval HAL_FAILED operation failed.
369  *
370  * @notapi
371  */
372 static bool mmc_detect_bus_clk(SDCDriver *sdcp, sdcbusclk_t *clk) {
373  uint32_t cmdarg;
374  uint32_t resp[1];
375  uint8_t *scratchpad = sdcp->config->scratchpad;
376 
377  /* Safe default.*/
378  *clk = SDC_CLK_25MHz;
379 
380  /* Use safe default when there is no space for data.*/
381  if (NULL == scratchpad) {
382  return HAL_SUCCESS;
383  }
384 
385  cmdarg = mmc_cmd6_construct(MMC_SWITCH_WRITE_BYTE, 185, 1, 0);
386  if (!(sdc_lld_send_cmd_short_crc(sdcp, MMCSD_CMD_SWITCH, cmdarg, resp) ||
387  MMCSD_R1_ERROR(resp[0]))) {
388  *clk = SDC_CLK_50MHz;
389  }
390 
391  return HAL_SUCCESS;
392 }

414 /**
415  * @brief Sets bus width for SDC.
416  *
417  * @param[in] sdcp pointer to the @p SDCDriver object
418  *
419  * @return The operation status.
420  * @retval HAL_SUCCESS operation succeeded.
421  * @retval HAL_FAILED operation failed.
422  *
423  * @notapi
424  */
425 static bool sdc_set_bus_width(SDCDriver *sdcp) {
426  uint32_t resp[1];
427 
428  if (SDC_MODE_1BIT == sdcp->config->bus_width) {
429  /* Nothing to do. Bus is already in 1bit mode.*/
430  return HAL_SUCCESS;
431  }
432  else if (SDC_MODE_4BIT == sdcp->config->bus_width) {
433  sdc_lld_set_bus_mode(sdcp, SDC_MODE_4BIT);
434  if (sdc_lld_send_cmd_short_crc(sdcp, MMCSD_CMD_APP_CMD, sdcp->rca, resp) ||
435  MMCSD_R1_ERROR(resp[0])) {
436  return HAL_FAILED;
437  }
438 
439  if (sdc_lld_send_cmd_short_crc(sdcp, MMCSD_CMD_SET_BUS_WIDTH, 2, resp) ||
440  MMCSD_R1_ERROR(resp[0])) {
441  return HAL_FAILED;
442  }
443  }
444  else {
445  /* SD card does not support 8bit bus.*/
446  return HAL_FAILED;
447  }
448 
449  return HAL_SUCCESS;
450 }

452 /**
453  * @brief Sets bus width for MMC.
454  *
455  * @param[in] sdcp pointer to the @p SDCDriver object
456  *
457  * @return The operation status.
458  * @retval HAL_SUCCESS operation succeeded.
459  * @retval HAL_FAILED operation failed.
460  *
461  * @notapi
462  */
463 static bool mmc_set_bus_width(SDCDriver *sdcp) {
464  uint32_t resp[1];
465  uint32_t cmdarg = mmc_cmd6_construct(MMC_SWITCH_WRITE_BYTE, 183, 0, 0);
466 
467  switch (sdcp->config->bus_width) {
468  case SDC_MODE_1BIT:
469  /* Nothing to do. Bus is already in 1bit mode.*/
470  return HAL_SUCCESS;
471  case SDC_MODE_4BIT:
472  cmdarg = mmc_cmd6_construct(MMC_SWITCH_WRITE_BYTE, 183, 1, 0);
473  break;
474  case SDC_MODE_8BIT:
475  cmdarg = mmc_cmd6_construct(MMC_SWITCH_WRITE_BYTE, 183, 2, 0);
476  break;
477  default:
478  osalDbgAssert(false, "unexpected case");
479  break;
480  }
481 
482  sdc_lld_set_bus_mode(sdcp, sdcp->config->bus_width);
483  if (sdc_lld_send_cmd_short_crc(sdcp, MMCSD_CMD_SWITCH, cmdarg, resp) ||
484  MMCSD_R1_ERROR(resp[0])) {
485  return HAL_FAILED;
486  }
487 
488  return HAL_SUCCESS;
489 }

491 /**
492  * @brief Wait for the card to complete pending operations.
493  *
494  * @param[in] sdcp pointer to the @p SDCDriver object
495  *
496  * @return The operation status.
497  * @retval HAL_SUCCESS operation succeeded.
498  * @retval HAL_FAILED operation failed.
499  *
500  * @notapi
501  */
502 bool _sdc_wait_for_transfer_state(SDCDriver *sdcp) {
503  uint32_t resp[1];
504 
505  while (true) {
506  if (sdc_lld_send_cmd_short_crc(sdcp, MMCSD_CMD_SEND_STATUS,
507  sdcp->rca, resp) ||
508  MMCSD_R1_ERROR(resp[0])) {
509  return HAL_FAILED;
510  }
511 
512  switch (MMCSD_R1_STS(resp[0])) {
513  case MMCSD_STS_TRAN:
514  return HAL_SUCCESS;
515  case MMCSD_STS_DATA:
516  case MMCSD_STS_RCV:
517  case MMCSD_STS_PRG:
518 #if SDC_NICE_WAITING == TRUE
519  osalThreadSleepMilliseconds(1);
520 #endif
521  continue;
522  default:
523  /* The card should have been initialized so any other state is not
524  valid and is reported as an error.*/
525  return HAL_FAILED;
526  }
527  }
528 }

608 /**
609  * @brief Performs the initialization procedure on the inserted card.
610  * @details This function should be invoked when a card is inserted and
611  * brings the driver in the @p BLK_READY state where it is possible
612  * to perform read and write operations.
613  *
614  * @param[in] sdcp pointer to the @p SDCDriver object
615  *
616  * @return The operation status.
617  * @retval HAL_SUCCESS operation succeeded.
618  * @retval HAL_FAILED operation failed.
619  *
620  * @api
621  */
622 bool sdcConnect(SDCDriver *sdcp) {
623  uint32_t resp[1];
624  sdcbusclk_t clk = SDC_CLK_25MHz;
625 
626  osalDbgCheck(sdcp != NULL);
627  osalDbgAssert((sdcp->state == BLK_ACTIVE) || (sdcp->state == BLK_READY),
628  "invalid state");
629 
630  /* Connection procedure in progress.*/
631  sdcp->state = BLK_CONNECTING;
632 
633  /* Card clock initialization.*/
634  sdc_lld_start_clk(sdcp);
635 
636  /* Enforces the initial card state.*/
637  sdc_lld_send_cmd_none(sdcp, MMCSD_CMD_GO_IDLE_STATE, 0);
638 
639  /* Detect card type.*/
640  if (HAL_FAILED == mode_detect(sdcp)) {
641  goto failed;
642  }
643 
644  /* Perform specific initialization procedure.*/
645  if ((sdcp->cardmode & SDC_MODE_CARDTYPE_MASK) == SDC_MODE_CARDTYPE_MMC) {
646  if (HAL_FAILED == mmc_init(sdcp)) {
647  goto failed;
648  }
649  }
650  else {
651  if (HAL_FAILED == sdc_init(sdcp)) {
652  goto failed;
653  }
654  }
655 
656  /* Reads CID.*/
657  if (sdc_lld_send_cmd_long_crc(sdcp, MMCSD_CMD_ALL_SEND_CID, 0, sdcp->cid)) {
658  goto failed;
659  }
660 
661  /* Asks for the RCA.*/
662  if (sdc_lld_send_cmd_short_crc(sdcp, MMCSD_CMD_SEND_RELATIVE_ADDR,
663  0, &sdcp->rca)) {
664  goto failed;
665  }
666 
667  /* Reads CSD.*/
668  if (sdc_lld_send_cmd_long_crc(sdcp, MMCSD_CMD_SEND_CSD,
669  sdcp->rca, sdcp->csd)) {
670  goto failed;
671  }
672 
673  /* Selects the card for operations.*/
674  if (sdc_lld_send_cmd_short_crc(sdcp, MMCSD_CMD_SEL_DESEL_CARD,
675  sdcp->rca, resp)) {
676  goto failed;
677  }
678 
679  /* Switches to high speed.*/
680  if (HAL_SUCCESS != detect_bus_clk(sdcp, &clk)) {
681  goto failed;
682  }
683  sdc_lld_set_data_clk(sdcp, clk);
684 
685  /* Reads extended CSD if needed and possible.*/
686  if (SDC_MODE_CARDTYPE_MMC == (sdcp->cardmode & SDC_MODE_CARDTYPE_MASK)) {
687 
688  /* The card is a MMC, checking if it is a large device.*/
689  if (_mmcsd_get_slice(sdcp->csd, MMCSD_CSD_MMC_CSD_STRUCTURE_SLICE) > 1U) {
690  uint8_t *ext_csd = sdcp->config->scratchpad;
691 
692  /* Size detection requires the buffer.*/
693  if (NULL == ext_csd) {
694  goto failed;
695  }
696

715  if (sdc_lld_send_cmd_short_crc(sdcp, MMCSD_CMD_SET_BLOCKLEN,
716  MMCSD_BLOCK_SIZE, resp) ||
717  MMCSD_R1_ERROR(resp[0])) {
718  goto failed;
719  }

722  switch (sdcp->cardmode & SDC_MODE_CARDTYPE_MASK) {
723  case SDC_MODE_CARDTYPE_SDV11:
724  case SDC_MODE_CARDTYPE_SDV20:

726  goto failed;
727  }

731  goto failed;
732  }
733  break;
734  default:
735  /* Unknown type.*/
736  goto failed;
737  }
738 
739  /* Initialization complete.*/
740  sdcp->state = BLK_READY;
741  return HAL_SUCCESS;

744 failed:
745  sdc_lld_stop_clk(sdcp);
746  sdcp->state = BLK_ACTIVE;
747  return HAL_FAILED;
748 }
