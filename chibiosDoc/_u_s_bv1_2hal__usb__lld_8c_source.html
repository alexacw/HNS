93 /**
94  * @brief Resets the packet memory allocator.
95  *
96  * @param[in] usbp pointer to the @p USBDriver object
97  */
98 static void usb_pm_reset(USBDriver *usbp) {
99 
100  /* The first 64 bytes are reserved for the descriptors table. The effective
101  available RAM for endpoint buffers is just 448 bytes.*/
102  usbp->pmnext = 64;
103 }
104 
105 /**
106  * @brief Resets the packet memory allocator.
107  *
108  * @param[in] usbp pointer to the @p USBDriver object
109  * @param[in] size size of the packet buffer to allocate
110  * @return The packet buffer address.
111  */
112 static uint32_t usb_pm_alloc(USBDriver *usbp, size_t size) {
113  uint32_t next;
114 
115  next = usbp->pmnext;
116  usbp->pmnext += (size + 1) & ~1;
117  osalDbgAssert(usbp->pmnext <= STM32_USB_PMA_SIZE, "PMA overflow");
118  return next;
119 }

121 /**
122  * @brief Reads from a dedicated packet buffer.
123  *
124  * @param[in] ep endpoint number
125  * @param[out] buf buffer where to copy the packet data
126  * @return The size of the receivee packet.
127  *
128  * @notapi
129  */
130 static size_t usb_packet_read_to_buffer(usbep_t ep, uint8_t *buf) {
131  size_t i, n;
132  stm32_usb_descriptor_t *udp = USB_GET_DESCRIPTOR(ep);
133  stm32_usb_pma_t *pmap = USB_ADDR2PTR(udp->RXADDR0);
134 #if STM32_USB_USE_ISOCHRONOUS
135  uint32_t epr = STM32_USB->EPR[ep];
136 
137  /* Double buffering is always enabled for isochronous endpoints, and
138  although we overlap the two buffers for simplicity, we still need
139  to read from the right counter. The DTOG_RX bit indicates the buffer
140  that is currently in use by the USB peripheral, that is, the buffer
141  in which the next received packet will be stored, so we need to
142  read the counter of the OTHER buffer, which is where the last
143  received packet was stored.*/
144  if (EPR_EP_TYPE_IS_ISO(epr) && !(epr & EPR_DTOG_RX))
145  n = (size_t)udp->RXCOUNT1 & RXCOUNT_COUNT_MASK;
146  else
147  n = (size_t)udp->RXCOUNT0 & RXCOUNT_COUNT_MASK;
148 #else
149  n = (size_t)udp->RXCOUNT0 & RXCOUNT_COUNT_MASK;
150 #endif
151 
152  i = n;
153 
154 #if STM32_USB_USE_FAST_COPY
155  while (i >= 16) {
156  uint32_t w;
157 
158  w = *(pmap + 0);
159  *(buf + 0) = (uint8_t)w;
160  *(buf + 1) = (uint8_t)(w >> 8);
161  w = *(pmap + 1);
162  *(buf + 2) = (uint8_t)w;
163  *(buf + 3) = (uint8_t)(w >> 8);
164  w = *(pmap + 2);
165  *(buf + 4) = (uint8_t)w;
166  *(buf + 5) = (uint8_t)(w >> 8);
167  w = *(pmap + 3);
168  *(buf + 6) = (uint8_t)w;
169  *(buf + 7) = (uint8_t)(w >> 8);
170  w = *(pmap + 4);
171  *(buf + 8) = (uint8_t)w;
172  *(buf + 9) = (uint8_t)(w >> 8);
173  w = *(pmap + 5);
174  *(buf + 10) = (uint8_t)w;
175  *(buf + 11) = (uint8_t)(w >> 8);
176  w = *(pmap + 6);
177  *(buf + 12) = (uint8_t)w;
178  *(buf + 13) = (uint8_t)(w >> 8);
179  w = *(pmap + 7);
180  *(buf + 14) = (uint8_t)w;
181  *(buf + 15) = (uint8_t)(w >> 8);
182 
183  i -= 16;
184  buf += 16;
185  pmap += 8;
186  }
187 #endif /* STM32_USB_USE_FAST_COPY */
188 
189  while (i >= 2) {
190  uint32_t w = *pmap++;
191  *buf++ = (uint8_t)w;
192  *buf++ = (uint8_t)(w >> 8);
193  i -= 2;
194  }
195 
196  if (i >= 1) {
197  *buf = (uint8_t)*pmap;
198  }
199 
200  return n;
201 }

203 /**
204  * @brief Writes to a dedicated packet buffer.
205  *
206  * @param[in] ep endpoint number
207  * @param[in] buf buffer where to fetch the packet data
208  * @param[in] n maximum number of bytes to copy. This value must
209  * not exceed the maximum packet size for this endpoint.
210  *
211  * @notapi
212  */
213 static void usb_packet_write_from_buffer(usbep_t ep,
214  const uint8_t *buf,
215  size_t n) {
216  stm32_usb_descriptor_t *udp = USB_GET_DESCRIPTOR(ep);
217  stm32_usb_pma_t *pmap = USB_ADDR2PTR(udp->TXADDR0);
218  int i = (int)n;
219 
220 #if STM32_USB_USE_ISOCHRONOUS
221  uint32_t epr = STM32_USB->EPR[ep];
222 
223  /* Double buffering is always enabled for isochronous endpoints, and
224  although we overlap the two buffers for simplicity, we still need
225  to write to the right counter. The DTOG_TX bit indicates the buffer
226  that is currently in use by the USB peripheral, that is, the buffer
227  from which the next packet will be sent, so we need to write the
228  counter of that buffer.*/
229  if (EPR_EP_TYPE_IS_ISO(epr) && (epr & EPR_DTOG_TX))
230  udp->TXCOUNT1 = (stm32_usb_pma_t)n;
231  else
232  udp->TXCOUNT0 = (stm32_usb_pma_t)n;
233 #else
234  udp->TXCOUNT0 = (stm32_usb_pma_t)n;
235 #endif
236 
237 #if STM32_USB_USE_FAST_COPY
238  while (i >= 16) {
239  uint32_t w;
240 
241  w = *(buf + 0);
242  w |= *(buf + 1) << 8;
243  *(pmap + 0) = (stm32_usb_pma_t)w;
244  w = *(buf + 2);
245  w |= *(buf + 3) << 8;
246  *(pmap + 1) = (stm32_usb_pma_t)w;
247  w = *(buf + 4);
248  w |= *(buf + 5) << 8;
249  *(pmap + 2) = (stm32_usb_pma_t)w;
250  w = *(buf + 6);
251  w |= *(buf + 7) << 8;
252  *(pmap + 3) = (stm32_usb_pma_t)w;
253  w = *(buf + 8);
254  w |= *(buf + 9) << 8;
255  *(pmap + 4) = (stm32_usb_pma_t)w;
256  w = *(buf + 10);
257  w |= *(buf + 11) << 8;
258  *(pmap + 5) = (stm32_usb_pma_t)w;
259  w = *(buf + 12);
260  w |= *(buf + 13) << 8;
261  *(pmap + 6) = (stm32_usb_pma_t)w;
262  w = *(buf + 14);
263  w |= *(buf + 15) << 8;
264  *(pmap + 7) = (stm32_usb_pma_t)w;
265 
266  i -= 16;
267  buf += 16;
268  pmap += 8;
269  }
270 #endif /* STM32_USB_USE_FAST_COPY */
271 
272  while (i > 0) {
273  uint32_t w;
274 
275  w = *buf++;
276  w |= *buf++ << 8;
277  *pmap++ = (stm32_usb_pma_t)w;
278  i -= 2;
279  }
280 }

282 /**
283  * @brief Common ISR code, serves the EP-related interrupts.
284  *
285  * @param[in] usbp pointer to the @p USBDriver object
286  * @param[in] ep endpoint number
287  *
288  * @notapi
289  */
290 static void usb_serve_endpoints(USBDriver *usbp, uint32_t ep) {
291  size_t n;
292  uint32_t epr = STM32_USB->EPR[ep];
293  const USBEndpointConfig *epcp = usbp->epc[ep];
294 
295  if (epr & EPR_CTR_TX) {
296  /* IN endpoint, transmission.*/
297  USBInEndpointState *isp = epcp->in_state;
298 
299  EPR_CLEAR_CTR_TX(ep);
300 
301  isp->txcnt += isp->txlast;
302  n = isp->txsize - isp->txcnt;
303  if (n > 0) {
304  /* Transfer not completed, there are more packets to send.*/
305  if (n > epcp->in_maxsize)
306  n = epcp->in_maxsize;
307 
308  /* Writes the packet from the defined buffer.*/
309  isp->txbuf += isp->txlast;
310  isp->txlast = n;
311  usb_packet_write_from_buffer(ep, isp->txbuf, n);
312 
313  /* Starting IN operation.*/
314  EPR_SET_STAT_TX(ep, EPR_STAT_TX_VALID);
315  }
316  else {
317  /* Transfer completed, invokes the callback.*/
318  _usb_isr_invoke_in_cb(usbp, ep);
319  }
320  }
321  if (epr & EPR_CTR_RX) {
322  /* OUT endpoint, receive.*/
323 
324  EPR_CLEAR_CTR_RX(ep);
325 
326  if (epr & EPR_SETUP) {
327  /* Setup packets handling, setup packets are handled using a
328  specific callback.*/
329  _usb_isr_invoke_setup_cb(usbp, ep);
330  }
331  else {
332  USBOutEndpointState *osp = epcp->out_state;
333 
334  /* Reads the packet into the defined buffer.*/
335  n = usb_packet_read_to_buffer(ep, osp->rxbuf);
336  osp->rxbuf += n;
337 
338  /* Transaction data updated.*/
339  osp->rxcnt += n;
340  osp->rxsize -= n;
341  osp->rxpkts -= 1;
342 
343  /* The transaction is completed if the specified number of packets
344  has been received or the current packet is a short packet.*/
345  if ((n < epcp->out_maxsize) || (osp->rxpkts == 0)) {
346  /* Transfer complete, invokes the callback.*/
347  _usb_isr_invoke_out_cb(usbp, ep);
348  }
349  else {
350  /* Transfer not complete, there are more packets to receive.*/
351  EPR_SET_STAT_RX(ep, EPR_STAT_RX_VALID);
352  }
353  }
354  }
355 }

361 #if STM32_USB_USE_USB1 || defined(__DOXYGEN__)

364 /**
365  * @brief USB high priority interrupt handler.
366  *
367  * @isr
368  */
369 OSAL_IRQ_HANDLER(STM32_USB1_HP_HANDLER) {
370  uint32_t istr;
371  USBDriver *usbp = &USBD1;
372 
373  OSAL_IRQ_PROLOGUE();
374 
375  /* Endpoint events handling.*/
376  istr = STM32_USB->ISTR;
377  while (istr & ISTR_CTR) {
378  usb_serve_endpoints(usbp, istr & ISTR_EP_ID_MASK);
379  istr = STM32_USB->ISTR;
380  }
381 
382  OSAL_IRQ_EPILOGUE();
383 }
384 #endif /* STM32_USB_USE_ISOCHRONOUS */
385 #endif /* STM32_USB1_LP_NUMBER != STM32_USB1_HP_NUMBER */

387 /**
388  * @brief USB low priority interrupt handler.
389  *
390  * @isr
391  */
392 OSAL_IRQ_HANDLER(STM32_USB1_LP_HANDLER) {
393  uint32_t istr;
394  USBDriver *usbp = &USBD1;
395 
396  OSAL_IRQ_PROLOGUE();
397 
398  istr = STM32_USB->ISTR;
399 
400  /* USB bus reset condition handling.*/
401  if (istr & ISTR_RESET) {
402  STM32_USB->ISTR = ~ISTR_RESET;
403 
404  _usb_reset(usbp);
405  }
406 
407  /* USB bus SUSPEND condition handling.*/
408  if (istr & ISTR_SUSP) {
409  STM32_USB->CNTR |= CNTR_FSUSP;
410 #if STM32_USB_LOW_POWER_ON_SUSPEND
411  STM32_USB->CNTR |= CNTR_LP_MODE;
412 #endif
413  STM32_USB->ISTR = ~ISTR_SUSP;
414 
415  _usb_suspend(usbp);
416  }
417 
418  /* USB bus WAKEUP condition handling.*/
419  if (istr & ISTR_WKUP) {
420  uint32_t fnr = STM32_USB->FNR;
421  if (!(fnr & FNR_RXDP)) {
422  STM32_USB->CNTR &= ~CNTR_FSUSP;
423 
424  _usb_wakeup(usbp);
425  }
426 #if STM32_USB_LOW_POWER_ON_SUSPEND
427  else {
428  /* Just noise, going back in SUSPEND mode, reference manual 22.4.5,
429  table 169.*/
430  STM32_USB->CNTR |= CNTR_LP_MODE;
431  }
432 #endif
433  STM32_USB->ISTR = ~ISTR_WKUP;
434  }
435 
436  /* SOF handling.*/
437  if (istr & ISTR_SOF) {
438  _usb_isr_invoke_sof_cb(usbp);
439  STM32_USB->ISTR = ~ISTR_SOF;
440  }
441 
442  /* Endpoint events handling.*/
443  while (istr & ISTR_CTR) {
444  usb_serve_endpoints(usbp, istr & ISTR_EP_ID_MASK);
445  istr = STM32_USB->ISTR;
446  }
447 
448  OSAL_IRQ_EPILOGUE();
449 }
450 #endif /* STM32_USB_USE_USB1 */

523 /**
524  * @brief USB low level reset routine.
525  *
526  * @param[in] usbp pointer to the @p USBDriver object
527  *
528  * @notapi
529  */
530 void usb_lld_reset(USBDriver *usbp) {
531  uint32_t cntr;
532 
533  /* Post reset initialization.*/
534  STM32_USB->BTABLE = BTABLE_ADDR;
535  STM32_USB->ISTR = 0;
536  STM32_USB->DADDR = DADDR_EF;
537  cntr = /*CNTR_ESOFM | */ CNTR_RESETM | CNTR_SUSPM |
538  CNTR_WKUPM | /*CNTR_ERRM | CNTR_PMAOVRM |*/ CNTR_CTRM;
539  /* The SOF interrupt is only enabled if a callback is defined for
540  this service because it is an high rate source.*/
541  if (usbp->config->sof_cb != NULL)
542  cntr |= CNTR_SOFM;
543  STM32_USB->CNTR = cntr;
544 
545  /* Resets the packet memory allocator.*/
546  usb_pm_reset(usbp);
547 
548  /* EP0 initialization.*/
549  usbp->epc[0] = &ep0config;
550  usb_lld_init_endpoint(usbp, 0);
551 }

565 /**
566  * @brief Enables an endpoint.
567  *
568  * @param[in] usbp pointer to the @p USBDriver object
569  * @param[in] ep endpoint number
570  *
571  * @notapi
572  */
573 void usb_lld_init_endpoint(USBDriver *usbp, usbep_t ep) {
574  uint16_t epr;
575  stm32_usb_descriptor_t *dp;
576  const USBEndpointConfig *epcp = usbp->epc[ep];
577 
578  /* Setting the endpoint type. Note that isochronous endpoints cannot be
579  bidirectional because it uses double buffering and both transmit and
580  receive descriptor fields are used for either direction.*/
581  switch (epcp->ep_mode & USB_EP_MODE_TYPE) {
582  case USB_EP_MODE_TYPE_ISOC:
583 #if STM32_USB_USE_ISOCHRONOUS
584  osalDbgAssert((epcp->in_state == NULL) || (epcp->out_state == NULL),
585  "isochronous EP cannot be IN and OUT");
586  epr = EPR_EP_TYPE_ISO;
587  break;
588 #else
589  osalDbgAssert(false, "isochronous support disabled");
590 #endif
591  /* Falls through.*/
592  case USB_EP_MODE_TYPE_BULK:
593  epr = EPR_EP_TYPE_BULK;
594  break;
595  case USB_EP_MODE_TYPE_INTR:
596  epr = EPR_EP_TYPE_INTERRUPT;
597  break;
598  default:
599  epr = EPR_EP_TYPE_CONTROL;
600  }
601 
602  dp = USB_GET_DESCRIPTOR(ep);
603 
604  /* IN endpoint handling.*/
605  if (epcp->in_state != NULL) {
606  dp->TXCOUNT0 = 0;
607  dp->TXADDR0 = usb_pm_alloc(usbp, epcp->in_maxsize);
608 
609 #if STM32_USB_USE_ISOCHRONOUS
610  if (epr == EPR_EP_TYPE_ISO) {
611  epr |= EPR_STAT_TX_VALID;
612  dp->TXCOUNT1 = dp->TXCOUNT0;
613  dp->TXADDR1 = dp->TXADDR0; /* Both buffers overlapped.*/
614  }
615  else {
616  epr |= EPR_STAT_TX_NAK;
617  }
618 #else
619  epr |= EPR_STAT_TX_NAK;
620 #endif
621  }
622 
623  /* OUT endpoint handling.*/
624  if (epcp->out_state != NULL) {
625  uint16_t nblocks;
626 
627  /* Endpoint size and address initialization.*/
628  if (epcp->out_maxsize > 62)
629  nblocks = (((((epcp->out_maxsize - 1) | 0x1f) + 1) / 32) << 10) |
630  0x8000;
631  else
632  nblocks = ((((epcp->out_maxsize - 1) | 1) + 1) / 2) << 10;
633  dp->RXCOUNT0 = nblocks;
634  dp->RXADDR0 = usb_pm_alloc(usbp, epcp->out_maxsize);
635 
636 #if STM32_USB_USE_ISOCHRONOUS
637  if (epr == EPR_EP_TYPE_ISO) {
638  epr |= EPR_STAT_RX_VALID;
639  dp->RXCOUNT1 = dp->RXCOUNT0;
640  dp->RXADDR1 = dp->RXADDR0; /* Both buffers overlapped.*/
641  }
642  else {
643  epr |= EPR_STAT_RX_NAK;
644  }
645 #else
646  epr |= EPR_STAT_RX_NAK;
647 #endif
648  }
649 
650  /* EPxR register setup.*/
651  EPR_SET(ep, epr | ep);
652  EPR_TOGGLE(ep, epr);
653 }
