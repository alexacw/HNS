152 /**
153  * @brief Shared end/half-of-tx service routine.
154  *
155  * @param[in] dacp pointer to the @p DACDriver object
156  * @param[in] flags pre-shifted content of the ISR register
157  */
158 static void dac_lld_serve_tx_interrupt(DACDriver *dacp, uint32_t flags) {
159 
160  if ((flags & (STM32_DMA_ISR_TEIF | STM32_DMA_ISR_DMEIF)) != 0) {
161  /* DMA errors handling.*/
162  _dac_isr_error_code(dacp, DAC_ERR_DMAFAILURE);
163  }
164  else {
165  if ((flags & STM32_DMA_ISR_HTIF) != 0) {
166  /* Half transfer processing.*/
167  _dac_isr_half_code(dacp);
168  }
169  if ((flags & STM32_DMA_ISR_TCIF) != 0) {
170  /* Transfer complete processing.*/
171  _dac_isr_full_code(dacp);
172  }
173  }
174 }

212 /**
213  * @brief Configures and activates the DAC peripheral.
214  *
215  * @param[in] dacp pointer to the @p DACDriver object
216  *
217  * @notapi
218  */
219 void dac_lld_start(DACDriver *dacp) {
220 
221  /* If the driver is in DAC_STOP state then a full initialization is
222  required.*/
223  if (dacp->state == DAC_STOP) {
224  dacchannel_t channel = 0;
225 
226  /* Enabling the clock source.*/
227 #if STM32_DAC_USE_DAC1_CH1
228  if (&DACD1 == dacp) {
229  rccEnableDAC1(true);
230  }
231 #endif
232 
233 #if STM32_DAC_USE_DAC1_CH2
234  if (&DACD2 == dacp) {
235  rccEnableDAC1(true);
236  channel = 1;
237  }
238 #endif
239 
240 #if STM32_DAC_USE_DAC2_CH1
241  if (&DACD3 == dacp) {
242  rccEnableDAC2(true);
243  }
244 #endif
245 
246 #if STM32_DAC_USE_DAC2_CH2
247  if (&DACD4 == dacp) {
248  rccEnableDAC2(true);
249  channel = 1;
250  }
251 #endif
252  /* Enabling DAC in SW triggering mode initially, initializing data to
253  zero.*/
254 #if STM32_DAC_DUAL_MODE == FALSE
255  dacp->params->dac->CR &= dacp->params->regmask;
256  dacp->params->dac->CR |= (DAC_CR_EN1 | dacp->config->cr) << dacp->params->regshift;
257  dac_lld_put_channel(dacp, channel, dacp->config->init);
258 #else
259  if ((dacp->config->datamode == DAC_DHRM_12BIT_RIGHT_DUAL) ||
260  (dacp->config->datamode == DAC_DHRM_12BIT_LEFT_DUAL) ||
261  (dacp->config->datamode == DAC_DHRM_8BIT_RIGHT_DUAL)) {
262  dacp->params->dac->CR = DAC_CR_EN2 | (dacp->config->cr << 16) | DAC_CR_EN1 | dacp->config->cr;
263  dac_lld_put_channel(dacp, 1U, dacp->config->init);
264  }
265  else {
266  dacp->params->dac->CR = DAC_CR_EN1 | dacp->config->cr;
267  }
268  dac_lld_put_channel(dacp, channel, dacp->config->init);
269 #endif
270  }
271 }

322 /**
323  * @brief Outputs a value directly on a DAC channel.
324  *
325  * @param[in] dacp pointer to the @p DACDriver object
326  * @param[in] channel DAC channel number
327  * @param[in] sample value to be output
328  *
329  * @api
330  */
331 void dac_lld_put_channel(DACDriver *dacp,
332  dacchannel_t channel,
333  dacsample_t sample) {
334 
335  switch (dacp->config->datamode) {
336  case DAC_DHRM_12BIT_RIGHT:
337 #if STM32_DAC_DUAL_MODE
338  case DAC_DHRM_12BIT_RIGHT_DUAL:
339 #endif
340  if (channel == 0U) {
341 #if STM32_DAC_DUAL_MODE
342  dacp->params->dac->DHR12R1 = (uint32_t)sample;
343 #else
344  *(&dacp->params->dac->DHR12R1 + dacp->params->dataoffset) = (uint32_t)sample;
345 #endif
346  }
347 #if (STM32_HAS_DAC1_CH2 || STM32_HAS_DAC2_CH2)
348  else {
349  dacp->params->dac->DHR12R2 = (uint32_t)sample;
350  }
351 #endif
352  break;
353  case DAC_DHRM_12BIT_LEFT:
354 #if STM32_DAC_DUAL_MODE
355  case DAC_DHRM_12BIT_LEFT_DUAL:
356 #endif
357  if (channel == 0U) {
358 #if STM32_DAC_DUAL_MODE
359  dacp->params->dac->DHR12L1 = (uint32_t)sample;
360 #else
361  *(&dacp->params->dac->DHR12L1 + dacp->params->dataoffset) = (uint32_t)sample;
362 #endif
363  }
364 #if (STM32_HAS_DAC1_CH2 || STM32_HAS_DAC2_CH2)
365  else {
366  dacp->params->dac->DHR12L2 = (uint32_t)sample;
367  }
368 #endif
369  break;
370  case DAC_DHRM_8BIT_RIGHT:
371 #if STM32_DAC_DUAL_MODE
372  case DAC_DHRM_8BIT_RIGHT_DUAL:
373 #endif
374  if (channel == 0U) {
375 #if STM32_DAC_DUAL_MODE
376  dacp->params->dac->DHR8R1 = (uint32_t)sample;
377 #else
378  *(&dacp->params->dac->DHR8R1 + dacp->params->dataoffset) = (uint32_t)sample;
379 #endif
380  }
381 #if (STM32_HAS_DAC1_CH2 || STM32_HAS_DAC2_CH2)
382  else {
383  dacp->params->dac->DHR8R2 = (uint32_t)sample;
384  }
385 #endif
386  break;
387  default:
388  osalDbgAssert(false, "unexpected DAC mode");
389  break;
390  }

393 /**
394  * @brief Starts a DAC conversion.
395  * @details Starts an asynchronous conversion operation.
396  * @note In @p DAC_DHRM_8BIT_RIGHT mode the parameters passed to the
397  * callback are wrong because two samples are packed in a single
398  * dacsample_t element. This will not be corrected, do not rely
399  * on those parameters.
400  * @note In @p DAC_DHRM_8BIT_RIGHT_DUAL mode two samples are treated
401  * as a single 16 bits sample and packed into a single dacsample_t
402  * element. The num_channels must be set to one in the group
403  * conversion configuration structure.
404  *
405  * @param[in] dacp pointer to the @p DACDriver object
406  *
407  * @notapi
408  */
409 void dac_lld_start_conversion(DACDriver *dacp) {
410  uint32_t n, cr, dmamode;
411 
412  /* Number of DMA operations per buffer.*/
413  n = dacp->depth * dacp->grpp->num_channels;
414 
415  /* Allocating the DMA channel.*/
416  bool b = dmaStreamAllocate(dacp->params->dma, dacp->params->dmairqprio,
417  (stm32_dmaisr_t)dac_lld_serve_tx_interrupt,
418  (void *)dacp);
419  osalDbgAssert(!b, "stream already allocated");
420 
421  /* DMA settings depend on the chosed DAC mode.*/
422  switch (dacp->config->datamode) {
423  /* Sets the DAC data register */
424  case DAC_DHRM_12BIT_RIGHT:
425  osalDbgAssert(dacp->grpp->num_channels == 1, "invalid number of channels");
426 
427  dmaStreamSetPeripheral(dacp->params->dma, &dacp->params->dac->DHR12R1 +
428  dacp->params->dataoffset);
429  dmamode = dacp->params->dmamode |
430  STM32_DMA_CR_PSIZE_HWORD | STM32_DMA_CR_MSIZE_HWORD;
431  break;
432  case DAC_DHRM_12BIT_LEFT:
433  osalDbgAssert(dacp->grpp->num_channels == 1, "invalid number of channels");
434 
435  dmaStreamSetPeripheral(dacp->params->dma, &dacp->params->dac->DHR12L1 +
436  dacp->params->dataoffset);
437  dmamode = dacp->params->dmamode |
438  STM32_DMA_CR_PSIZE_HWORD | STM32_DMA_CR_MSIZE_HWORD;
439  break;
440  case DAC_DHRM_8BIT_RIGHT:
441  osalDbgAssert(dacp->grpp->num_channels == 1, "invalid number of channels");
442 
443  dmaStreamSetPeripheral(dacp->params->dma, &dacp->params->dac->DHR8R1 +
444  dacp->params->dataoffset);
445  dmamode = dacp->params->dmamode |
446  STM32_DMA_CR_PSIZE_BYTE | STM32_DMA_CR_MSIZE_BYTE;
447 
448  /* In this mode the size of the buffer is halved because two samples
449  packed in a single dacsample_t element.*/
450  n = (n + 1) / 2;
451  break;
452 #if STM32_DAC_DUAL_MODE == TRUE
453  case DAC_DHRM_12BIT_RIGHT_DUAL:
454  osalDbgAssert(dacp->grpp->num_channels == 2, "invalid number of channels");
455 
456  dmaStreamSetPeripheral(dacp->params->dma, &dacp->params->dac->DHR12RD);
457  dmamode = dacp->params->dmamode |
458  STM32_DMA_CR_PSIZE_WORD | STM32_DMA_CR_MSIZE_WORD;
459  n /= 2;
460  break;
461  case DAC_DHRM_12BIT_LEFT_DUAL:
462  osalDbgAssert(dacp->grpp->num_channels == 2, "invalid number of channels");
463 
464  dmaStreamSetPeripheral(dacp->params->dma, &dacp->params->dac->DHR12LD);
465  dmamode = dacp->params->dmamode |
466  STM32_DMA_CR_PSIZE_WORD | STM32_DMA_CR_MSIZE_WORD;
467  n /= 2;
468  break;
469  case DAC_DHRM_8BIT_RIGHT_DUAL:
470  osalDbgAssert(dacp->grpp->num_channels == 1, "invalid number of channels");
471 
472  dmaStreamSetPeripheral(dacp->params->dma, &dacp->params->dac->DHR8RD);
473  dmamode = dacp->params->dmamode |
474  STM32_DMA_CR_PSIZE_HWORD | STM32_DMA_CR_MSIZE_HWORD;
475  n /= 2;
476  break;
477 #endif
478  default:
479  osalDbgAssert(false, "unexpected DAC mode");
480  return;
481  }
482 
483  dmaStreamSetMemory0(dacp->params->dma, dacp->samples);
484  dmaStreamSetTransactionSize(dacp->params->dma, n);
485  dmaStreamSetMode(dacp->params->dma, dmamode |
486  STM32_DMA_CR_DMEIE | STM32_DMA_CR_TEIE |
487  STM32_DMA_CR_HTIE | STM32_DMA_CR_TCIE);
488  dmaStreamEnable(dacp->params->dma);
489 
490  /* DAC configuration.*/
491 #if STM32_DAC_DUAL_MODE == FALSE
492  cr = DAC_CR_DMAEN1 | (dacp->grpp->trigger << 3) | DAC_CR_TEN1 | DAC_CR_EN1 | dacp->config->cr;
493  dacp->params->dac->CR &= dacp->params->regmask;
494  dacp->params->dac->CR |= cr << dacp->params->regshift;
495 #else
496  dacp->params->dac->CR = 0;
497  cr = DAC_CR_DMAEN1 | (dacp->grpp->trigger << 3) | DAC_CR_TEN1 | DAC_CR_EN1 | dacp->config->cr
498  | (dacp->grpp->trigger << 19) | DAC_CR_TEN2 | DAC_CR_EN2 | (dacp->config->cr << 16);
499  dacp->params->dac->CR = cr;
500 #endif
501 }

503 /**
504  * @brief Stops an ongoing conversion.
505  * @details This function stops the currently ongoing conversion and returns
506  * the driver in the @p DAC_READY state. If there was no conversion
507  * being processed then the function does nothing.
508  *
509  * @param[in] dacp pointer to the @p DACDriver object
510  *
511  * @iclass
512  */
513 void dac_lld_stop_conversion(DACDriver *dacp) {
514 
515  /* DMA channel disabled and released.*/
516  dmaStreamDisable(dacp->params->dma);
517  dmaStreamRelease(dacp->params->dma);
518 
519 #if STM32_DAC_DUAL_MODE == FALSE
520  dacp->params->dac->CR &= dacp->params->regmask;
521  dacp->params->dac->CR |= (DAC_CR_EN1 | dacp->config->cr) << dacp->params->regshift;
522 #else
523  if ((dacp->config->datamode == DAC_DHRM_12BIT_RIGHT_DUAL) ||
524  (dacp->config->datamode == DAC_DHRM_12BIT_LEFT_DUAL) ||
525  (dacp->config->datamode == DAC_DHRM_8BIT_RIGHT_DUAL)) {
526  dacp->params->dac->CR = DAC_CR_EN2 | (dacp->config->cr << 16) | DAC_CR_EN1 | dacp->config->cr;
527  }
528  else {
529  dacp->params->dac->CR = DAC_CR_EN1 | dacp->config->cr;
530  }
531 #endif
532 }
