323 /**
324  * @brief Programs the stream mode settings.
325  * @note This function can be invoked in both ISR or thread context.
326  * @pre The stream must have been allocated using @p dmaStreamAllocate().
327  * @post After use the stream can be released using @p dmaStreamRelease().
328  *
329  * @param[in] dmastp pointer to a stm32_dma_stream_t structure
330  * @param[in] mode value to be written in the CCR register
331  *
332  * @special
333  */
334 #if STM32_DMA_SUPPORTS_CSELR || defined(__DOXYGEN__)
335 #define dmaStreamSetMode(dmastp, mode) { \
336  uint32_t cselr = *(dmastp)->cselr; \
337  cselr &= ~(0x0000000FU << (dmastp)->shift); \

396 /**
397  * @brief Starts a memory to memory operation using the specified stream.
398  * @note The default transfer data mode is "byte to byte" but it can be
399  * changed by specifying extra options in the @p mode parameter.
400  * @pre The stream must have been allocated using @p dmaStreamAllocate().
401  * @post After use the stream can be released using @p dmaStreamRelease().
402  *
403  * @param[in] dmastp pointer to a stm32_dma_stream_t structure
404  * @param[in] mode value to be written in the CCR register, this value
405  * is implicitly ORed with:
406  * - @p STM32_DMA_CR_MINC
407  * - @p STM32_DMA_CR_PINC
408  * - @p STM32_DMA_CR_DIR_M2M
409  * - @p STM32_DMA_CR_EN
410  * .
411  * @param[in] src source address
412  * @param[in] dst destination address
413  * @param[in] n number of data units to copy
414  */
415 #define dmaStartMemCopy(dmastp, mode, src, dst, n) { \
416  dmaStreamSetPeripheral(dmastp, src); \
417  dmaStreamSetMemory0(dmastp, dst); \
418  dmaStreamSetTransactionSize(dmastp, n); \
419  dmaStreamSetMode(dmastp, (mode) | \
420  STM32_DMA_CR_MINC | STM32_DMA_CR_PINC | \
421  STM32_DMA_CR_DIR_M2M | STM32_DMA_CR_EN); \
422 }

424 /**
425  * @brief Polled wait for DMA transfer end.
426  * @pre The stream must have been allocated using @p dmaStreamAllocate().
427  * @post After use the stream can be released using @p dmaStreamRelease().
428  *
429  * @param[in] dmastp pointer to a stm32_dma_stream_t structure

431 #define dmaWaitCompletion(dmastp) { \
432  while ((dmastp)->channel->CNDTR > 0U) \

437 /**
438  * @brief Serves a DMA IRQ.
439  *
440  * @param[in] dmastp pointer to a stm32_dma_stream_t structure
441  */
442 #define dmaServeInterrupt(dmastp) { \
443  uint32_t flags; \
444  uint32_t idx = (dmastp)->selfindex; \
445  \
446  flags = ((dmastp)->dma->ISR >> (dmastp)->shift) & STM32_DMA_ISR_MASK; \
447  if (flags & (dmastp)->channel->CCR) { \
448  (dmastp)->dma->IFCR = flags << (dmastp)->shift; \
449  if (_stm32_dma_isr_redir[idx].dma_func) { \
450  _stm32_dma_isr_redir[idx].dma_func(_stm32_dma_isr_redir[idx].dma_param, flags); \
451  } \
452  } \
453 }
