54 /**
55  * @brief Enables the clock of one or more peripheral on the APB1 bus.
56  * @note The @p lp parameter is ignored in this family.
57  *
58  * @param[in] mask APB1 peripherals mask
59  * @param[in] lp low power enable flag
60  *
61  * @api
62  */
63 #define rccEnableAPB1(mask, lp) { \
64  RCC->APB1ENR |= (mask); \
65 }
66 
67 /**
68  * @brief Disables the clock of one or more peripheral on the APB1 bus.
69  * @note The @p lp parameter is ignored in this family.
70  *
71  * @param[in] mask APB1 peripherals mask
72  *
73  * @api
74  */
75 #define rccDisableAPB1(mask) { \
76  RCC->APB1ENR &= ~(mask); \
77 }
78 
79 /**
80  * @brief Resets one or more peripheral on the APB1 bus.
81  *
82  * @param[in] mask APB1 peripherals mask
83  *
84  * @api
85  */
86 #define rccResetAPB1(mask) { \
87  RCC->APB1RSTR |= (mask); \
88  RCC->APB1RSTR = 0; \
89 }
