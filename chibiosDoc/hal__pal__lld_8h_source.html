73 /**
74  * @brief Forms a line identifier.
75  * @details A port/pad pair are encoded into an @p ioline_t type. The encoding
76  * of this type is platform-dependent.
77  * @note In this driver the pad number is encoded in the lower 4 bits of
78  * the GPIO address which are guaranteed to be zero.
79  */
80 #define PAL_LINE(port, pad) \
81  ((ioline_t)((uint32_t)(port)) | ((uint32_t)(pad)))

316 /**
317  * @brief Writes a group of bits.
318  * @details This function is implemented by writing the GPIO BSRR register, the
319  * implementation has no side effects.
320  * @note Writing on pads programmed as pull-up or pull-down has the side
321  * effect to modify the resistor setting because the output latched
322  * data is used for the resistor selection.
323  *
324  * @param[in] port port identifier
325  * @param[in] mask group mask
326  * @param[in] offset the group bit offset within the port
327  * @param[in] bits bits to be written. Values exceeding the group
328  * width are masked.
329  *
330  * @notapi
331  */
332 #define pal_lld_writegroup(port, mask, offset, bits) \
333  ((port)->BSRR = ((~(bits) & (mask)) << (16 + (offset))) | \
334  (((bits) & (mask)) << (offset)))
