325 /**
326  * @brief Structure representing an CAN driver.
327  */
328 struct CANDriver {
329  /**
330  * @brief Driver state.
331  */
332  canstate_t state;
333  /**
334  * @brief Current configuration data.
335  */
336  const CANConfig *config;
337  /**
338  * @brief Transmission threads queue.
339  */
340  threads_queue_t txqueue;
341  /**
342  * @brief Receive threads queue.
343  */
344  threads_queue_t rxqueue;
345 #if !defined(CAN_ENFORCE_USE_CALLBACKS)
346  /**
347  * @brief One or more frames become available.
348  * @note After broadcasting this event it will not be broadcasted again
349  * until the received frames queue has been completely emptied. It
350  * is <b>not</b> broadcasted for each received frame. It is
351  * responsibility of the application to empty the queue by
352  * repeatedly invoking @p canReceive() when listening to this event.
353  * This behavior minimizes the interrupt served by the system
354  * because CAN traffic.
355  * @note The flags associated to the listeners will indicate which
356  * receive mailboxes become non-empty.
357  */
358  event_source_t rxfull_event;
359  /**
360  * @brief One or more transmission mailbox become available.
361  * @note The flags associated to the listeners will indicate which
362  * transmit mailboxes become empty.
363  * @note The upper 16 bits are transmission error flags associated
364  * to the transmit mailboxes.
365  */
366  event_source_t txempty_event;
367  /**
368  * @brief A CAN bus error happened.
369  * @note The flags associated to the listeners will indicate that
370  * receive error(s) have occurred.
371  * @note In this implementation the upper 16 bits are filled with the
372  * unprocessed content of the ESR register.
373  */
374  event_source_t error_event;
375 #if CAN_USE_SLEEP_MODE || defined (__DOXYGEN__)
376  /**
377  * @brief Entering sleep state event.
378  */
379  event_source_t sleep_event;
380  /**
381  * @brief Exiting sleep state event.
382  */
383  event_source_t wakeup_event;
384 #endif /* CAN_USE_SLEEP_MODE */
385 #else /* defined(CAN_ENFORCE_USE_CALLBACKS) */
386  /**
387  * @brief One or more frames become available.
388  * @note After calling this function it will not be called again
389  * until the received frames queue has been completely emptied. It
390  * is <b>not</b> called for each received frame. It is
391  * responsibility of the application to empty the queue by
392  * repeatedly invoking @p chTryReceiveI().
393  * This behavior minimizes the interrupt served by the system
394  * because CAN traffic.
395  */
396  can_callback_t rxfull_cb;
397  /**
398  * @brief One or more transmission mailbox become available.
399  * @note The flags associated to the callback will indicate which
400  * transmit mailboxes become empty.
401  */
402  can_callback_t txempty_cb;
403  /**
404  * @brief A CAN bus error happened.
405  */
406  can_callback_t error_cb;
407 #if (CAN_USE_SLEEP_MODE == TRUE) || defined (__DOXYGEN__)
408  /**
409  * @brief Exiting sleep state.
410  */
411  can_callback_t wakeup_cb;
412 #endif
413 #endif
414  /* End of the mandatory fields.*/
415  /**
416  * @brief Pointer to the CAN registers.
417  */
418  CAN_TypeDef *can;
419 };
