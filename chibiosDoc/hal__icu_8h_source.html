171 /**

173  *
174  * @param[in] icup pointer to the @p ICUDriver object
175  *
176  * @notapi
177  */
178 #define _icu_isr_invoke_width_cb(icup) do { \
179  if (((icup)->state == ICU_ACTIVE) && \
180  ((icup)->config->width_cb != NULL)) \
181  (icup)->config->width_cb(icup); \
182 } while (0)
183 
184 /**
185  * @brief Common ISR code, ICU period event.
186  * @note A period event brings the driver into the @p ICU_ACTIVE state.
187  *
188  * @param[in] icup pointer to the @p ICUDriver object
189  *
190  * @notapi
191  */
192 #define _icu_isr_invoke_period_cb(icup) do { \
193  if (((icup)->state == ICU_ACTIVE) && \
194  ((icup)->config->period_cb != NULL)) \
195  (icup)->config->period_cb(icup); \
196  (icup)->state = ICU_ACTIVE; \
197 } while (0)
198
