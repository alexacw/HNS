175 /**
176  * @brief Conversion group configuration structure.
177  * @details This implementation-dependent structure describes a conversion
178  * operation.
179  * @note The use of this configuration structure requires knowledge of
180  * STM32 ADC cell registers interface, please refer to the STM32
181  * reference manual for details.
182  */
183 typedef struct {
184  /**
185  * @brief Enables the circular buffer mode for the group.
186  */
187  bool circular;
188  /**
189  * @brief Number of the analog channels belonging to the conversion group.
190  */
191  adc_channels_num_t num_channels;
192  /**
193  * @brief Callback function associated to the group or @p NULL.
194  */
195  adccallback_t end_cb;
196  /**
197  * @brief Error callback or @p NULL.
198  */
199  adcerrorcallback_t error_cb;
200  /* End of the mandatory fields.*/
201  /**
202  * @brief ADC CR1 register initialization data.
203  * @note All the required bits must be defined into this field except
204  * @p ADC_CR1_SCAN that is enforced inside the driver.
205  */
206  uint32_t cr1;
207  /**
208  * @brief ADC CR2 register initialization data.
209  * @note All the required bits must be defined into this field except
210  * @p ADC_CR2_DMA, @p ADC_CR2_CONT and @p ADC_CR2_ADON that are
211  * enforced inside the driver.
212  */
213  uint32_t cr2;
214  /**
215  * @brief ADC SMPR1 register initialization data.
216  * @details In this field must be specified the sample times for channels
217  * 10...17.
218  */
219  uint32_t smpr1;
220  /**
221  * @brief ADC SMPR2 register initialization data.
222  * @details In this field must be specified the sample times for channels
223  * 0...9.
224  */
225  uint32_t smpr2;
226  /**
227  * @brief ADC SQR1 register initialization data.
228  * @details Conversion group sequence 13...16 + sequence length.
229  */
230  uint32_t sqr1;
231  /**
232  * @brief ADC SQR2 register initialization data.
233  * @details Conversion group sequence 7...12.
234  */
235  uint32_t sqr2;
236  /**
237  * @brief ADC SQR3 register initialization data.
238  * @details Conversion group sequence 1...6.
239  */
240  uint32_t sqr3;
241 } ADCConversionGroup;
