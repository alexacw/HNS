258 /**
259  * @brief STM32 Serial Driver configuration structure.
260  * @details An instance of this structure must be passed to @p sdStart()
261  * in order to configure and start a serial driver operations.
262  * @note This structure content is architecture dependent, each driver
263  * implementation defines its own version and the custom static
264  * initializers.
265  */
266 typedef struct {
267  /**
268  * @brief Bit rate.
269  */
270  uint32_t speed;
271  /* End of the mandatory fields.*/
272  /**
273  * @brief Initialization value for the CR1 register.
274  */
275  uint16_t cr1;
276  /**
277  * @brief Initialization value for the CR2 register.
278  */
279  uint16_t cr2;
280  /**
281  * @brief Initialization value for the CR3 register.
282  */
283  uint16_t cr3;
284 } SerialConfig;
285 
286 /**
287  * @brief @p SerialDriver specific data.
288  */
289 #define _serial_driver_data \
290  _base_asynchronous_channel_data \
291  /* Driver state.*/ \
292  sdstate_t state; \
293  /* Input queue.*/ \
294  input_queue_t iqueue; \
295  /* Output queue.*/ \
296  output_queue_t oqueue; \
297  /* Input circular buffer.*/ \
298  uint8_t ib[SERIAL_BUFFERS_SIZE]; \
299  /* Output circular buffer.*/ \
300  uint8_t ob[SERIAL_BUFFERS_SIZE]; \
301  /* End of the mandatory fields.*/ \
302  /* Pointer to the USART registers block.*/ \
303  USART_TypeDef *usart; \
304  /* Mask to be applied on received frames.*/ \
305  uint8_t rxmask;
