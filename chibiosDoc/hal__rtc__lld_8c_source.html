54 /**
55  * @brief Wait for synchronization of RTC registers with APB1 bus.
56  * @details This function must be invoked before trying to read RTC registers
57  * in the backup domain: DIV, CNT, ALR. CR registers can always
58  * be read.
59  *
60  * @notapi
61  */
62 static void rtc_apb1_sync(void) {
63 
64  while ((RTC->CRL & RTC_CRL_RSF) == 0)
65  ;
66 }

68 /**
69  * @brief Wait for for previous write operation complete.
70  * @details This function must be invoked before writing to any RTC registers
71  *
72  * @notapi
73  */
74 static void rtc_wait_write_completed(void) {
75 
76  while ((RTC->CRL & RTC_CRL_RTOFF) == 0)
77  ;
78 }

80 /**
81  * @brief Acquires write access to RTC registers.
82  * @details Before writing to the backup domain RTC registers the previous
83  * write operation must be completed. Use this function before
84  * writing to PRL, CNT, ALR registers.
85  *
86  * @notapi
87  */
88 static void rtc_acquire_access(void) {
89 
90  rtc_wait_write_completed();
91  RTC->CRL |= RTC_CRL_CNF;
92 }

119 /**
120  * @brief Converts time from seconds/milliseconds to timespec.
121  *
122  * @param[in] tv_sec seconds value
123  * @param[in] tv_msec milliseconds value
124  * @param[out] timespec pointer to a @p RTCDateTime structure
125  *
126  * @notapi
127  */
128 static void rtc_decode(uint32_t tv_sec,
129  uint32_t tv_msec,
130  RTCDateTime *timespec) {
131  struct tm tim;
132  struct tm *t;
133 
134  /* If the conversion is successful the function returns a pointer
135  to the object the result was written into.*/
136 #if defined(__GNUC__) || defined(__CC_ARM)
137  t = localtime_r((time_t *)&(tv_sec), &tim);
138  osalDbgAssert(t != NULL, "conversion failed");
139 #else
140  t = localtime(&tv_sec);
141  memcpy(&tim, t, sizeof(struct tm));
142 #endif
143 
144  rtcConvertStructTmToDateTime(&tim, tv_msec, timespec);
145 }

151 /**
152  * @brief RTC interrupt handler.
153  *
154  * @isr
155  */
156 OSAL_IRQ_HANDLER(STM32_RTC1_HANDLER) {
157  uint16_t flags;
158 
159  OSAL_IRQ_PROLOGUE();
160 
161  /* Code hits this wait only when AHB1 bus was previously powered off by any
162  reason (standby, reset, etc). In other cases there is no waiting.*/
163  rtc_apb1_sync();
164 
165  /* Mask of all enabled and pending sources.*/
166  flags = RTCD1.rtc->CRH & RTCD1.rtc->CRL;
167  RTCD1.rtc->CRL &= ~(RTC_CRL_SECF | RTC_CRL_ALRF | RTC_CRL_OWF);
168 
169  if (flags & RTC_CRL_SECF)
170  RTCD1.callback(&RTCD1, RTC_EVENT_SECOND);
171 
172  if (flags & RTC_CRL_ALRF)
173  RTCD1.callback(&RTCD1, RTC_EVENT_ALARM);
174 
175  if (flags & RTC_CRL_OWF)
176  RTCD1.callback(&RTCD1, RTC_EVENT_OVERFLOW);
177 
178  OSAL_IRQ_EPILOGUE();
179 }

185 /**
186  * @brief Load value of RTCCLK to prescaler registers.
187  * @note The pre-scaler must not be set on every reset as RTC clock
188  * counts are lost when it is set.
189  * @note This function designed to be called from
190  * hal_lld_backup_domain_init(). Because there is only place
191  * where possible to detect BKP domain reset event reliably.
192  *
193  * @notapi
194  */
195 void rtc_lld_set_prescaler(void) {
196  syssts_t sts;
197 
198  /* Entering a reentrant critical zone.*/
199  sts = osalSysGetStatusAndLockX();
200 
201  rtc_acquire_access();
202  RTC->PRLH = (uint16_t)((STM32_RTCCLK - 1) >> 16) & 0x000F;
203  RTC->PRLL = (uint16_t)(((STM32_RTCCLK - 1)) & 0xFFFF);
204  rtc_release_access();
205 
206  /* Leaving a reentrant critical zone.*/
207  osalSysRestoreStatusX(sts);
208 }

210 /**
211  * @brief Initialize RTC.
212  *
213  * @notapi
214  */
215 void rtc_lld_init(void) {
216 
217  /* RTC object initialization.*/
218  rtcObjectInit(&RTCD1);
219 
220  /* RTC pointer initialization.*/
221  RTCD1.rtc = RTC;
222 
223  /* RSF bit must be cleared by software after an APB1 reset or an APB1 clock
224  stop. Otherwise its value will not be actual. */
225  RTCD1.rtc->CRL &= ~RTC_CRL_RSF;
226 
227  /* Required because access to PRL.*/
228  rtc_apb1_sync();
229 
230  /* All interrupts initially disabled.*/
231  rtc_wait_write_completed();
232  RTCD1.rtc->CRH = 0;
233 
234  /* Callback initially disabled.*/
235  RTCD1.callback = NULL;
236 
237  /* IRQ vector permanently assigned to this driver.*/
238  nvicEnableVector(STM32_RTC1_NUMBER, STM32_RTC_IRQ_PRIORITY);
239 }

274 /**
275  * @brief Set alarm time.
276  *
277  * @note Default value after BKP domain reset is 0xFFFFFFFF
278  * @note The function can be called from any context.
279  *
280  * @param[in] rtcp pointer to RTC driver structure
281  * @param[in] alarm alarm identifier
282  * @param[in] alarmspec pointer to a @p RTCAlarm structure
283  *
284  * @notapi
285  */
286 void rtc_lld_set_alarm(RTCDriver *rtcp,
287  rtcalarm_t alarm_number,
288  const RTCAlarm *alarmspec) {
289  syssts_t sts;
290  (void)alarm_number;
291 
292  /* Entering a reentrant critical zone.*/
293  sts = osalSysGetStatusAndLockX();
294 
295  rtc_acquire_access();
296  if (alarmspec != NULL) {
297  rtcp->rtc->ALRH = (uint16_t)(alarmspec->tv_sec >> 16);
298  rtcp->rtc->ALRL = (uint16_t)(alarmspec->tv_sec & 0xFFFF);
299  }
300  else {
301  rtcp->rtc->ALRH = 0;
302  rtcp->rtc->ALRL = 0;
303  }
304  rtc_release_access();
305 
306  /* Leaving a reentrant critical zone.*/
307  osalSysRestoreStatusX(sts);
308 }

341 /**
342  * @brief Enables or disables RTC callbacks.
343  * @details This function enables or disables callbacks, use a @p NULL pointer
344  * in order to disable a callback.
345  * @note The function can be called from any context.
346  *
347  * @param[in] rtcp pointer to RTC driver structure
348  * @param[in] callback callback function pointer or @p NULL
349  *
350  * @notapi
351  */
352 void rtc_lld_set_callback(RTCDriver *rtcp, rtccb_t callback) {
353  syssts_t sts;
354 
355  /* Entering a reentrant critical zone.*/
356  sts = osalSysGetStatusAndLockX();
357 
358  if (callback != NULL) {
359 
360  /* IRQ sources enabled only after setting up the callback.*/
361  rtcp->callback = callback;
362 
363  rtc_wait_write_completed();
364  rtcp->rtc->CRL &= ~(RTC_CRL_OWF | RTC_CRL_ALRF | RTC_CRL_SECF);
365  rtcp->rtc->CRH = RTC_CRH_OWIE | RTC_CRH_ALRIE | RTC_CRH_SECIE;
366  }
367  else {
368  rtc_wait_write_completed();
369  rtcp->rtc->CRH = 0;
370 
371  /* Callback set to NULL only after disabling the IRQ sources.*/
372  rtcp->callback = NULL;
373  }
374 
375  /* Leaving a reentrant critical zone.*/
376  osalSysRestoreStatusX(sts);
377 }

379 /**
380  * @brief Get seconds and (optionally) milliseconds from RTC.
381  * @note The function can be called from any context.
382  *
383  * @param[in] rtcp pointer to RTC driver structure
384  * @param[out] tv_sec pointer to seconds value
385  * @param[out] tv_msec pointer to milliseconds value, set it
386  * to @p NULL if not needed
387  *
388  * @api
389  */
390 void rtcSTM32GetSecMsec(RTCDriver *rtcp, uint32_t *tv_sec, uint32_t *tv_msec) {
391  uint32_t time_frac;
392  syssts_t sts;
393 
394  osalDbgCheck((NULL != tv_sec) && (NULL != rtcp));
395 
396  /* Entering a reentrant critical zone.*/
397  sts = osalSysGetStatusAndLockX();
398 
399  /* Required because access to CNT and DIV.*/
400  rtc_apb1_sync();
401 
402  /* wait for previous write accesses to complete.*/
403  rtc_wait_write_completed();
404 
405  /* Loops until two consecutive read returning the same value.*/
406  do {
407  *tv_sec = ((uint32_t)(rtcp->rtc->CNTH) << 16) + rtcp->rtc->CNTL;
408  time_frac = (((uint32_t)rtcp->rtc->DIVH) << 16) + (uint32_t)rtcp->rtc->DIVL;
409  } while ((*tv_sec) != (((uint32_t)(rtcp->rtc->CNTH) << 16) + rtcp->rtc->CNTL));
410 
411  /* Leaving a reentrant critical zone.*/
412  osalSysRestoreStatusX(sts);
413 
414  if (NULL != tv_msec)
415  *tv_msec = (((uint32_t)STM32_RTCCLK - 1 - time_frac) * 1000) / STM32_RTCCLK;
416 }
