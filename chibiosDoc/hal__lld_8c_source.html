49 /**
50  * @brief Initializes the backup domain.
51  * @note WARNING! Changing clock source impossible without resetting
52  * of the whole BKP domain.
53  */
54 static void hal_lld_backup_domain_init(void) {
55 
56  /* Backup domain access enabled and left open.*/
57  PWR->CR |= PWR_CR_DBP;
58 
59 #if HAL_USE_RTC
60  /* Reset BKP domain if different clock source selected.*/
61  if ((RCC->BDCR & STM32_RTCSEL_MASK) != STM32_RTCSEL){
62  /* Backup domain reset.*/
63  RCC->BDCR = RCC_BDCR_BDRST;
64  RCC->BDCR = 0;
65  }
66 
67  /* If enabled then the LSE is started.*/
68 #if STM32_LSE_ENABLED
69 #if defined(STM32_LSE_BYPASS)
70  /* LSE Bypass.*/
71  RCC->BDCR |= RCC_BDCR_LSEON | RCC_BDCR_LSEBYP;
72 #else
73  /* No LSE Bypass.*/
74  RCC->BDCR |= RCC_BDCR_LSEON;
75 #endif
76  while ((RCC->BDCR & RCC_BDCR_LSERDY) == 0)
77  ; /* Waits until LSE is stable. */
78 #endif /* STM32_LSE_ENABLED */
79 
80 #if STM32_RTCSEL != STM32_RTCSEL_NOCLOCK
81  /* If the backup domain hasn't been initialized yet then proceed with
82  initialization.*/
83  if ((RCC->BDCR & RCC_BDCR_RTCEN) == 0) {
84  /* Selects clock source.*/
85  RCC->BDCR |= STM32_RTCSEL;
86 
87  /* Prescaler value loaded in registers.*/
88  rtc_lld_set_prescaler();
89 
90  /* RTC clock enabled.*/
91  RCC->BDCR |= RCC_BDCR_RTCEN;
92  }
93 #endif /* STM32_RTCSEL != STM32_RTCSEL_NOCLOCK */
94 #endif /* HAL_USE_RTC */
95 }

127 /**
128  * @brief Low level HAL driver initialization.
129  *
130  * @notapi
131  */
132 void hal_lld_init(void) {
133 
134  /* Reset of all peripherals.*/
135  rccResetAPB1(0xFFFFFFFF);
136  rccResetAPB2(0xFFFFFFFF);
137 
138  /* PWR and BD clocks enabled.*/
139  rccEnablePWRInterface(true);
140  rccEnableBKPInterface(true);
141 
142  /* Initializes the backup domain.*/
143  hal_lld_backup_domain_init();
144 
145  /* DMA subsystems initialization.*/
146 #if defined(STM32_DMA_REQUIRED)
147  dmaInit();
148 #endif
149 
150  /* IRQ subsystem initialization.*/
151  irqInit();
152 
153  /* Programmable voltage detector enable.*/
154 #if STM32_PVD_ENABLE
155  PWR->CR |= PWR_CR_PVDE | (STM32_PLS & STM32_PLS_MASK);
156 #endif /* STM32_PVD_ENABLE */
157 }

159 /**
160  * @brief STM32 clocks and PLL initialization.
161  * @note All the involved constants come from the file @p board.h.
162  * @note This function should be invoked just after the system reset.
163  *
164  * @special
165  */
166 #if defined(STM32F10X_LD) || defined(STM32F10X_LD_VL) || \
167  defined(STM32F10X_MD) || defined(STM32F10X_MD_VL) || \
168  defined(STM32F10X_HD) || defined(STM32F10X_XL) || \
169  defined(__DOXYGEN__)
170 /*
171  * Clocks initialization for all sub-families except CL.
172  */
173 void stm32_clock_init(void) {
174 
175 #if !STM32_NO_INIT
176  /* HSI setup, it enforces the reset situation in order to handle possible
177  problems with JTAG probes and re-initializations.*/
178  RCC->CR |= RCC_CR_HSION; /* Make sure HSI is ON. */
179  while (!(RCC->CR & RCC_CR_HSIRDY))
180  ; /* Wait until HSI is stable. */
181  RCC->CR &= RCC_CR_HSITRIM | RCC_CR_HSION; /* CR Reset value. */
182  RCC->CFGR = 0; /* CFGR reset value. */
183  while ((RCC->CFGR & RCC_CFGR_SWS) != RCC_CFGR_SWS_HSI)
184  ; /* Waits until HSI is selected. */
185 
186 #if STM32_HSE_ENABLED
187 #if defined(STM32_HSE_BYPASS)
188  /* HSE Bypass.*/
189  RCC->CR |= RCC_CR_HSEON | RCC_CR_HSEBYP;
190 #endif
191  /* HSE activation.*/
192  RCC->CR |= RCC_CR_HSEON;
193  while (!(RCC->CR & RCC_CR_HSERDY))
194  ; /* Waits until HSE is stable. */
195 #endif
196 
197 #if STM32_LSI_ENABLED
198  /* LSI activation.*/
199  RCC->CSR |= RCC_CSR_LSION;
200  while ((RCC->CSR & RCC_CSR_LSIRDY) == 0)
201  ; /* Waits until LSI is stable. */
202 #endif
203 
204 #if STM32_ACTIVATE_PLL
205  /* PLL activation.*/
206  RCC->CFGR |= STM32_PLLMUL | STM32_PLLXTPRE | STM32_PLLSRC;
207  RCC->CR |= RCC_CR_PLLON;
208  while (!(RCC->CR & RCC_CR_PLLRDY))
209  ; /* Waits until PLL is stable. */
210 #endif
211 
212  /* Clock settings.*/
213 #if STM32_HAS_USB
214  RCC->CFGR = STM32_MCOSEL | STM32_USBPRE | STM32_PLLMUL | STM32_PLLXTPRE |
215  STM32_PLLSRC | STM32_ADCPRE | STM32_PPRE2 | STM32_PPRE1 |
216  STM32_HPRE;
217 #else
218  RCC->CFGR = STM32_MCOSEL | STM32_PLLMUL | STM32_PLLXTPRE |
219  STM32_PLLSRC | STM32_ADCPRE | STM32_PPRE2 | STM32_PPRE1 |
220  STM32_HPRE;
221 #endif
222 
223  /* Flash setup and final clock selection. */
224  FLASH->ACR = STM32_FLASHBITS;
225 
226  /* Switching to the configured clock source if it is different from HSI.*/
227 #if (STM32_SW != STM32_SW_HSI)
228  /* Switches clock source.*/
229  RCC->CFGR |= STM32_SW;
230  while ((RCC->CFGR & RCC_CFGR_SWS) != (STM32_SW << 2))
231  ; /* Waits selection complete. */
232 #endif
233 
234 #if !STM32_HSI_ENABLED
235  RCC->CR &= ~RCC_CR_HSION;
236 #endif
237 #endif /* !STM32_NO_INIT */
238 }

240 #elif defined(STM32F10X_CL)
241 /*
242  * Clocks initialization for the CL sub-family.
243  */
244 void stm32_clock_init(void) {
245 
246 #if !STM32_NO_INIT
247  /* HSI setup, it enforces the reset situation in order to handle possible
248  problems with JTAG probes and re-initializations.*/
249  RCC->CR |= RCC_CR_HSION; /* Make sure HSI is ON. */
250  while (!(RCC->CR & RCC_CR_HSIRDY))
251  ; /* Wait until HSI is stable. */
252 
253  /* HSI is selected as new source without touching the other fields in
254  CFGR. Clearing the register has to be postponed after HSI is the
255  new source.*/
256  RCC->CFGR &= ~RCC_CFGR_SW; /* Reset SW */
257  RCC->CFGR |= RCC_CFGR_SWS_HSI; /* Select HSI as internal*/
258  while ((RCC->CFGR & RCC_CFGR_SWS) != RCC_CFGR_SWS_HSI)
259  ; /* Wait until HSI is selected. */
260 
261  /* Registers finally cleared to reset values.*/
262  RCC->CR &= RCC_CR_HSITRIM | RCC_CR_HSION; /* CR Reset value. */
263  RCC->CFGR = 0; /* CFGR reset value. */
264 
265 #if STM32_HSE_ENABLED
266 #if defined(STM32_HSE_BYPASS)
267  /* HSE Bypass.*/
268  RCC->CR |= RCC_CR_HSEBYP;
269 #endif
270  /* HSE activation.*/
271  RCC->CR |= RCC_CR_HSEON;
272  while (!(RCC->CR & RCC_CR_HSERDY))
273  ; /* Waits until HSE is stable. */
274 #endif
275 
276 #if STM32_LSI_ENABLED
277  /* LSI activation.*/
278  RCC->CSR |= RCC_CSR_LSION;
279  while ((RCC->CSR & RCC_CSR_LSIRDY) == 0)
280  ; /* Waits until LSI is stable. */
281 #endif
282 
283  /* Settings of various dividers and multipliers in CFGR2.*/
284  RCC->CFGR2 = STM32_PLL3MUL | STM32_PLL2MUL | STM32_PREDIV2 |
285  STM32_PREDIV1 | STM32_PREDIV1SRC;
286 
287  /* PLL2 setup, if activated.*/
288 #if STM32_ACTIVATE_PLL2
289  RCC->CR |= RCC_CR_PLL2ON;
290  while (!(RCC->CR & RCC_CR_PLL2RDY))
291  ; /* Waits until PLL2 is stable. */
292 #endif
293 
294  /* PLL3 setup, if activated.*/
295 #if STM32_ACTIVATE_PLL3
296  RCC->CR |= RCC_CR_PLL3ON;
297  while (!(RCC->CR & RCC_CR_PLL3RDY))
298  ; /* Waits until PLL3 is stable. */
299 #endif
300 
301  /* PLL1 setup, if activated.*/
302 #if STM32_ACTIVATE_PLL1
303  RCC->CFGR |= STM32_PLLMUL | STM32_PLLSRC;
304  RCC->CR |= RCC_CR_PLLON;
305  while (!(RCC->CR & RCC_CR_PLLRDY))
306  ; /* Waits until PLL1 is stable. */
307 #endif
308 
309  /* Clock settings.*/
310 #if STM32_HAS_OTG1
311  RCC->CFGR = STM32_MCOSEL | STM32_OTGFSPRE | STM32_PLLMUL | STM32_PLLSRC |
312  STM32_ADCPRE | STM32_PPRE2 | STM32_PPRE1 | STM32_HPRE;
313 #else
314  RCC->CFGR = STM32_MCO | STM32_PLLMUL | STM32_PLLSRC |
315  STM32_ADCPRE | STM32_PPRE2 | STM32_PPRE1 | STM32_HPRE;
316 #endif
317 
318  /* Flash setup and final clock selection. */
319  FLASH->ACR = STM32_FLASHBITS; /* Flash wait states depending on clock. */
320 
321  /* Switching to the configured clock source if it is different from HSI.*/
322 #if (STM32_SW != STM32_SW_HSI)
323  RCC->CFGR |= STM32_SW; /* Switches on the selected clock source. */
324  while ((RCC->CFGR & RCC_CFGR_SWS) != (STM32_SW << 2))
325  ;
326 #endif
327 
328 #if !STM32_HSI_ENABLED
329  RCC->CR &= ~RCC_CR_HSION;
330 #endif
331 #endif /* !STM32_NO_INIT */
332 }
