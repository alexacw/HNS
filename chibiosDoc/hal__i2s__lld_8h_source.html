305 /**
306  * @brief I2S notification callback type.
307  *
308  * @param[in] i2sp pointer to the @p I2SDriver object
309  * @param[in] offset offset in buffers of the data to read/write
310  * @param[in] n number of samples to read/write
311  */
312 typedef void (*i2scallback_t)(I2SDriver *i2sp, size_t offset, size_t n);
313 
314 /**
315  * @brief Driver configuration structure.
316  * @note It could be empty on some architectures.
317  */
318 typedef struct {
319  /**
320  * @brief Transmission buffer pointer.
321  * @note Can be @p NULL if TX is not required.
322  */
323  const void *tx_buffer;
324  /**
325  * @brief Receive buffer pointer.
326  * @note Can be @p NULL if RX is not required.
327  */
328  void *rx_buffer;
329  /**
330  * @brief TX and RX buffers size as number of samples.
331  */
332  size_t size;
333  /**
334  * @brief Callback function called during streaming.
335  */
336  i2scallback_t end_cb;
337  /* End of the mandatory fields.*/
338  /**
339  * @brief Configuration of the I2SCFGR register.
340  * @details See the STM32 reference manual, this register is used for
341  * the I2S configuration, the following bits must not be
342  * specified because handled directly by the driver:
343  * - I2SMOD
344  * - I2SE
345  * - I2SCFG
346  * .
347  */
348  int16_t i2scfgr;
349  /**
350  * @brief Configuration of the I2SPR register.
351  * @details See the STM32 reference manual, this register is used for
352  * the I2S clock setup.
353  */
354  int16_t i2spr;
355 } I2SConfig;

357 /**
358  * @brief Structure representing an I2S driver.
359  */
360 struct I2SDriver {
361  /**
362  * @brief Driver state.
363  */
364  i2sstate_t state;
365  /**
366  * @brief Current configuration data.
367  */
368  const I2SConfig *config;
369  /* End of the mandatory fields.*/
370  /**
371  * @brief Pointer to the SPIx registers block.
372  */
373  SPI_TypeDef *spi;
374  /**
375  * @brief Calculated part of the I2SCFGR register.
376  */
377  uint16_t cfg;
378  /**
379  * @brief Receive DMA stream or @p NULL.
380  */
381  const stm32_dma_stream_t *dmarx;
382  /**
383  * @brief Transmit DMA stream or @p NULL.
384  */
385  const stm32_dma_stream_t *dmatx;
386  /**
387  * @brief RX DMA mode bit mask.
388  */
389  uint32_t rxdmamode;
390  /**
391  * @brief TX DMA mode bit mask.
392  */
393  uint32_t txdmamode;
394 };
