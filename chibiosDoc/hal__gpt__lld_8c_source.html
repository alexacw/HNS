490

674  /* Prescaler value calculation.*/
675  psc = (uint16_t)((gptp->clock / gptp->config->frequency) - 1);
676  osalDbgAssert(((uint32_t)(psc + 1) * gptp->config->frequency) == gptp->clock,
677  "invalid frequency");
678 
679  /* Timer configuration.*/
680  gptp->tim->CR1 = 0; /* Initially stopped. */
681  gptp->tim->CR2 = gptp->config->cr2;
682  gptp->tim->PSC = psc; /* Prescaler value. */
683  gptp->tim->SR = 0; /* Clear pending IRQs. */
684  gptp->tim->DIER = gptp->config->dier & /* DMA-related DIER bits. */
685  ~STM32_TIM_DIER_IRQ_MASK;
686 }

812 /**
813  * @brief Starts the timer in continuous mode.
814  *
815  * @param[in] gptp pointer to the @p GPTDriver object
816  * @param[in] interval period in ticks
817  *
818  * @notapi
819  */
820 void gpt_lld_start_timer(GPTDriver *gptp, gptcnt_t interval) {
821 
822  gptp->tim->ARR = (uint32_t)interval; /* Time constant. */
823  gptp->tim->EGR = STM32_TIM_EGR_UG; /* Update event. */
824  gptp->tim->CNT = 0; /* Reset counter. */
825 
826  /* NOTE: After generating the UG event it takes several clock cycles before
827  SR bit 0 goes to 1. This is why the clearing of CNT has been inserted
828  before the clearing of SR, to give it some time.*/
829  gptp->tim->SR = 0; /* Clear pending IRQs. */
830  if (NULL != gptp->config->callback)
831  gptp->tim->DIER |= STM32_TIM_DIER_UIE; /* Update Event IRQ enabled.*/
832  gptp->tim->CR1 = STM32_TIM_CR1_ARPE | STM32_TIM_CR1_URS | STM32_TIM_CR1_CEN;
833 }

845  gptp->tim->SR = 0; /* Clear pending IRQs. */

851 /**
852  * @brief Starts the timer in one shot mode and waits for completion.
853  * @details This function specifically polls the timer waiting for completion
854  * in order to not have extra delays caused by interrupt servicing,
855  * this function is only recommended for short delays.
856  *
857  * @param[in] gptp pointer to the @p GPTDriver object
858  * @param[in] interval time interval in ticks
859  *
860  * @notapi
861  */
862 void gpt_lld_polled_delay(GPTDriver *gptp, gptcnt_t interval) {
863 
864  gptp->tim->ARR = (uint32_t)interval; /* Time constant. */
865  gptp->tim->EGR = STM32_TIM_EGR_UG; /* Update event. */
866  gptp->tim->SR = 0; /* Clear pending IRQs. */
867  gptp->tim->CR1 = STM32_TIM_CR1_OPM | STM32_TIM_CR1_URS | STM32_TIM_CR1_CEN;
868  while (!(gptp->tim->SR & STM32_TIM_SR_UIF))
869  ;
870  gptp->tim->SR = 0; /* Clear pending IRQs. */
871 }
