384 /**

407  * @brief Channels configurations.

409  PWMChannelConfig channels[PWM_CHANNELS];

423  /**

473 /**
474  * @brief Changes the period the PWM peripheral.
475  * @details This function changes the period of a PWM unit that has already
476  * been activated using @p pwmStart().
477  * @pre The PWM unit must have been activated using @p pwmStart().
478  * @post The PWM unit period is changed to the new value.
479  * @note The function has effect at the next cycle start.
480  * @note If a period is specified that is shorter than the pulse width
481  * programmed in one of the channels then the behavior is not
482  * guaranteed.
483  *
484  * @param[in] pwmp pointer to a @p PWMDriver object
485  * @param[in] period new cycle time in ticks
486  *
487  * @notapi
488  */
489 #define pwm_lld_change_period(pwmp, period) \
490  ((pwmp)->tim->ARR = ((period) - 1))
