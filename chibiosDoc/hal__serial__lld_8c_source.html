94 /**
95  * @brief USART initialization.
96  * @details This function must be invoked with interrupts disabled.
97  *
98  * @param[in] sdp pointer to a @p SerialDriver object
99  * @param[in] config the architecture-dependent serial driver configuration
100  */
101 static void usart_init(SerialDriver *sdp, const SerialConfig *config) {
102  USART_TypeDef *u = sdp->usart;
103 
104  /* Baud rate setting.*/
105 #if STM32_HAS_USART6
106  if ((sdp->usart == USART1) || (sdp->usart == USART6))
107 #else
108  if (sdp->usart == USART1)
109 #endif
110  u->BRR = STM32_PCLK2 / config->speed;
111  else
112  u->BRR = STM32_PCLK1 / config->speed;
113 
114  /* Note that some bits are enforced.*/
115  u->CR2 = config->cr2 | USART_CR2_LBDIE;
116  u->CR3 = config->cr3 | USART_CR3_EIE;
117  u->CR1 = config->cr1 | USART_CR1_UE | USART_CR1_PEIE |
118  USART_CR1_RXNEIE | USART_CR1_TE |
119  USART_CR1_RE;
120  u->SR = 0;
121  (void)u->SR; /* SR reset step 1.*/
122  (void)u->DR; /* SR reset step 2.*/
123 
124  /* Deciding mask to be applied on the data register on receive, this is
125  required in order to mask out the parity bit.*/
126  if ((config->cr1 & (USART_CR1_M | USART_CR1_PCE)) == USART_CR1_PCE) {
127  sdp->rxmask = 0x7F;
128  }
129  else {
130  sdp->rxmask = 0xFF;
131  }
132 }

147 /**
148  * @brief Error handling routine.
149  *
150  * @param[in] sdp pointer to a @p SerialDriver object
151  * @param[in] sr USART SR register value
152  */
153 static void set_error(SerialDriver *sdp, uint16_t sr) {
154  eventflags_t sts = 0;
155 
156  if (sr & USART_SR_ORE)
157  sts |= SD_OVERRUN_ERROR;
158  if (sr & USART_SR_PE)
159  sts |= SD_PARITY_ERROR;
160  if (sr & USART_SR_FE)
161  sts |= SD_FRAMING_ERROR;
162  if (sr & USART_SR_NE)
163  sts |= SD_NOISE_ERROR;
164  chnAddFlagsI(sdp, sts);
165 }

167 /**
168  * @brief Common IRQ handler.
169  *
170  * @param[in] sdp communication channel associated to the USART
171  */
172 static void serve_interrupt(SerialDriver *sdp) {
173  USART_TypeDef *u = sdp->usart;
174  uint16_t cr1 = u->CR1;
175  uint16_t sr = u->SR;
176 
177  /* Special case, LIN break detection.*/
178  if (sr & USART_SR_LBD) {
179  osalSysLockFromISR();
180  chnAddFlagsI(sdp, SD_BREAK_DETECTED);
181  u->SR = ~USART_SR_LBD;
182  osalSysUnlockFromISR();
183  }
184 
185  /* Data available.*/
186  osalSysLockFromISR();
187  while (sr & (USART_SR_RXNE | USART_SR_ORE | USART_SR_NE | USART_SR_FE |
188  USART_SR_PE)) {
189  uint8_t b;
190 
191  /* Error condition detection.*/
192  if (sr & (USART_SR_ORE | USART_SR_NE | USART_SR_FE | USART_SR_PE))
193  set_error(sdp, sr);
194  b = (uint8_t)u->DR & sdp->rxmask;
195  if (sr & USART_SR_RXNE)
196  sdIncomingDataI(sdp, b);
197  sr = u->SR;
198  }
199  osalSysUnlockFromISR();
200 
201  /* Transmission buffer empty.*/
202  if ((cr1 & USART_CR1_TXEIE) && (sr & USART_SR_TXE)) {
203  msg_t b;
204  osalSysLockFromISR();
205  b = oqGetI(&sdp->oqueue);
206  if (b < MSG_OK) {
207  chnAddFlagsI(sdp, CHN_OUTPUT_EMPTY);
208  u->CR1 = (cr1 & ~USART_CR1_TXEIE) | USART_CR1_TCIE;
209  }
210  else
211  u->DR = b;
212  osalSysUnlockFromISR();
213  }
214 
215  /* Physical transmission end.*/
216  if ((cr1 & USART_CR1_TCIE) && (sr & USART_SR_TC)) {
217  osalSysLockFromISR();
218  if (oqIsEmptyI(&sdp->oqueue)) {
219  chnAddFlagsI(sdp, CHN_TRANSMISSION_END);
220  u->CR1 = cr1 & ~USART_CR1_TCIE;
221  }
222  osalSysUnlockFromISR();
223  }
224 }
