502 /**
503  * @brief Starts a I2S data exchange.
504  *
505  * @param[in] i2sp pointer to the @p I2SDriver object
506  *
507  * @notapi
508  */
509 void i2s_lld_start_exchange(I2SDriver *i2sp) {
510  size_t size = i2sp->config->size;
511 
512  /* In 32 bit modes the DMA has to perform double operations because fetches
513  are always performed using 16 bit accesses.
514  DATLEN CHLEN SIZE
515  00 (16) 0 (16) 16
516  00 (16) 1 (32) 16
517  01 (24) X 32
518  10 (32) X 32
519  11 (NA) X NA
520  */
521  if ((i2sp->config->i2scfgr & SPI_I2SCFGR_DATLEN) != 0)
522  size *= 2;
523 
524  /* RX DMA setup.*/
525  if (NULL != i2sp->dmarx) {
526  dmaStreamSetMode(i2sp->dmarx, i2sp->rxdmamode);
527  dmaStreamSetPeripheral(i2sp->dmarx, &i2sp->spi->DR);
528  dmaStreamSetMemory0(i2sp->dmarx, i2sp->config->rx_buffer);
529  dmaStreamSetTransactionSize(i2sp->dmarx, size);
530  dmaStreamEnable(i2sp->dmarx);
531  }
532 
533  /* TX DMA setup.*/
534  if (NULL != i2sp->dmatx) {
535  dmaStreamSetMode(i2sp->dmatx, i2sp->txdmamode);
536  dmaStreamSetPeripheral(i2sp->dmatx, &i2sp->spi->DR);
537  dmaStreamSetMemory0(i2sp->dmatx, i2sp->config->tx_buffer);
538  dmaStreamSetTransactionSize(i2sp->dmatx, size);
539  dmaStreamEnable(i2sp->dmatx);
540  }
541 
542  /* Starting transfer.*/
543  i2sp->spi->I2SCFGR |= SPI_I2SCFGR_I2SE;
544 }

546 /**
547  * @brief Stops the ongoing data exchange.
548  * @details The ongoing data exchange, if any, is stopped, if the driver
549  * was not active the function does nothing.
550  *
551  * @param[in] i2sp pointer to the @p I2SDriver object
552  *
553  * @notapi
554  */
555 void i2s_lld_stop_exchange(I2SDriver *i2sp) {
556 
557  /* Stop TX DMA, if enabled.*/
558  if (NULL != i2sp->dmatx) {
559  dmaStreamDisable(i2sp->dmatx);
560 
561  /* From the RM: To switch off the I2S, by clearing I2SE, it is mandatory
562  to wait for TXE = 1 and BSY = 0.*/
563  while ((i2sp->spi->SR & (SPI_SR_TXE | SPI_SR_BSY)) != SPI_SR_TXE)
564  ;
565  }
566 
567  /* Stop SPI/I2S peripheral.*/
568  i2sp->spi->I2SCFGR &= ~SPI_I2SCFGR_I2SE;
569 
570  /* Stop RX DMA, if enabled.*/
571  if (NULL != i2sp->dmarx)
572  dmaStreamDisable(i2sp->dmarx);
573 }
