51 static uint16_t get_hword(uint8_t *p) {
52  uint16_t hw;
53 
54  hw = (uint16_t)*p++;
55  hw |= (uint16_t)*p << 8U;
56  return hw;
57 }

72 /**
73  * @brief Standard requests handler.
74  * @details This is the standard requests default handler, most standard
75  * requests are handled here, the user can override the standard
76  * handling using the @p requests_hook_cb hook in the
77  * @p USBConfig structure.
78  *
79  * @param[in] usbp pointer to the @p USBDriver object
80  * @return The request handling exit code.
81  * @retval false Request not recognized by the handler or error.
82  * @retval true Request handled.
83  */
84 static bool default_handler(USBDriver *usbp) {
85  const USBDescriptor *dp;
86 
87  /* Decoding the request.*/
88  switch ((((uint32_t)usbp->setup[0] & (USB_RTYPE_RECIPIENT_MASK |
89  USB_RTYPE_TYPE_MASK)) |
90  ((uint32_t)usbp->setup[1] << 8U))) {
91  case (uint32_t)USB_RTYPE_RECIPIENT_DEVICE | ((uint32_t)USB_REQ_GET_STATUS << 8):
92  /* Just returns the current status word.*/
93  usbSetupTransfer(usbp, (uint8_t *)&usbp->status, 2, NULL);
94  return true;
95  case (uint32_t)USB_RTYPE_RECIPIENT_DEVICE | ((uint32_t)USB_REQ_CLEAR_FEATURE << 8):
96  /* Only the DEVICE_REMOTE_WAKEUP is handled here, any other feature
97  number is handled as an error.*/
98  if (usbp->setup[2] == USB_FEATURE_DEVICE_REMOTE_WAKEUP) {
99  usbp->status &= ~2U;
100  usbSetupTransfer(usbp, NULL, 0, NULL);
101  return true;
102  }
103  return false;
104  case (uint32_t)USB_RTYPE_RECIPIENT_DEVICE | ((uint32_t)USB_REQ_SET_FEATURE << 8):
105  /* Only the DEVICE_REMOTE_WAKEUP is handled here, any other feature
106  number is handled as an error.*/
107  if (usbp->setup[2] == USB_FEATURE_DEVICE_REMOTE_WAKEUP) {
108  usbp->status |= 2U;
109  usbSetupTransfer(usbp, NULL, 0, NULL);
110  return true;
111  }
112  return false;
113  case (uint32_t)USB_RTYPE_RECIPIENT_DEVICE | ((uint32_t)USB_REQ_SET_ADDRESS << 8):
114  /* The SET_ADDRESS handling can be performed here or postponed after
115  the status packed depending on the USB_SET_ADDRESS_MODE low
116  driver setting.*/
117 #if USB_SET_ADDRESS_MODE == USB_EARLY_SET_ADDRESS
118  if ((usbp->setup[0] == USB_RTYPE_RECIPIENT_DEVICE) &&
119  (usbp->setup[1] == USB_REQ_SET_ADDRESS)) {
120  set_address(usbp);
121  }
122  usbSetupTransfer(usbp, NULL, 0, NULL);
123 #else
124  usbSetupTransfer(usbp, NULL, 0, set_address);
125 #endif
126  return true;
127  case (uint32_t)USB_RTYPE_RECIPIENT_DEVICE | ((uint32_t)USB_REQ_GET_DESCRIPTOR << 8):
128  case (uint32_t)USB_RTYPE_RECIPIENT_INTERFACE | ((uint32_t)USB_REQ_GET_DESCRIPTOR << 8):
129  /* Handling descriptor requests from the host.*/
130  dp = usbp->config->get_descriptor_cb(usbp, usbp->setup[3],
131  usbp->setup[2],
132  get_hword(&usbp->setup[4]));
133  if (dp == NULL) {
134  return false;
135  }
136  /*lint -save -e9005 [11.8] Removing const is fine.*/
137  usbSetupTransfer(usbp, (uint8_t *)dp->ud_string, dp->ud_size, NULL);
138  /*lint -restore*/
139  return true;
140  case (uint32_t)USB_RTYPE_RECIPIENT_DEVICE | ((uint32_t)USB_REQ_GET_CONFIGURATION << 8):
141  /* Returning the last selected configuration.*/
142  usbSetupTransfer(usbp, &usbp->configuration, 1, NULL);
143  return true;
144  case (uint32_t)USB_RTYPE_RECIPIENT_DEVICE | ((uint32_t)USB_REQ_SET_CONFIGURATION << 8):
145  /* Handling configuration selection from the host only if it is different
146  from the current configuration.*/
147  if (usbp->configuration != usbp->setup[2]) {
148  /* If the USB device is already active then we have to perform the clear
149  procedure on the current configuration.*/
150  if (usbp->state == USB_ACTIVE) {
151  /* Current configuration cleared.*/
152  osalSysLockFromISR ();
153  usbDisableEndpointsI(usbp);
154  osalSysUnlockFromISR ();
155  usbp->configuration = 0U;
156  usbp->state = USB_SELECTED;
157  _usb_isr_invoke_event_cb(usbp, USB_EVENT_UNCONFIGURED);
158  }
159  if (usbp->setup[2] != 0U) {
160  /* New configuration.*/
161  usbp->configuration = usbp->setup[2];
162  usbp->state = USB_ACTIVE;
163  _usb_isr_invoke_event_cb(usbp, USB_EVENT_CONFIGURED);
164  }
165  }
166  usbSetupTransfer(usbp, NULL, 0, NULL);
167  return true;
168  case (uint32_t)USB_RTYPE_RECIPIENT_INTERFACE | ((uint32_t)USB_REQ_GET_STATUS << 8):
169  case (uint32_t)USB_RTYPE_RECIPIENT_ENDPOINT | ((uint32_t)USB_REQ_SYNCH_FRAME << 8):
170  /* Just sending two zero bytes, the application can change the behavior
171  using a hook..*/
172  /*lint -save -e9005 [11.8] Removing const is fine.*/
173  usbSetupTransfer(usbp, (uint8_t *)zero_status, 2, NULL);
174  /*lint -restore*/
175  return true;
176  case (uint32_t)USB_RTYPE_RECIPIENT_ENDPOINT | ((uint32_t)USB_REQ_GET_STATUS << 8):
177  /* Sending the EP status.*/
178  if ((usbp->setup[4] & 0x80U) != 0U) {
179  switch (usb_lld_get_status_in(usbp, usbp->setup[4] & 0x0FU)) {
180  case EP_STATUS_STALLED:
181  /*lint -save -e9005 [11.8] Removing const is fine.*/
182  usbSetupTransfer(usbp, (uint8_t *)halted_status, 2, NULL);
183  /*lint -restore*/
184  return true;
185  case EP_STATUS_ACTIVE:
186  /*lint -save -e9005 [11.8] Removing const is fine.*/
187  usbSetupTransfer(usbp, (uint8_t *)active_status, 2, NULL);
188  /*lint -restore*/
189  return true;
190  case EP_STATUS_DISABLED:
191  default:
192  return false;
193  }
194  }
195  else {
196  switch (usb_lld_get_status_out(usbp, usbp->setup[4] & 0x0FU)) {
197  case EP_STATUS_STALLED:
198  /*lint -save -e9005 [11.8] Removing const is fine.*/
199  usbSetupTransfer(usbp, (uint8_t *)halted_status, 2, NULL);
200  /*lint -restore*/
201  return true;
202  case EP_STATUS_ACTIVE:
203  /*lint -save -e9005 [11.8] Removing const is fine.*/
204  usbSetupTransfer(usbp, (uint8_t *)active_status, 2, NULL);
205  /*lint -restore*/
206  return true;
207  case EP_STATUS_DISABLED:
208  default:
209  return false;
210  }
211  }
212  case (uint32_t)USB_RTYPE_RECIPIENT_ENDPOINT | ((uint32_t)USB_REQ_CLEAR_FEATURE << 8):
213  /* Only ENDPOINT_HALT is handled as feature.*/
214  if (usbp->setup[2] != USB_FEATURE_ENDPOINT_HALT) {
215  return false;
216  }
217  /* Clearing the EP status, not valid for EP0, it is ignored in that case.*/
218  if ((usbp->setup[4] & 0x0FU) != 0U) {
219  if ((usbp->setup[4] & 0x80U) != 0U) {
220  usb_lld_clear_in(usbp, usbp->setup[4] & 0x0FU);
221  }
222  else {
223  usb_lld_clear_out(usbp, usbp->setup[4] & 0x0FU);
224  }
225  }
226  usbSetupTransfer(usbp, NULL, 0, NULL);
227  return true;
228  case (uint32_t)USB_RTYPE_RECIPIENT_ENDPOINT | ((uint32_t)USB_REQ_SET_FEATURE << 8):
229  /* Only ENDPOINT_HALT is handled as feature.*/
230  if (usbp->setup[2] != USB_FEATURE_ENDPOINT_HALT) {
231  return false;
232  }
233  /* Stalling the EP, not valid for EP0, it is ignored in that case.*/
234  if ((usbp->setup[4] & 0x0FU) != 0U) {
235  if ((usbp->setup[4] & 0x80U) != 0U) {
236  usb_lld_stall_in(usbp, usbp->setup[4] & 0x0FU);
237  }
238  else {
239  usb_lld_stall_out(usbp, usbp->setup[4] & 0x0FU);
240  }
241  }
242  usbSetupTransfer(usbp, NULL, 0, NULL);
243  return true;
244  case (uint32_t)USB_RTYPE_RECIPIENT_DEVICE | ((uint32_t)USB_REQ_SET_DESCRIPTOR << 8):
245  case (uint32_t)USB_RTYPE_RECIPIENT_INTERFACE | ((uint32_t)USB_REQ_CLEAR_FEATURE << 8):
246  case (uint32_t)USB_RTYPE_RECIPIENT_INTERFACE | ((uint32_t)USB_REQ_SET_FEATURE << 8):
247  case (uint32_t)USB_RTYPE_RECIPIENT_INTERFACE | ((uint32_t)USB_REQ_GET_INTERFACE << 8):
248  case (uint32_t)USB_RTYPE_RECIPIENT_INTERFACE | ((uint32_t)USB_REQ_SET_INTERFACE << 8):
249  /* All the above requests are not handled here, if you need them then
250  use the hook mechanism and provide handling.*/
251  default:
252  return false;
253  }
254 }

292 /**
293  * @brief Configures and activates the USB peripheral.
294  *
295  * @param[in] usbp pointer to the @p USBDriver object
296  * @param[in] config pointer to the @p USBConfig object
297  *
298  * @api
299  */
300 void usbStart(USBDriver *usbp, const USBConfig *config) {
301  unsigned i;
302 
303  osalDbgCheck((usbp != NULL) && (config != NULL));
304 
305  osalSysLock();
306  osalDbgAssert((usbp->state == USB_STOP) || (usbp->state == USB_READY),
307  "invalid state");
308  usbp->config = config;
309  for (i = 0; i <= (unsigned)USB_MAX_ENDPOINTS; i++) {
310  usbp->epc[i] = NULL;
311  }
312  usb_lld_start(usbp);
313  usbp->state = USB_READY;
314  osalSysUnlock();
315 }

317 /**
318  * @brief Deactivates the USB peripheral.
319  *
320  * @param[in] usbp pointer to the @p USBDriver object
321  *
322  * @api
323  */
324 void usbStop(USBDriver *usbp) {
325  unsigned i;
326 
327  osalDbgCheck(usbp != NULL);
328 
329  osalSysLock();
330 
331  osalDbgAssert((usbp->state == USB_STOP) || (usbp->state == USB_READY) ||
332  (usbp->state == USB_SELECTED) || (usbp->state == USB_ACTIVE) ||
333  (usbp->state == USB_SUSPENDED),
334  "invalid state");
335 
336  usb_lld_stop(usbp);
337  usbp->config = NULL;
338  usbp->state = USB_STOP;
339 
340  /* Resetting all ongoing synchronous operations.*/
341  for (i = 0; i <= (unsigned)USB_MAX_ENDPOINTS; i++) {
342 #if USB_USE_WAIT == TRUE
343  if (usbp->epc[i] != NULL) {
344  if (usbp->epc[i]->in_state != NULL) {
345  osalThreadResumeI(&usbp->epc[i]->in_state->thread, MSG_RESET);
346  }
347  if (usbp->epc[i]->out_state != NULL) {
348  osalThreadResumeI(&usbp->epc[i]->out_state->thread, MSG_RESET);
349  }
350  }
351 #endif
352  usbp->epc[i] = NULL;
353  }
354  osalOsRescheduleS();
355 
356  osalSysUnlock();
357 }

359 /**
360  * @brief Enables an endpoint.
361  * @details This function enables an endpoint, both IN and/or OUT directions
362  * depending on the configuration structure.
363  * @note This function must be invoked in response of a SET_CONFIGURATION
364  * or SET_INTERFACE message.
365  *
366  * @param[in] usbp pointer to the @p USBDriver object
367  * @param[in] ep endpoint number
368  * @param[in] epcp the endpoint configuration
369  *
370  * @iclass
371  */
372 void usbInitEndpointI(USBDriver *usbp, usbep_t ep,
373  const USBEndpointConfig *epcp) {
374 
375  osalDbgCheckClassI();
376  osalDbgCheck((usbp != NULL) && (epcp != NULL));
377  osalDbgAssert(usbp->state == USB_ACTIVE,
378  "invalid state");
379  osalDbgAssert(usbp->epc[ep] == NULL, "already initialized");
380 
381  /* Logically enabling the endpoint in the USBDriver structure.*/
382  usbp->epc[ep] = epcp;
383 
384  /* Clearing the state structures, custom fields as well.*/
385  if (epcp->in_state != NULL) {
386  memset(epcp->in_state, 0, sizeof(USBInEndpointState));
387  }
388  if (epcp->out_state != NULL) {
389  memset(epcp->out_state, 0, sizeof(USBOutEndpointState));
390  }
391 
392  /* Low level endpoint activation.*/
393  usb_lld_init_endpoint(usbp, ep);
394 }

396 /**
397  * @brief Disables all the active endpoints.
398  * @details This function disables all the active endpoints except the
399  * endpoint zero.
400  * @note This function must be invoked in response of a SET_CONFIGURATION
401  * message with configuration number zero.
402  *
403  * @param[in] usbp pointer to the @p USBDriver object
404  *
405  * @iclass
406  */
407 void usbDisableEndpointsI(USBDriver *usbp) {
408  unsigned i;
409 
410  osalDbgCheckClassI();
411  osalDbgCheck(usbp != NULL);
412  osalDbgAssert(usbp->state == USB_ACTIVE, "invalid state");
413 
414  usbp->transmitting &= 1U;
415  usbp->receiving &= 1U;
416 
417  for (i = 1; i <= (unsigned)USB_MAX_ENDPOINTS; i++) {
418 #if USB_USE_WAIT == TRUE
419  /* Signaling the event to threads waiting on endpoints.*/
420  if (usbp->epc[i] != NULL) {
421  if (usbp->epc[i]->in_state != NULL) {
422  osalThreadResumeI(&usbp->epc[i]->in_state->thread, MSG_RESET);
423  }
424  if (usbp->epc[i]->out_state != NULL) {
425  osalThreadResumeI(&usbp->epc[i]->out_state->thread, MSG_RESET);
426  }
427  }
428 #endif
429  usbp->epc[i] = NULL;
430  }
431 
432  /* Low level endpoints deactivation.*/
433  usb_lld_disable_endpoints(usbp);
434 }

436 /**
437  * @brief Starts a receive transaction on an OUT endpoint.
438  * @note This function is meant to be called from ISR context outside
439  * critical zones because there is a potentially slow operation
440  * inside.
441  *
442  * @param[in] usbp pointer to the @p USBDriver object
443  * @param[in] ep endpoint number
444  * @param[out] buf buffer where to copy the received data
445  * @param[in] n transaction size. It is recommended a multiple of
446  * the packet size because the excess is discarded.
447  *
448  * @iclass
449  */
450 void usbStartReceiveI(USBDriver *usbp, usbep_t ep,
451  uint8_t *buf, size_t n) {
452  USBOutEndpointState *osp;
453 
454  osalDbgCheckClassI();
455  osalDbgCheck((usbp != NULL) && (ep <= (usbep_t)USB_MAX_ENDPOINTS));
456  osalDbgAssert(!usbGetReceiveStatusI(usbp, ep), "already receiving");
457 
458  /* Marking the endpoint as active.*/
459  usbp->receiving |= (uint16_t)((unsigned)1U << (unsigned)ep);
460 
461  /* Setting up the transfer.*/
462  /*lint -save -e661 [18.1] pclint is confused by the check on ep.*/
463  osp = usbp->epc[ep]->out_state;
464  /*lint -restore*/
465  osp->rxbuf = buf;
466  osp->rxsize = n;
467  osp->rxcnt = 0;
468 #if USB_USE_WAIT == TRUE
469  osp->thread = NULL;
470 #endif
471 
472  /* Starting transfer.*/
473  usb_lld_start_out(usbp, ep);
474 }

662

665

667  usbp->status = 0;

759 /**
760  * @brief Default EP0 SETUP callback.
761  * @details This function is used by the low level driver as default handler
762  * for EP0 SETUP events.
763  *
764  * @param[in] usbp pointer to the @p USBDriver object
765  * @param[in] ep endpoint number, always zero
766  *
767  * @notapi
768  */
769 void _usb_ep0setup(USBDriver *usbp, usbep_t ep) {
770  size_t max;
771 
772  /* Is the EP0 state machine in the correct state for handling setup
773  packets?*/
774  if (usbp->ep0state != USB_EP0_STP_WAITING) {
775  /* This is unexpected could require handling with a warning event.*/
776  /* TODO: handling here.*/
777 
778  /* Resetting the EP0 state machine and going ahead.*/
779  usbp->ep0state = USB_EP0_STP_WAITING;
780  }
781 
782  /* Reading the setup data into the driver buffer.*/
783  usbReadSetup(usbp, ep, usbp->setup);
784 
785  /* First verify if the application has an handler installed for this
786  request.*/
787  /*lint -save -e9007 [13.5] No side effects, it is intentional.*/
788  if ((usbp->config->requests_hook_cb == NULL) ||
789  !(usbp->config->requests_hook_cb(usbp))) {
790  /*lint -restore*/
791  /* Invoking the default handler, if this fails then stalls the
792  endpoint zero as error.*/
793  /*lint -save -e9007 [13.5] No side effects, it is intentional.*/
794  if (((usbp->setup[0] & USB_RTYPE_TYPE_MASK) != USB_RTYPE_TYPE_STD) ||
795  !default_handler(usbp)) {
796  /*lint -restore*/
797  /* Error response, the state machine goes into an error state, the low
798  level layer will have to reset it to USB_EP0_WAITING_SETUP after
799  receiving a SETUP packet.*/
800  usb_lld_stall_in(usbp, 0);
801  usb_lld_stall_out(usbp, 0);
802  _usb_isr_invoke_event_cb(usbp, USB_EVENT_STALLED);
803  usbp->ep0state = USB_EP0_ERROR;
804  return;
805  }
806  }
807 #if (USB_SET_ADDRESS_ACK_HANDLING == USB_SET_ADDRESS_ACK_HW)
808  if (usbp->setup[1] == USB_REQ_SET_ADDRESS) {
809  /* Zero-length packet sent by hardware */
810  return;
811  }
812 #endif
813  /* Transfer preparation. The request handler must have populated
814  correctly the fields ep0next, ep0n and ep0endcb using the macro
815  usbSetupTransfer().*/
816  max = (size_t)get_hword(&usbp->setup[6]);
817  /* The transfer size cannot exceed the specified amount.*/
818  if (usbp->ep0n > max) {
819  usbp->ep0n = max;
820  }
821  if ((usbp->setup[0] & USB_RTYPE_DIR_MASK) == USB_RTYPE_DIR_DEV2HOST) {
822  /* IN phase.*/
823  if (usbp->ep0n != 0U) {
824  /* Starts the transmit phase.*/
825  usbp->ep0state = USB_EP0_IN_TX;
826  osalSysLockFromISR();
827  usbStartTransmitI(usbp, 0, usbp->ep0next, usbp->ep0n);
828  osalSysUnlockFromISR();
829  }
830  else {
831  /* No transmission phase, directly receiving the zero sized status
832  packet.*/
833  usbp->ep0state = USB_EP0_OUT_WAITING_STS;
834 #if (USB_EP0_STATUS_STAGE == USB_EP0_STATUS_STAGE_SW)
835  osalSysLockFromISR();
836  usbStartReceiveI(usbp, 0, NULL, 0);
837  osalSysUnlockFromISR();
838 #else
839  usb_lld_end_setup(usbp, ep);
840 #endif
841  }
842  }
843  else {
844  /* OUT phase.*/
845  if (usbp->ep0n != 0U) {
846  /* Starts the receive phase.*/
847  usbp->ep0state = USB_EP0_OUT_RX;
848  osalSysLockFromISR();
849  usbStartReceiveI(usbp, 0, (uint8_t *)usbp->ep0next, usbp->ep0n);
850  osalSysUnlockFromISR();
851  }
852  else {
853  /* No receive phase, directly sending the zero sized status
854  packet.*/
855  usbp->ep0state = USB_EP0_IN_SENDING_STS;
856 #if (USB_EP0_STATUS_STAGE == USB_EP0_STATUS_STAGE_SW)
857  osalSysLockFromISR();
858  usbStartTransmitI(usbp, 0, NULL, 0);
859  osalSysUnlockFromISR();
860 #else
861  usb_lld_end_setup(usbp, ep);
862 #endif
863  }
864  }
865 }

867 /**
868  * @brief Default EP0 IN callback.
869  * @details This function is used by the low level driver as default handler
870  * for EP0 IN events.
871  *
872  * @param[in] usbp pointer to the @p USBDriver object
873  * @param[in] ep endpoint number, always zero
874  *
875  * @notapi
876  */
877 void _usb_ep0in(USBDriver *usbp, usbep_t ep) {
878  size_t max;
879 
880  (void)ep;
881  switch (usbp->ep0state) {
882  case USB_EP0_IN_TX:
883  max = (size_t)get_hword(&usbp->setup[6]);
884  /* If the transmitted size is less than the requested size and it is a
885  multiple of the maximum packet size then a zero size packet must be
886  transmitted.*/
887  if ((usbp->ep0n < max) &&
888  ((usbp->ep0n % usbp->epc[0]->in_maxsize) == 0U)) {
889  osalSysLockFromISR();
890  usbStartTransmitI(usbp, 0, NULL, 0);
891  osalSysUnlockFromISR();
892  usbp->ep0state = USB_EP0_IN_WAITING_TX0;
893  return;
894  }
895  /* Falls through.*/
896  case USB_EP0_IN_WAITING_TX0:
897  /* Transmit phase over, receiving the zero sized status packet.*/
898  usbp->ep0state = USB_EP0_OUT_WAITING_STS;
899 #if (USB_EP0_STATUS_STAGE == USB_EP0_STATUS_STAGE_SW)
900  osalSysLockFromISR();
901  usbStartReceiveI(usbp, 0, NULL, 0);
902  osalSysUnlockFromISR();
903 #else
904  usb_lld_end_setup(usbp, ep);
905 #endif
906  return;
907  case USB_EP0_IN_SENDING_STS:
908  /* Status packet sent, invoking the callback if defined.*/
909  if (usbp->ep0endcb != NULL) {
910  usbp->ep0endcb(usbp);
911  }
912  usbp->ep0state = USB_EP0_STP_WAITING;
913  return;
914  case USB_EP0_STP_WAITING:
915  case USB_EP0_OUT_WAITING_STS:
916  case USB_EP0_OUT_RX:
917  /* All the above are invalid states in the IN phase.*/
918  osalDbgAssert(false, "EP0 state machine error");
919  /* Falls through.*/
920  case USB_EP0_ERROR:
921  /* Error response, the state machine goes into an error state, the low
922  level layer will have to reset it to USB_EP0_WAITING_SETUP after
923  receiving a SETUP packet.*/
924  usb_lld_stall_in(usbp, 0);
925  usb_lld_stall_out(usbp, 0);
926  _usb_isr_invoke_event_cb(usbp, USB_EVENT_STALLED);
927  usbp->ep0state = USB_EP0_ERROR;
928  return;
929  default:
930  osalDbgAssert(false, "EP0 state machine invalid state");
931  }
932 }
