339  * @brief Callback for cycle period measurement.

343  * @brief Callback for timer overflow.

345  icucallback_t overflow_cb;
346  /* End of the mandatory fields.*/

350  */
351  icuchannel_t channel;

354  * @note The value of this field should normally be equal to zero.

398 /**
399  * @brief Returns the width of the latest pulse.
400  * @details The pulse width is defined as number of ticks between the start
401  * edge and the stop edge.
402  *
403  * @param[in] icup pointer to the @p ICUDriver object
404  * @return The number of ticks.
405  *
406  * @notapi
407  */

409 
410 /**

416  * @return The number of ticks.
