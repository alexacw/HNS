116  * @brief Aborts an I2C transaction.
117  *
118  * @param[in] i2cp pointer to the @p I2CDriver object

122 static void i2c_lld_abort_operation(I2CDriver *i2cp) {
123  I2C_TypeDef *dp = i2cp->i2c;
124 
125  /* Stops the I2C peripheral.*/
126  dp->CR1 = I2C_CR1_SWRST;
127  dp->CR1 = 0;
128  dp->CR2 = 0;
129  dp->SR1 = 0;
130 
131  /* Stops the associated DMA streams.*/
132  dmaStreamDisable(i2cp->dmatx);
133  dmaStreamDisable(i2cp->dmarx);
134 }

136 /**
137  * @brief Set clock speed.
138  *
139  * @param[in] i2cp pointer to the @p I2CDriver object
140  *
141  * @notapi
142  */
143 static void i2c_lld_set_clock(I2CDriver *i2cp) {
144  I2C_TypeDef *dp = i2cp->i2c;
145  uint16_t regCCR, clock_div;
146  int32_t clock_speed = i2cp->config->clock_speed;
147  i2cdutycycle_t duty = i2cp->config->duty_cycle;
148 
149  osalDbgCheck((i2cp != NULL) &&
150  (clock_speed > 0) &&
151  (clock_speed <= 4000000));
152 
153  /* CR2 Configuration.*/
154  dp->CR2 &= (uint16_t)~I2C_CR2_FREQ;
155  dp->CR2 |= (uint16_t)I2C_CLK_FREQ;
156 
157  /* CCR Configuration.*/
158  regCCR = 0;
159  clock_div = I2C_CCR_CCR;
160 
161  if (clock_speed <= 100000) {
162  /* Configure clock_div in standard mode.*/
163  osalDbgAssert(duty == STD_DUTY_CYCLE, "invalid standard mode duty cycle");
164 
165  /* Standard mode clock_div calculate: Tlow/Thigh = 1/1.*/
166  osalDbgAssert((STM32_PCLK1 % (clock_speed * 2)) == 0,
167  "PCLK1 must be divisible without remainder");
168  clock_div = (uint16_t)(STM32_PCLK1 / (clock_speed * 2));
169 
170  osalDbgAssert(clock_div >= 0x04,
171  "clock divider less then 0x04 not allowed");
172  regCCR |= (clock_div & I2C_CCR_CCR);
173 
174  /* Sets the Maximum Rise Time for standard mode.*/
175  dp->TRISE = I2C_CLK_FREQ + 1;
176  }
177  else if (clock_speed <= 400000) {
178  /* Configure clock_div in fast mode.*/
179  osalDbgAssert((duty == FAST_DUTY_CYCLE_2) ||
180  (duty == FAST_DUTY_CYCLE_16_9),
181  "invalid fast mode duty cycle");
182 
183  if (duty == FAST_DUTY_CYCLE_2) {
184  /* Fast mode clock_div calculate: Tlow/Thigh = 2/1.*/
185  osalDbgAssert((STM32_PCLK1 % (clock_speed * 3)) == 0,
186  "PCLK1 must be divided without remainder");
187  clock_div = (uint16_t)(STM32_PCLK1 / (clock_speed * 3));
188  }
189  else if (duty == FAST_DUTY_CYCLE_16_9) {
190  /* Fast mode clock_div calculate: Tlow/Thigh = 16/9.*/
191  osalDbgAssert((STM32_PCLK1 % (clock_speed * 25)) == 0,
192  "PCLK1 must be divided without remainder");
193  clock_div = (uint16_t)(STM32_PCLK1 / (clock_speed * 25));
194  regCCR |= I2C_CCR_DUTY;
195  }
196 
197  osalDbgAssert(clock_div >= 0x01,
198  "clock divider less then 0x04 not allowed");
199  regCCR |= (I2C_CCR_FS | (clock_div & I2C_CCR_CCR));
200 
201  /* Sets the Maximum Rise Time for fast mode.*/
202  dp->TRISE = (I2C_CLK_FREQ * 300 / 1000) + 1;
203  }
204 
205  osalDbgAssert((clock_div <= I2C_CCR_CCR), "the selected clock is too low");
206 
207  dp->CCR = regCCR;
208 }

210 /**
211  * @brief Set operation mode of I2C hardware.
212  *
213  * @param[in] i2cp pointer to the @p I2CDriver object
214  *
215  * @notapi
216  */
217 static void i2c_lld_set_opmode(I2CDriver *i2cp) {
218  I2C_TypeDef *dp = i2cp->i2c;
219  i2copmode_t opmode = i2cp->config->op_mode;
220  uint16_t regCR1;
221 
222  regCR1 = dp->CR1;
223  switch (opmode) {
224  case OPMODE_I2C:
225  regCR1 &= (uint16_t)~(I2C_CR1_SMBUS|I2C_CR1_SMBTYPE);
226  break;
227  case OPMODE_SMBUS_DEVICE:
228  regCR1 |= I2C_CR1_SMBUS;
229  regCR1 &= (uint16_t)~(I2C_CR1_SMBTYPE);
230  break;
231  case OPMODE_SMBUS_HOST:
232  regCR1 |= (I2C_CR1_SMBUS|I2C_CR1_SMBTYPE);
233  break;
234  }
235  dp->CR1 = regCR1;
236 }

238 /**
239  * @brief I2C shared ISR code.
240  *
241  * @param[in] i2cp pointer to the @p I2CDriver object
242  *
243  * @notapi
244  */
245 static void i2c_lld_serve_event_interrupt(I2CDriver *i2cp) {
246  I2C_TypeDef *dp = i2cp->i2c;
247  uint32_t regSR2 = dp->SR2;
248  uint32_t event = dp->SR1;
249 
250  /* Interrupts are disabled just before dmaStreamEnable() because there
251  is no need of interrupts until next transaction begin. All the work is
252  done by the DMA.*/
253  switch (I2C_EV_MASK & (event | (regSR2 << 16))) {
254  case I2C_EV5_MASTER_MODE_SELECT:
255  if ((i2cp->addr >> 8) > 0) {
256  /* 10-bit address: 1 1 1 1 0 X X R/W */
257  dp->DR = 0xF0 | (0x6 & (i2cp->addr >> 8)) | (0x1 & i2cp->addr);
258  } else {
259  dp->DR = i2cp->addr;
260  }
261  break;
262  case I2C_EV9_MASTER_ADD10:
263  /* Set second addr byte (10-bit addressing)*/
264  dp->DR = (0xFF & (i2cp->addr >> 1));
265  break;
266  case I2C_EV6_MASTER_REC_MODE_SELECTED:
267  dp->CR2 &= ~I2C_CR2_ITEVTEN;
268  dmaStreamEnable(i2cp->dmarx);
269  dp->CR2 |= I2C_CR2_LAST; /* Needed in receiver mode. */
270  if (dmaStreamGetTransactionSize(i2cp->dmarx) < 2)
271  dp->CR1 &= ~I2C_CR1_ACK;
272  break;
273  case I2C_EV6_MASTER_TRA_MODE_SELECTED:
274  dp->CR2 &= ~I2C_CR2_ITEVTEN;
275  dmaStreamEnable(i2cp->dmatx);
276  break;
277  case I2C_EV8_2_MASTER_BYTE_TRANSMITTED:
278  /* Catches BTF event after the end of transmission.*/
279  if (dmaStreamGetTransactionSize(i2cp->dmarx) > 0) {
280  /* Starts "read after write" operation, LSB = 1 -> receive.*/
281  i2cp->addr |= 0x01;
282  dp->CR1 |= I2C_CR1_START | I2C_CR1_ACK;
283  return;
284  }
285  dp->CR2 &= ~I2C_CR2_ITEVTEN;
286  dp->CR1 |= I2C_CR1_STOP;
287  _i2c_wakeup_isr(i2cp);
288  break;
289  default:
290  break;
291  }
292  /* Clear ADDR flag. */
293  if (event & (I2C_SR1_ADDR | I2C_SR1_ADD10))
294  (void)dp->SR2;
295 }

297 /**
298  * @brief DMA RX end IRQ handler.
299  *
300  * @param[in] i2cp pointer to the @p I2CDriver object
301  * @param[in] flags pre-shifted content of the ISR register
302  *
303  * @notapi
304  */
305 static void i2c_lld_serve_rx_end_irq(I2CDriver *i2cp, uint32_t flags) {
306  I2C_TypeDef *dp = i2cp->i2c;
307 
308  /* DMA errors handling.*/
309 #if defined(STM32_I2C_DMA_ERROR_HOOK)
310  if ((flags & (STM32_DMA_ISR_TEIF | STM32_DMA_ISR_DMEIF)) != 0) {
311  STM32_I2C_DMA_ERROR_HOOK(i2cp);
312  }
313 #else
314  (void)flags;
315 #endif
316 
317  dmaStreamDisable(i2cp->dmarx);
318 
319  dp->CR2 &= ~I2C_CR2_LAST;
320  dp->CR1 &= ~I2C_CR1_ACK;
321  dp->CR1 |= I2C_CR1_STOP;
322  _i2c_wakeup_isr(i2cp);
323 }

326  * @brief DMA TX end IRQ handler.

328  * @param[in] i2cp pointer to the @p I2CDriver object

331  */
332 static void i2c_lld_serve_tx_end_irq(I2CDriver *i2cp, uint32_t flags) {
333  I2C_TypeDef *dp = i2cp->i2c;
334 
335  /* DMA errors handling.*/
336 #if defined(STM32_I2C_DMA_ERROR_HOOK)
337  if ((flags & (STM32_DMA_ISR_TEIF | STM32_DMA_ISR_DMEIF)) != 0) {
338  STM32_I2C_DMA_ERROR_HOOK(i2cp);
339  }
340 #else
341  (void)flags;
342 #endif
343 
344  dmaStreamDisable(i2cp->dmatx);
345  /* Enables interrupts to catch BTF event meaning transmission part complete.
346  Interrupt handler will decide to generate STOP or to begin receiving part
347  of R/W transaction itself.*/
348  dp->CR2 |= I2C_CR2_ITEVTEN;
349 }

351 /**
352  * @brief I2C error handler.
353  *
354  * @param[in] i2cp pointer to the @p I2CDriver object
355  * @param[in] sr content of the SR1 register to be decoded
356  *
357  * @notapi
358  */
359 static void i2c_lld_serve_error_interrupt(I2CDriver *i2cp, uint16_t sr) {
360 
361  /* Clears interrupt flags just to be safe.*/
362  dmaStreamDisable(i2cp->dmatx);
363  dmaStreamDisable(i2cp->dmarx);
364 
365  i2cp->errors = I2C_NO_ERROR;
366 
367  if (sr & I2C_SR1_BERR) /* Bus error. */
368  i2cp->errors |= I2C_BUS_ERROR;
369 
370  if (sr & I2C_SR1_ARLO) /* Arbitration lost. */
371  i2cp->errors |= I2C_ARBITRATION_LOST;
372 
373  if (sr & I2C_SR1_AF) { /* Acknowledge fail. */
374  i2cp->i2c->CR2 &= ~I2C_CR2_ITEVTEN;
375  i2cp->i2c->CR1 |= I2C_CR1_STOP; /* Setting stop bit. */
376  i2cp->errors |= I2C_ACK_FAILURE;
377  }
378 
379  if (sr & I2C_SR1_OVR) /* Overrun. */
380  i2cp->errors |= I2C_OVERRUN;
381 
382  if (sr & I2C_SR1_TIMEOUT) /* SMBus Timeout. */
383  i2cp->errors |= I2C_TIMEOUT;
384 
385  if (sr & I2C_SR1_PECERR) /* PEC error. */
386  i2cp->errors |= I2C_PEC_ERROR;
387 
388  if (sr & I2C_SR1_SMBALERT) /* SMBus alert. */
389  i2cp->errors |= I2C_SMB_ALERT;
390 
391  /* If some error has been identified then sends wakes the waiting thread.*/
392  if (i2cp->errors != I2C_NO_ERROR)
393  _i2c_wakeup_error_isr(i2cp);
394 }

530 /**
531  * @brief Configures and activates the I2C peripheral.
532  *
533  * @param[in] i2cp pointer to the @p I2CDriver object
534  *
535  * @notapi
536  */
537 void i2c_lld_start(I2CDriver *i2cp) {
538  I2C_TypeDef *dp = i2cp->i2c;
539 
540  /* If in stopped state then enables the I2C and DMA clocks.*/
541  if (i2cp->state == I2C_STOP) {
542 
543  i2cp->txdmamode = STM32_DMA_CR_PSIZE_BYTE | STM32_DMA_CR_MSIZE_BYTE |
544  STM32_DMA_CR_MINC | STM32_DMA_CR_DMEIE |
545  STM32_DMA_CR_TEIE | STM32_DMA_CR_TCIE |
546  STM32_DMA_CR_DIR_M2P;
547  i2cp->rxdmamode = STM32_DMA_CR_PSIZE_BYTE | STM32_DMA_CR_MSIZE_BYTE |
548  STM32_DMA_CR_MINC | STM32_DMA_CR_DMEIE |
549  STM32_DMA_CR_TEIE | STM32_DMA_CR_TCIE |
550  STM32_DMA_CR_DIR_P2M;
551 
552 #if STM32_I2C_USE_I2C1
553  if (&I2CD1 == i2cp) {
554  bool b;
555 
556  rccResetI2C1();
557  b = dmaStreamAllocate(i2cp->dmarx,
558  STM32_I2C_I2C1_IRQ_PRIORITY,
559  (stm32_dmaisr_t)i2c_lld_serve_rx_end_irq,
560  (void *)i2cp);
561  osalDbgAssert(!b, "stream already allocated");
562  b = dmaStreamAllocate(i2cp->dmatx,
563  STM32_I2C_I2C1_IRQ_PRIORITY,
564  (stm32_dmaisr_t)i2c_lld_serve_tx_end_irq,
565  (void *)i2cp);
566  osalDbgAssert(!b, "stream already allocated");
567  rccEnableI2C1(true);
568  nvicEnableVector(I2C1_EV_IRQn, STM32_I2C_I2C1_IRQ_PRIORITY);
569  nvicEnableVector(I2C1_ER_IRQn, STM32_I2C_I2C1_IRQ_PRIORITY);
570 
571  i2cp->rxdmamode |= STM32_DMA_CR_CHSEL(I2C1_RX_DMA_CHANNEL) |

691 /**
692  * @brief Receives data via the I2C bus as master.
693  * @details Number of receiving bytes must be more than 1 on STM32F1x. This is
694  * hardware restriction.
695  *
696  * @param[in] i2cp pointer to the @p I2CDriver object
697  * @param[in] addr slave device address
698  * @param[out] rxbuf pointer to the receive buffer
699  * @param[in] rxbytes number of bytes to be received
700  * @param[in] timeout the number of ticks before the operation timeouts,
701  * the following special values are allowed:
702  * - @a TIME_INFINITE no timeout.
703  * .
704  * @return The operation status.
705  * @retval MSG_OK if the function succeeded.
706  * @retval MSG_RESET if one or more I2C errors occurred, the errors can
707  * be retrieved using @p i2cGetErrors().
708  * @retval MSG_TIMEOUT if a timeout occurred before operation end. <b>After a
709  * timeout the driver must be stopped and restarted
710  * because the bus is in an uncertain state</b>.
711  *
712  * @notapi
713  */
714 msg_t i2c_lld_master_receive_timeout(I2CDriver *i2cp, i2caddr_t addr,
715  uint8_t *rxbuf, size_t rxbytes,
716  sysinterval_t timeout) {
717  I2C_TypeDef *dp = i2cp->i2c;
718  systime_t start, end;
719 
720 #if defined(STM32F1XX_I2C)
721  osalDbgCheck(rxbytes > 1);
722 #endif
723 
724  /* Resetting error flags for this transfer.*/
725  i2cp->errors = I2C_NO_ERROR;
726 
727  /* Initializes driver fields, LSB = 1 -> receive.*/
728  i2cp->addr = (addr << 1) | 0x01;
729 
730  /* Releases the lock from high level driver.*/
731  osalSysUnlock();
732 
733  /* RX DMA setup.*/
734  dmaStreamSetMode(i2cp->dmarx, i2cp->rxdmamode);
735  dmaStreamSetMemory0(i2cp->dmarx, rxbuf);
736  dmaStreamSetTransactionSize(i2cp->dmarx, rxbytes);
737 
738  /* Calculating the time window for the timeout on the busy bus condition.*/
739  start = osalOsGetSystemTimeX();
740  end = osalTimeAddX(start, OSAL_MS2I(STM32_I2C_BUSY_TIMEOUT));
741 
742  /* Waits until BUSY flag is reset or, alternatively, for a timeout
743  condition.*/
744  while (true) {
745  osalSysLock();
746 
747  /* If the bus is not busy then the operation can continue, note, the
748  loop is exited in the locked state.*/
749  if (!(dp->SR2 & I2C_SR2_BUSY) && !(dp->CR1 & I2C_CR1_STOP))
750  break;
751 
752  /* If the system time went outside the allowed window then a timeout
753  condition is returned.*/
754  if (!osalTimeIsInRangeX(osalOsGetSystemTimeX(), start, end))
755  return MSG_TIMEOUT;
756 
757  osalSysUnlock();
758  }
759 
760  /* Starts the operation.*/
761  dp->CR2 |= I2C_CR2_ITEVTEN;
762  dp->CR1 |= I2C_CR1_START | I2C_CR1_ACK;
763 
764  /* Waits for the operation completion or a timeout.*/
765  return osalThreadSuspendTimeoutS(&i2cp->thread, timeout);
766 }
