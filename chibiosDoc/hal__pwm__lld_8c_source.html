359 /**
360  * @brief Configures and activates the PWM peripheral.
361  * @note Starting a driver that is already in the @p PWM_READY state
362  * disables all the active channels.
363  *
364  * @param[in] pwmp pointer to a @p PWMDriver object
365  *
366  * @notapi
367  */
368 void pwm_lld_start(PWMDriver *pwmp) {
369  uint32_t psc;
370  uint32_t ccer;
371 
372  if (pwmp->state == PWM_STOP) {
373  /* Clock activation and timer reset.*/
374 #if STM32_PWM_USE_TIM1
375  if (&PWMD1 == pwmp) {
376  rccEnableTIM1(true);
377  rccResetTIM1();
378 #if !defined(STM32_TIM1_SUPPRESS_ISR)
379  nvicEnableVector(STM32_TIM1_UP_NUMBER, STM32_PWM_TIM1_IRQ_PRIORITY);
380  nvicEnableVector(STM32_TIM1_CC_NUMBER, STM32_PWM_TIM1_IRQ_PRIORITY);
381 #endif
382 #if defined(STM32_TIM1CLK)
383  pwmp->clock = STM32_TIM1CLK;
384 #else
385  pwmp->clock = STM32_TIMCLK2;
386 #endif
387  }
388 #endif
389 
390 #if STM32_PWM_USE_TIM2
391  if (&PWMD2 == pwmp) {
392  rccEnableTIM2(true);
393  rccResetTIM2();
394 #if !defined(STM32_TIM2_SUPPRESS_ISR)
395  nvicEnableVector(STM32_TIM2_NUMBER, STM32_PWM_TIM2_IRQ_PRIORITY);
396 #endif
397 #if defined(STM32_TIM2CLK)
398  pwmp->clock = STM32_TIM2CLK;
399 #else
400  pwmp->clock = STM32_TIMCLK1;
401 #endif
402  }
403 #endif
404 
405 #if STM32_PWM_USE_TIM3
406  if (&PWMD3 == pwmp) {
407  rccEnableTIM3(true);
408  rccResetTIM3();
409 #if !defined(STM32_TIM3_SUPPRESS_ISR)
410  nvicEnableVector(STM32_TIM3_NUMBER, STM32_PWM_TIM3_IRQ_PRIORITY);
411 #endif
412 #if defined(STM32_TIM3CLK)
413  pwmp->clock = STM32_TIM3CLK;
414 #else
415  pwmp->clock = STM32_TIMCLK1;
416 #endif
417  }
418 #endif
419 
420 #if STM32_PWM_USE_TIM4
421  if (&PWMD4 == pwmp) {
422  rccEnableTIM4(true);
423  rccResetTIM4();
424 #if !defined(STM32_TIM4_SUPPRESS_ISR)
425  nvicEnableVector(STM32_TIM4_NUMBER, STM32_PWM_TIM4_IRQ_PRIORITY);
426 #endif
427 #if defined(STM32_TIM4CLK)
428  pwmp->clock = STM32_TIM4CLK;
429 #else
430  pwmp->clock = STM32_TIMCLK1;
431 #endif
432  }
433 #endif
434 
435 #if STM32_PWM_USE_TIM5
436  if (&PWMD5 == pwmp) {
437  rccEnableTIM5(true);
438  rccResetTIM5();
439 #if !defined(STM32_TIM5_SUPPRESS_ISR)
440  nvicEnableVector(STM32_TIM5_NUMBER, STM32_PWM_TIM5_IRQ_PRIORITY);
441 #endif
442 #if defined(STM32_TIM5CLK)
443  pwmp->clock = STM32_TIM5CLK;
444 #else
445  pwmp->clock = STM32_TIMCLK1;
446 #endif
447  }
448 #endif
449 
450 #if STM32_PWM_USE_TIM8
451  if (&PWMD8 == pwmp) {
452  rccEnableTIM8(true);
453  rccResetTIM8();
454 #if !defined(STM32_TIM8_SUPPRESS_ISR)
455  nvicEnableVector(STM32_TIM8_UP_NUMBER, STM32_PWM_TIM8_IRQ_PRIORITY);
456  nvicEnableVector(STM32_TIM8_CC_NUMBER, STM32_PWM_TIM8_IRQ_PRIORITY);
457 #endif
458 #if defined(STM32_TIM8CLK)
459  pwmp->clock = STM32_TIM8CLK;
460 #else
461  pwmp->clock = STM32_TIMCLK2;
462 #endif
463  }
464 #endif
465 
466 #if STM32_PWM_USE_TIM9
467  if (&PWMD9 == pwmp) {
468  rccEnableTIM9(true);
469  rccResetTIM9();
470 #if !defined(STM32_TIM9_SUPPRESS_ISR)
471  nvicEnableVector(STM32_TIM9_NUMBER, STM32_PWM_TIM9_IRQ_PRIORITY);
472 #endif
473 #if defined(STM32_TIM9CLK)
474  pwmp->clock = STM32_TIM9CLK;
475 #else
476  pwmp->clock = STM32_TIMCLK2;
477 #endif
478  }
479 #endif
480 
481  /* All channels configured in PWM1 mode with preload enabled and will
482  stay that way until the driver is stopped.*/
483  pwmp->tim->CCMR1 = STM32_TIM_CCMR1_OC1M(6) | STM32_TIM_CCMR1_OC1PE |
484  STM32_TIM_CCMR1_OC2M(6) | STM32_TIM_CCMR1_OC2PE;
485  pwmp->tim->CCMR2 = STM32_TIM_CCMR2_OC3M(6) | STM32_TIM_CCMR2_OC3PE |
486  STM32_TIM_CCMR2_OC4M(6) | STM32_TIM_CCMR2_OC4PE;
487 #if STM32_TIM_MAX_CHANNELS > 4
488  pwmp->tim->CCMR3 = STM32_TIM_CCMR3_OC5M(6) | STM32_TIM_CCMR3_OC5PE |
489  STM32_TIM_CCMR3_OC6M(6) | STM32_TIM_CCMR3_OC6PE;
490 #endif
491  }
492  else {
493  /* Driver re-configuration scenario, it must be stopped first.*/
494  pwmp->tim->CR1 = 0; /* Timer disabled. */
495  pwmp->tim->CCR[0] = 0; /* Comparator 1 disabled. */
496  pwmp->tim->CCR[1] = 0; /* Comparator 2 disabled. */
497  pwmp->tim->CCR[2] = 0; /* Comparator 3 disabled. */
498  pwmp->tim->CCR[3] = 0; /* Comparator 4 disabled. */
499 #if STM32_TIM_MAX_CHANNELS > 4
500  if (pwmp->channels > 4) {
501  pwmp->tim->CCXR[0] = 0; /* Comparator 5 disabled. */
502  pwmp->tim->CCXR[1] = 0; /* Comparator 6 disabled. */
503  }
504 #endif
505  pwmp->tim->CNT = 0; /* Counter reset to zero. */
506  }
507 
508  /* Timer configuration.*/
509  psc = (pwmp->clock / pwmp->config->frequency) - 1;
510  osalDbgAssert((psc <= 0xFFFF) &&
511  ((psc + 1) * pwmp->config->frequency) == pwmp->clock,
512  "invalid frequency");
513  pwmp->tim->PSC = psc;
514  pwmp->tim->ARR = pwmp->period - 1;
515  pwmp->tim->CR2 = pwmp->config->cr2;
516 
517  /* Output enables and polarities setup.*/
518  ccer = 0;
519  switch (pwmp->config->channels[0].mode & PWM_OUTPUT_MASK) {
520  case PWM_OUTPUT_ACTIVE_LOW:
521  ccer |= STM32_TIM_CCER_CC1P;
522  /* Falls through.*/
523  case PWM_OUTPUT_ACTIVE_HIGH:
524  ccer |= STM32_TIM_CCER_CC1E;
525  /* Falls through.*/
526  default:
527  ;
528  }
529  switch (pwmp->config->channels[1].mode & PWM_OUTPUT_MASK) {
530  case PWM_OUTPUT_ACTIVE_LOW:
531  ccer |= STM32_TIM_CCER_CC2P;
532  /* Falls through.*/
533  case PWM_OUTPUT_ACTIVE_HIGH:
534  ccer |= STM32_TIM_CCER_CC2E;
535  /* Falls through.*/
536  default:
537  ;
538  }
539  switch (pwmp->config->channels[2].mode & PWM_OUTPUT_MASK) {
540  case PWM_OUTPUT_ACTIVE_LOW:
541  ccer |= STM32_TIM_CCER_CC3P;
542  /* Falls through.*/
543  case PWM_OUTPUT_ACTIVE_HIGH:
544  ccer |= STM32_TIM_CCER_CC3E;
545  /* Falls through.*/
546  default:
547  ;
548  }
549  switch (pwmp->config->channels[3].mode & PWM_OUTPUT_MASK) {
550  case PWM_OUTPUT_ACTIVE_LOW:
551  ccer |= STM32_TIM_CCER_CC4P;
552  /* Falls through.*/
553  case PWM_OUTPUT_ACTIVE_HIGH:
554  ccer |= STM32_TIM_CCER_CC4E;
555  /* Falls through.*/
556  default:
557  ;
558  }
559 #if STM32_PWM_USE_ADVANCED
560 #if STM32_PWM_USE_TIM1 && !STM32_PWM_USE_TIM8
561  if (&PWMD1 == pwmp) {
562 #endif
563 #if !STM32_PWM_USE_TIM1 && STM32_PWM_USE_TIM8
564  if (&PWMD8 == pwmp) {
565 #endif
566 #if STM32_PWM_USE_TIM1 && STM32_PWM_USE_TIM8
567  if ((&PWMD1 == pwmp) || (&PWMD8 == pwmp)) {
568 #endif
569  switch (pwmp->config->channels[0].mode & PWM_COMPLEMENTARY_OUTPUT_MASK) {
570  case PWM_COMPLEMENTARY_OUTPUT_ACTIVE_LOW:
571  ccer |= STM32_TIM_CCER_CC1NP;
572  /* Falls through.*/
573  case PWM_COMPLEMENTARY_OUTPUT_ACTIVE_HIGH:
574  ccer |= STM32_TIM_CCER_CC1NE;
575  /* Falls through.*/
576  default:
577  ;
578  }
579  switch (pwmp->config->channels[1].mode & PWM_COMPLEMENTARY_OUTPUT_MASK) {
580  case PWM_COMPLEMENTARY_OUTPUT_ACTIVE_LOW:
581  ccer |= STM32_TIM_CCER_CC2NP;
582  /* Falls through.*/
583  case PWM_COMPLEMENTARY_OUTPUT_ACTIVE_HIGH:
584  ccer |= STM32_TIM_CCER_CC2NE;
585  /* Falls through.*/
586  default:
587  ;
588  }
589  switch (pwmp->config->channels[2].mode & PWM_COMPLEMENTARY_OUTPUT_MASK) {
590  case PWM_COMPLEMENTARY_OUTPUT_ACTIVE_LOW:
591  ccer |= STM32_TIM_CCER_CC3NP;
592  /* Falls through.*/
593  case PWM_COMPLEMENTARY_OUTPUT_ACTIVE_HIGH:
594  ccer |= STM32_TIM_CCER_CC3NE;
595  /* Falls through.*/
596  default:
597  ;
598  }
599  }
600 #endif /* STM32_PWM_USE_ADVANCED*/
601 
602  pwmp->tim->CCER = ccer;
603  pwmp->tim->EGR = STM32_TIM_EGR_UG; /* Update event. */
604  pwmp->tim->SR = 0; /* Clear pending IRQs. */
605  pwmp->tim->DIER = pwmp->config->dier & /* DMA-related DIER settings. */
606  ~STM32_TIM_DIER_IRQ_MASK;
607 #if STM32_PWM_USE_TIM1 || STM32_PWM_USE_TIM8
608 #if STM32_PWM_USE_ADVANCED
609  pwmp->tim->BDTR = pwmp->config->bdtr | STM32_TIM_BDTR_MOE;
610 #else
611  pwmp->tim->BDTR = STM32_TIM_BDTR_MOE;
612 #endif
613 #endif
614  /* Timer configured and started.*/
615  pwmp->tim->CR1 = STM32_TIM_CR1_ARPE | STM32_TIM_CR1_URS |
616  STM32_TIM_CR1_CEN;
617 }

619 /**
620  * @brief Deactivates the PWM peripheral.
621  *
622  * @param[in] pwmp pointer to a @p PWMDriver object
623  *
624  * @notapi
625  */
626 void pwm_lld_stop(PWMDriver *pwmp) {
627 
628  /* If in ready state then disables the PWM clock.*/
629  if (pwmp->state == PWM_READY) {
630  pwmp->tim->CR1 = 0; /* Timer disabled. */
631  pwmp->tim->DIER = 0; /* All IRQs disabled. */
632  pwmp->tim->SR = 0; /* Clear eventual pending IRQs. */
633 #if STM32_PWM_USE_TIM1 || STM32_PWM_USE_TIM8
634  pwmp->tim->BDTR = 0;
635 #endif
636 
637 #if STM32_PWM_USE_TIM1
638  if (&PWMD1 == pwmp) {

839  * @brief Common TIM2...TIM5,TIM9 IRQ handler.
840  * @note It is assumed that the various sources are only activated if the
841  * associated callback pointer is not equal to @p NULL in order to not
842  * perform an extra check in a potentially critical interrupt handler.
843  *
844  * @param[in] pwmp pointer to a @p PWMDriver object
845  *
846  * @notapi
847  */
848 void pwm_lld_serve_interrupt(PWMDriver *pwmp) {
849  uint32_t sr;
850 
851  sr = pwmp->tim->SR;
852  sr &= pwmp->tim->DIER & STM32_TIM_DIER_IRQ_MASK;
853  pwmp->tim->SR = ~sr;
854  if (((sr & STM32_TIM_SR_CC1IF) != 0) &&
855  (pwmp->config->channels[0].callback != NULL))
856  pwmp->config->channels[0].callback(pwmp);
857  if (((sr & STM32_TIM_SR_CC2IF) != 0) &&
858  (pwmp->config->channels[1].callback != NULL))
859  pwmp->config->channels[1].callback(pwmp);
860  if (((sr & STM32_TIM_SR_CC3IF) != 0) &&
861  (pwmp->config->channels[2].callback != NULL))
862  pwmp->config->channels[2].callback(pwmp);
863  if (((sr & STM32_TIM_SR_CC4IF) != 0) &&
864  (pwmp->config->channels[3].callback != NULL))
865  pwmp->config->channels[3].callback(pwmp);
866  if (((sr & STM32_TIM_SR_UIF) != 0) && (pwmp->config->callback != NULL))
867  pwmp->config->callback(pwmp);
868 }
