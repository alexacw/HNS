94 static msg_t _ctl(void *ip, unsigned int operation, void *arg) {
95  SerialDriver *sdp = (SerialDriver *)ip;
96 
97  osalDbgCheck(sdp != NULL);
98 
99  switch (operation) {
100  case CHN_CTL_NOP:
101  osalDbgCheck(arg == NULL);
102  break;
103  case CHN_CTL_INVALID:
104  osalDbgAssert(false, "invalid CTL operation");
105  break;
106  default:
107 #if defined(SD_LLD_IMPLEMENTS_CTL)
108  /* Delegating to the LLD if supported.*/
109  return sd_lld_control(sdp, operation, arg);
110 #else
111  break;
112 #endif
113  }
114  return MSG_OK;
115 }

174 /**
175  * @brief Configures and starts the driver.
176  *
177  * @param[in] sdp pointer to a @p SerialDriver object
178  * @param[in] config the architecture-dependent serial driver configuration.
179  * If this parameter is set to @p NULL then a default
180  * configuration is used.

182  * @api

184 void sdStart(SerialDriver *sdp, const SerialConfig *config) {
185 
186  osalDbgCheck(sdp != NULL);
187 
188  osalSysLock();
189  osalDbgAssert((sdp->state == SD_STOP) || (sdp->state == SD_READY),
190  "invalid state");
191  sd_lld_start(sdp, config);
192  sdp->state = SD_READY;
193  osalSysUnlock();
194 }

196 /**
197  * @brief Stops the driver.
198  * @details Any thread waiting on the driver's queues will be awakened with
199  * the message @p MSG_RESET.
200  *
201  * @param[in] sdp pointer to a @p SerialDriver object
202  *
203  * @api
204  */
205 void sdStop(SerialDriver *sdp) {
206 
207  osalDbgCheck(sdp != NULL);
208 
209  osalSysLock();
210 
211  osalDbgAssert((sdp->state == SD_STOP) || (sdp->state == SD_READY),
212  "invalid state");
213 
214  sd_lld_stop(sdp);
215  sdp->state = SD_STOP;
216  oqResetI(&sdp->oqueue);
217  iqResetI(&sdp->iqueue);
218  osalOsRescheduleS();
219 
220  osalSysUnlock();
221 }

223 /**
224  * @brief Handles incoming data.
225  * @details This function must be called from the input interrupt service
226  * routine in order to enqueue incoming data and generate the
227  * related events.
228  * @note The incoming data event is only generated when the input queue
229  * becomes non-empty.
230  * @note In order to gain some performance it is suggested to not use
231  * this function directly but copy this code directly into the
232  * interrupt service routine.
233  *
234  * @param[in] sdp pointer to a @p SerialDriver structure
235  * @param[in] b the byte to be written in the driver's Input Queue
236  *
237  * @iclass
238  */
239 void sdIncomingDataI(SerialDriver *sdp, uint8_t b) {
240 
241  osalDbgCheckClassI();
242  osalDbgCheck(sdp != NULL);
243 
244  if (iqIsEmptyI(&sdp->iqueue))
245  chnAddFlagsI(sdp, CHN_INPUT_AVAILABLE);
246  if (iqPutI(&sdp->iqueue, b) < MSG_OK)
247  chnAddFlagsI(sdp, SD_QUEUE_FULL_ERROR);
248 }

250 /**
251  * @brief Handles outgoing data.
252  * @details Must be called from the output interrupt service routine in order
253  * to get the next byte to be transmitted.
254  * @note In order to gain some performance it is suggested to not use
255  * this function directly but copy this code directly into the
256  * interrupt service routine.
257  *
258  * @param[in] sdp pointer to a @p SerialDriver structure
259  * @return The byte value read from the driver's output queue.
260  * @retval MSG_TIMEOUT if the queue is empty (the lower driver usually
261  * disables the interrupt source when this happens).
262  *
263  * @iclass
264  */
265 msg_t sdRequestDataI(SerialDriver *sdp) {
266  msg_t b;
267 
268  osalDbgCheckClassI();
269  osalDbgCheck(sdp != NULL);
270 
271  b = oqGetI(&sdp->oqueue);
272  if (b < MSG_OK)
273  chnAddFlagsI(sdp, CHN_OUTPUT_EMPTY);
274  return b;
275 }
